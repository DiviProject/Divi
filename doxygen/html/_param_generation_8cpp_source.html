40 void
41 CalculateParams(ZerocoinParams &params, CBigNum N, string aux, uint32_t securityLevel)
42 {
43  params.initialized = false;
44  params.accumulatorParams.initialized = false;
45 
46  // Verify that |N| is > 1023 bits.
47  uint32_t NLen = N.bitSize();
48  if (NLen < 1023) {
49  throw std::runtime_error("Modulus must be at least 1023 bits");
50  }
51 
52  // Verify that "securityLevel" is at least 80 bits (minimum).
53  if (securityLevel < 80) {
54  throw std::runtime_error("Security level must be at least 80 bits.");
55  }
56 
57  // Set the accumulator modulus to "N".
58  params.accumulatorParams.accumulatorModulus = N;
59 
60  // Calculate the required size of the field "F_p" into which
61  // we're embedding the coin commitment group. This may throw an
62  // exception if the securityLevel is too large to be supported
63  // by the current modulus.
64  uint32_t pLen = 0;
65  uint32_t qLen = 0;
66  calculateGroupParamLengths(NLen - 2, securityLevel, &pLen, &qLen);
67 
68  // Calculate candidate parameters ("p", "q") for the coin commitment group
69  // using a deterministic process based on "N", the "aux" string, and
70  // the dedicated string "COMMITMENTGROUP".
71  params.coinCommitmentGroup = deriveIntegerGroupParams(calculateSeed(N, aux, securityLevel, STRING_COMMIT_GROUP),
72  pLen, qLen);
73 
74  // Next, we derive parameters for a second Accumulated Value commitment group.
75  // This is a Schnorr group with the specific property that the order of the group
76  // must be exactly equal to "q" from the commitment group. We set
77  // the modulus of the new group equal to "2q+1" and test to see if this is prime.
78  params.serialNumberSoKCommitmentGroup = deriveIntegerGroupFromOrder(params.coinCommitmentGroup.modulus);
79 
80  // Calculate the parameters for the internal commitment
81  // using the same process.
82  params.accumulatorParams.accumulatorPoKCommitmentGroup = deriveIntegerGroupParams(calculateSeed(N, aux, securityLevel, STRING_AIC_GROUP),
83  qLen + 300, qLen + 1);
84 
85  // Calculate the parameters for the accumulator QRN commitment generators. This isn't really
86  // a whole group, just a pair of random generators in QR_N.
87  uint32_t resultCtr;
88  params.accumulatorParams.accumulatorQRNCommitmentGroup.g = generateIntegerFromSeed(NLen - 1,
89  calculateSeed(N, aux, securityLevel, STRING_QRNCOMMIT_GROUPG),
90  &resultCtr).pow_mod(CBigNum(2),N);
91  params.accumulatorParams.accumulatorQRNCommitmentGroup.h = generateIntegerFromSeed(NLen - 1,
92  calculateSeed(N, aux, securityLevel, STRING_QRNCOMMIT_GROUPH),
93  &resultCtr).pow_mod(CBigNum(2), N);
94 
95  // Calculate the accumulator base, which we calculate as "u = C**2 mod N"
96  // where C is an arbitrary value. In the unlikely case that "u = 1" we increment
97  // "C" and repeat.
98  CBigNum constant(ACCUMULATOR_BASE_CONSTANT);
99  params.accumulatorParams.accumulatorBase = CBigNum(1);
100  for (uint32_t count = 0; count < MAX_ACCUMGEN_ATTEMPTS && params.accumulatorParams.accumulatorBase.isOne(); count++) {
101  params.accumulatorParams.accumulatorBase = constant.pow_mod(CBigNum(2), params.accumulatorParams.accumulatorModulus);
102  }
103 
104  // Compute the accumulator range. The upper range is the largest possible coin commitment value.
105  // The lower range is sqrt(upper range) + 1. Since OpenSSL doesn't have
106  // a square root function we use a slightly higher approximation.
107  params.accumulatorParams.maxCoinValue = params.coinCommitmentGroup.modulus;
108  params.accumulatorParams.minCoinValue = CBigNum(2).pow((params.coinCommitmentGroup.modulus.bitSize() / 2) + 3);
109 
110  // If all went well, mark params as successfully initialized.
111  params.accumulatorParams.initialized = true;
112 
113  // If all went well, mark params as successfully initialized.
114  params.initialized = true;
115 }

246 deriveIntegerGroupParams(uint256 seed, uint32_t pLen, uint32_t qLen)
247 {
248  IntegerGroupParams result;
249  CBigNum p;
250  CBigNum q;
251  uint256 pSeed, qSeed;
252 
253  // Calculate "p" and "q" and "domain_parameter_seed" from the
254  // "seed" buffer above, using the procedure described in NIST
255  // FIPS 186-3, Appendix A.1.2.
256  calculateGroupModulusAndOrder(seed, pLen, qLen, &(result.modulus),
257  &(result.groupOrder), &pSeed, &qSeed);
258 
259  // Calculate the generators "g", "h" using the process described in
260  // NIST FIPS 186-3, Appendix A.2.3. This algorithm takes ("p", "q",
261  // "domain_parameter_seed", "index"). We use "index" value 1
262  // to generate "g" and "index" value 2 to generate "h".
263  result.g = calculateGroupGenerator(seed, pSeed, qSeed, result.modulus, result.groupOrder, 1);
264  result.h = calculateGroupGenerator(seed, pSeed, qSeed, result.modulus, result.groupOrder, 2);
265 
266  // Perform some basic tests to make sure we have good parameters
267  if ((uint32_t)(result.modulus.bitSize()) < pLen || // modulus is pLen bits long
268  (uint32_t)(result.groupOrder.bitSize()) < qLen || // order is qLen bits long
269  !(result.modulus.isPrime()) || // modulus is prime
270  !(result.groupOrder.isPrime()) || // order is prime
271  !((result.g.pow_mod(result.groupOrder, result.modulus)).isOne()) || // g^order mod modulus = 1
272  !((result.h.pow_mod(result.groupOrder, result.modulus)).isOne()) || // h^order mod modulus = 1
273  ((result.g.pow_mod(CBigNum(100), result.modulus)).isOne()) || // g^100 mod modulus != 1
274  ((result.h.pow_mod(CBigNum(100), result.modulus)).isOne()) || // h^100 mod modulus != 1
275  result.g == result.h || // g != h
276  result.g.isOne()) { // g != 1
277  // If any of the above tests fail, throw an exception
278  throw std::runtime_error("Group parameters are not valid");
279  }
280 
281  return result;
282 }

292 deriveIntegerGroupFromOrder(CBigNum &groupOrder)
293 {
294  IntegerGroupParams result;
295 
296  // Set the order to "groupOrder"
297  result.groupOrder = groupOrder;
298 
299  // Try possible values for "modulus" of the form "groupOrder * 2 * i" where
300  // "p" is prime and i is a counter starting at 1.
301  for (uint32_t i = 1; i < NUM_SCHNORRGEN_ATTEMPTS; i++) {
302  // Set modulus equal to "groupOrder * 2 * i"
303  result.modulus = (result.groupOrder * CBigNum(i*2)) + CBigNum(1);
304 
305  // Test the result for primality
306  // TODO: This is a probabilistic routine and thus not the right choice
307  if (result.modulus.isPrime(256)) {
308 
309  // Success.
310  //
311  // Calculate the generators "g", "h" using the process described in
312  // NIST FIPS 186-3, Appendix A.2.3. This algorithm takes ("p", "q",
313  // "domain_parameter_seed", "index"). We use "index" value 1
314  // to generate "g" and "index" value 2 to generate "h".
315  uint256 seed = calculateSeed(groupOrder, "", 128, "");
316  uint256 pSeed = calculateHash(seed);
317  uint256 qSeed = calculateHash(pSeed);
318  result.g = calculateGroupGenerator(seed, pSeed, qSeed, result.modulus, result.groupOrder, 1);
319  result.h = calculateGroupGenerator(seed, pSeed, qSeed, result.modulus, result.groupOrder, 2);
320 
321  // Perform some basic tests to make sure we have good parameters
322  if (!(result.modulus.isPrime()) || // modulus is prime
323  !(result.groupOrder.isPrime()) || // order is prime
324  !((result.g.pow_mod(result.groupOrder, result.modulus)).isOne()) || // g^order mod modulus = 1
325  !((result.h.pow_mod(result.groupOrder, result.modulus)).isOne()) || // h^order mod modulus = 1
326  ((result.g.pow_mod(CBigNum(100), result.modulus)).isOne()) || // g^100 mod modulus != 1
327  ((result.h.pow_mod(CBigNum(100), result.modulus)).isOne()) || // h^100 mod modulus != 1
328  result.g == result.h || // g != h
329  result.g.isOne()) { // g != 1
330  // If any of the above tests fail, throw an exception
331  throw std::runtime_error("Group parameters are not valid");
332  }
333 
334  return result;
335  }
336  }
337 
338  // If we reached this point group generation has failed. Throw an exception.
339  throw std::runtime_error("Too many attempts to generate Schnorr group.");
340 }

355 void
356 calculateGroupModulusAndOrder(uint256 seed, uint32_t pLen, uint32_t qLen,
357  CBigNum *resultModulus, CBigNum *resultGroupOrder,
358  uint256 *resultPseed, uint256 *resultQseed)
359 {
360  // Verify that the seed length is >= qLen
361  if (qLen > (sizeof(seed)) * 8) {
362  // TODO: The use of 256-bit seeds limits us to 256-bit group orders. We should probably change this.
363  // throw std::runtime_error("Seed is too short to support the required security level.");
364  }
365 
366 #ifdef ZEROCOIN_DEBUG
367  cout << "calculateGroupModulusAndOrder: pLen = " << pLen << endl;
368 #endif
369 
370  // Generate a random prime for the group order.
371  // This may throw an exception, which we'll pass upwards.
372  // Result is the value "resultGroupOrder", "qseed" and "qgen_counter".
373  uint256 qseed;
374  uint32_t qgen_counter;
375  *resultGroupOrder = generateRandomPrime(qLen, seed, &qseed, &qgen_counter);
376 
377  // Using pLen / 2 + 1 as the length and qseed as the input_seed, use the random prime
378  // routine to obtain p0 , pseed, and pgen_counter. We pass exceptions upward.
379  uint32_t p0len = ceil((pLen / 2.0) + 1);
380  uint256 pseed;
381  uint32_t pgen_counter;
382  CBigNum p0 = generateRandomPrime(p0len, qseed, &pseed, &pgen_counter);
383 
384  // Set x = 0, old_counter = pgen_counter
385  uint32_t old_counter = pgen_counter;
386 
387  // Generate a random integer "x" of pLen bits
388  uint32_t iterations;
389  CBigNum x = generateIntegerFromSeed(pLen, pseed, &iterations);
390  pseed += (iterations + 1);
391 
392  // Set x = 2^{pLen1} + (x mod 2^{pLen1}).
393  CBigNum powerOfTwo = CBigNum(2).pow(pLen-1);
394  x = powerOfTwo + (x % powerOfTwo);
395 
396  // t = x / (2 * resultGroupOrder * p0).
397  // TODO: we don't have a ceiling function
398  CBigNum t = x / (CBigNum(2) * (*resultGroupOrder) * p0);
399 
400  // Now loop until we find a valid prime "p" or we fail due to
401  // pgen_counter exceeding ((4*pLen) + old_counter).
402  for ( ; pgen_counter <= ((4*pLen) + old_counter) ; pgen_counter++) {
403  // If (2 * t * resultGroupOrder * p0 + 1) > 2^{pLen}, then
404  // t = 2^{pLen1} / (2 * resultGroupOrder * p0).
405  powerOfTwo = CBigNum(2).pow(pLen);
406  CBigNum prod = (CBigNum(2) * t * (*resultGroupOrder) * p0) + CBigNum(1);
407  if (prod > powerOfTwo) {
408  // TODO: implement a ceil function
409  t = CBigNum(2).pow(pLen-1) / (CBigNum(2) * (*resultGroupOrder) * p0);
410  }
411 
412  // Compute a candidate prime resultModulus = 2tqp0 + 1.
413  *resultModulus = (CBigNum(2) * t * (*resultGroupOrder) * p0) + CBigNum(1);
414 
415  // Verify that resultModulus is prime. First generate a pseudorandom integer "a".
416  CBigNum a = generateIntegerFromSeed(pLen, pseed, &iterations);
417  pseed += iterations + 1;
418 
419  // Set a = 2 + (a mod (resultModulus3)).
420  a = CBigNum(2) + (a % ((*resultModulus) - CBigNum(3)));
421 
422  // Set z = a^{2 * t * resultGroupOrder} mod resultModulus
423  CBigNum z = a.pow_mod(CBigNum(2) * t * (*resultGroupOrder), (*resultModulus));
424 
425  // If GCD(z1, resultModulus) == 1 AND (z^{p0} mod resultModulus == 1)
426  // then we have found our result. Return.
427  if ((resultModulus->gcd(z - CBigNum(1))).isOne() &&
428  (z.pow_mod(p0, (*resultModulus))).isOne()) {
429  // Success! Return the seeds and primes.
430  *resultPseed = pseed;
431  *resultQseed = qseed;
432  return;
433  }
434 
435  // This prime did not work out. Increment "t" and try again.
436  t = t + CBigNum(1);
437  } // loop continues until pgen_counter exceeds a limit
438 
439  // We reach this point only if we exceeded our maximum iteration count.
440  // Throw an exception.
441  throw std::runtime_error("Unable to generate a prime modulus for the group");
442 }

458 calculateGroupGenerator(uint256 seed, uint256 pSeed, uint256 qSeed, CBigNum modulus, CBigNum groupOrder, uint32_t index)
459 {
460  CBigNum result;
461 
462  // Verify that 0 <= index < 256
463  if (index > 255) {
464  throw std::runtime_error("Invalid index for group generation");
465  }
466 
467  // Compute e = (modulus - 1) / groupOrder
468  CBigNum e = (modulus - CBigNum(1)) / groupOrder;
469 
470  // Loop until we find a generator
471  for (uint32_t count = 1; count < MAX_GENERATOR_ATTEMPTS; count++) {
472  // hash = Hash(seed || pSeed || qSeed || ggen || index || count
473  uint256 hash = calculateGeneratorSeed(seed, pSeed, qSeed, "ggen", index, count);
474  CBigNum W(hash);
475 
476  // Compute result = W^e mod p
477  result = W.pow_mod(e, modulus);
478 
479  // If result > 1, we have a generator
480  if (result > 1) {
481  return result;
482  }
483  }
484 
485  // We only get here if we failed to find a generator
486  throw std::runtime_error("Unable to find a generator, too many attempts");

501 CBigNum
502 generateRandomPrime(uint32_t primeBitLen, uint256 in_seed, uint256 *out_seed,
503  uint32_t *prime_gen_counter)
504 {
505  // Verify that primeBitLen is not too small
506  if (primeBitLen < 2) {
507  throw std::runtime_error("Prime length is too short");
508  }
509 
510  // If primeBitLen < 33 bits, perform the base case.
511  if (primeBitLen < 33) {
512  CBigNum result(0);
513 
514  // Set prime_seed = in_seed, prime_gen_counter = 0.
515  uint256 prime_seed = in_seed;
516  (*prime_gen_counter) = 0;
517 
518  // Loop up to "4 * primeBitLen" iterations.
519  while ((*prime_gen_counter) < (4 * primeBitLen)) {
520 
521  // Generate a pseudorandom integer "c" of length primeBitLength bits
522  uint32_t iteration_count;
523  CBigNum c = generateIntegerFromSeed(primeBitLen, prime_seed, &iteration_count);
524 #ifdef ZEROCOIN_DEBUG
525  cout << "generateRandomPrime: primeBitLen = " << primeBitLen << endl;
526  cout << "Generated c = " << c << endl;
527 #endif
528 
529  prime_seed += (iteration_count + 1);
530  (*prime_gen_counter)++;
531 
532  // Set "intc" to be the least odd integer >= "c" we just generated
533  uint32_t intc = c.getulong();
534  intc = (2 * floor(intc / 2.0)) + 1;
535 #ifdef ZEROCOIN_DEBUG
536  cout << "Should be odd. c = " << intc << endl;
537  cout << "The big num is: c = " << c << endl;
538 #endif
539 
540  // Perform trial division on this (relatively small) integer to determine if "intc"
541  // is prime. If so, return success.
542  if (primalityTestByTrialDivision(intc)) {
543  // Return "intc" converted back into a CBigNum and "prime_seed". We also updated
544  // the variable "prime_gen_counter" in previous statements.
545  result = intc;
546  *out_seed = prime_seed;
547 
548  // Success
549  return result;
550  }
551  } // while()
552 
553  // If we reached this point there was an error finding a candidate prime
554  // so throw an exception.
555  throw std::runtime_error("Unable to find prime in Shawe-Taylor algorithm");
556 
557  // END OF BASE CASE
558  }
559  // If primeBitLen >= 33 bits, perform the recursive case.
560  else {
561  // Recurse to find a new random prime of roughly half the size
562  uint32_t newLength = ceil((double)primeBitLen / 2.0) + 1;
563  CBigNum c0 = generateRandomPrime(newLength, in_seed, out_seed, prime_gen_counter);
564 
565  // Generate a random integer "x" of primeBitLen bits using the output
566  // of the previous call.
567  uint32_t numIterations;
568  CBigNum x = generateIntegerFromSeed(primeBitLen, *out_seed, &numIterations);
569  (*out_seed) += numIterations + 1;
570 
571  // Compute "t" = x / (2 * c0
572  // TODO no Ceiling call
573  CBigNum t = x / (CBigNum(2) * c0);
574 
575  // Repeat the following procedure until we find a prime (or time out)
576  for (uint32_t testNum = 0; testNum < MAX_PRIMEGEN_ATTEMPTS; testNum++) {
577 
578  // If ((2 * t * c0) + 1 > 2^{primeBitLen}),
579  // then t = 2^{primeBitLen} 1 / (2 * c0).
580  if ((CBigNum(2) * t * c0) > (CBigNum(2).pow(CBigNum(primeBitLen)))) {
581  t = ((CBigNum(2).pow(CBigNum(primeBitLen))) - CBigNum(1)) / (CBigNum(2) * c0);
582  }
583 
584  // Set c = (2 * t * c0) + 1
585  CBigNum c = (CBigNum(2) * t * c0) + CBigNum(1);
586 
587  // Increment prime_gen_counter
588  (*prime_gen_counter)++;
589 
590  // Test "c" for primality as follows:
591  // 1. First pick an integer "a" in between 2 and (c - 2)
592  CBigNum a = generateIntegerFromSeed(c.bitSize(), (*out_seed), &numIterations);
593  a = CBigNum(2) + (a % (c - CBigNum(3)));
594  (*out_seed) += (numIterations + 1);
595 
596  // 2. Compute "z" = a^{2*t} mod c
597  CBigNum z = a.pow_mod(CBigNum(2) * t, c);
598 
599  // 3. Check if "c" is prime.
600  // Specifically, verify that gcd((z-1), c) == 1 AND (z^c0 mod c) == 1
601  // If so we return "c" as our result.
602  if (c.gcd(z - CBigNum(1)).isOne() && z.pow_mod(c0, c).isOne()) {
603  // Return "c", out_seed and prime_gen_counter
604  // (the latter two of which were already updated)
605  return c;
606  }
607 
608  // 4. If the test did not succeed, increment "t" and loop
609  t = t + CBigNum(1);
610  } // end of test loop
611  }
612 
613  // We only reach this point if the test loop has iterated MAX_PRIMEGEN_ATTEMPTS
614  // and failed to identify a valid prime. Throw an exception.
615  throw std::runtime_error("Unable to generate random prime (too many tests)");
616 }
