202  explicit CScriptNum(const std::vector<unsigned char>& vch, bool fRequireMinimal)
203  {
204  if (vch.size() > nMaxNumSize) {
205  throw scriptnum_error("script number overflow");
206  }
207  if (fRequireMinimal && vch.size() > 0) {
208  // Check that the number is encoded with the minimum possible
209  // number of bytes.
210  //
211  // If the most-significant-byte - excluding the sign bit - is zero
212  // then we're not minimal. Note how this test also rejects the
213  // negative-zero encoding, 0x80.
214  if ((vch.back() & 0x7f) == 0) {
215  // One exception: if there's more than one byte and the most
216  // significant bit of the second-most-significant-byte is set
217  // it would conflict with the sign bit. An example of this case
218  // is +-255, which encode to 0xff00 and 0xff80 respectively.
219  // (big-endian).
220  if (vch.size() <= 1 || (vch[vch.size() - 2] & 0x80) == 0) {
221  throw scriptnum_error("non-minimally encoded script number");
222  }
223  }
224  }
225  m_value = set_vch(vch);
226  }

292  static std::vector<unsigned char> serialize(const int64_t& value)
293  {
294  if(value == 0)
295  return std::vector<unsigned char>();
296 
297  std::vector<unsigned char> result;
298  const bool neg = value < 0;
299  uint64_t absvalue = neg ? -value : value;
300 
301  while(absvalue)
302  {
303  result.push_back(absvalue & 0xff);
304  absvalue >>= 8;
305  }
306 
307 // - If the most significant byte is >= 0x80 and the value is positive, push a
308 // new zero-byte to make the significant byte < 0x80 again.
309 
310 // - If the most significant byte is >= 0x80 and the value is negative, push a
311 // new 0x80 byte that will be popped off when converting to an integral.
312 
313 // - If the most significant byte is < 0x80 and the value is negative, add
314 // 0x80 to it, since it will be subtracted and interpreted as a negative when
315 // converting to an integral.
316 
317  if (result.back() & 0x80)
318  result.push_back(neg ? 0x80 : 0);
319  else if (neg)
320  result.back() |= 0x80;
321 
322  return result;
323  }

327 private:
328  static int64_t set_vch(const std::vector<unsigned char>& vch)
329  {
330  if (vch.empty())
331  return 0;
332 
333  int64_t result = 0;
334  for (size_t i = 0; i != vch.size(); ++i)
335  result |= static_cast<int64_t>(vch[i]) << 8*i;
336 
337  // If the input vector's most significant byte is 0x80, remove it from
338  // the result's msb and return a negative.
339  if (vch.back() & 0x80)
340  return -((int64_t)(result & ~(0x80ULL << (8 * (vch.size() - 1)))));
341 
342  return result;
343  }

479  bool GetOp2(const_iterator& pc, opcodetype& opcodeRet, std::vector<unsigned char>* pvchRet) const
480  {
481  opcodeRet = OP_INVALIDOPCODE;
482  if (pvchRet)
483  pvchRet->clear();
484  if (pc >= end())
485  return false;
486 
487  // Read instruction
488  if (end() - pc < 1)
489  return false;
490  unsigned int opcode = *pc++;
491 
492  // Immediate operand
493  if (opcode <= OP_PUSHDATA4)
494  {
495  unsigned int nSize = 0;
496  if (opcode < OP_PUSHDATA1)
497  {
498  nSize = opcode;
499  }
500  else if (opcode == OP_PUSHDATA1)
501  {
502  if (end() - pc < 1)
503  return false;
504  nSize = *pc++;
505  }
506  else if (opcode == OP_PUSHDATA2)
507  {
508  if (end() - pc < 2)
509  return false;
510  nSize = 0;
511  memcpy(&nSize, &pc[0], 2);
512  pc += 2;
513  }
514  else if (opcode == OP_PUSHDATA4)
515  {
516  if (end() - pc < 4)
517  return false;
518  memcpy(&nSize, &pc[0], 4);
519  pc += 4;
520  }
521  if (end() - pc < 0 || (unsigned int)(end() - pc) < nSize)
522  return false;
523  if (pvchRet)
524  pvchRet->assign(pc, pc + nSize);
525  pc += nSize;
526  }
527 
528  opcodeRet = (opcodetype)opcode;
529  return true;
530  }
