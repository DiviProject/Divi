17 void CCoins::CalcMaskSize(unsigned int& nBytes, unsigned int& nNonzeroBytes) const
18 {
19  unsigned int nLastUsedByte = 0;
20  for (unsigned int b = 0; 2 + b * 8 < vout.size(); b++) {
21  bool fZero = true;
22  for (unsigned int i = 0; i < 8 && 2 + b * 8 + i < vout.size(); i++) {
23  if (!vout[2 + b * 8 + i].IsNull()) {
24  fZero = false;
25  continue;
26  }
27  }
28  if (!fZero) {
29  nLastUsedByte = b + 1;
30  nNonzeroBytes++;
31  }
32  }
33  nBytes += nLastUsedByte;

86 CCoinsMap::const_iterator CCoinsViewCache::FetchCoins(const uint256& txid) const
87 {
88  CCoinsMap::iterator it = cacheCoins.find(txid);
89  if (it != cacheCoins.end())
90  return it;
91  CCoins tmp;
92  if (!base->GetCoins(txid, tmp))
93  return cacheCoins.end();
94  CCoinsMap::iterator ret = cacheCoins.insert(std::make_pair(txid, CCoinsCacheEntry())).first;
95  tmp.swap(ret->second.coins);
96  if (ret->second.coins.IsPruned()) {
97  // The parent only has an empty entry for this txid; we can consider our
98  // version as fresh.
99  ret->second.flags = CCoinsCacheEntry::FRESH;
100  }
101  return ret;
102 }

114 CCoinsModifier CCoinsViewCache::ModifyCoins(const uint256& txid)
115 {
116  assert(!hasModifier);
117  std::pair<CCoinsMap::iterator, bool> ret = cacheCoins.insert(std::make_pair(txid, CCoinsCacheEntry()));
118  if (ret.second) {
119  if (!base->GetCoins(txid, ret.first->second.coins)) {
120  // The parent view does not have this entry; mark it as fresh.
121  ret.first->second.coins.Clear();
122  ret.first->second.flags = CCoinsCacheEntry::FRESH;
123  } else if (ret.first->second.coins.IsPruned()) {
124  // The parent view only has a pruned entry for this; mark it as fresh.
125  ret.first->second.flags = CCoinsCacheEntry::FRESH;
126  }
127  }
128  // Assume that whenever ModifyCoins is called, the entry will be modified.
129  ret.first->second.flags |= CCoinsCacheEntry::DIRTY;
130  return CCoinsModifier(*this, ret.first);
131 }

143 bool CCoinsViewCache::HaveCoins(const uint256& txid) const
144 {
145  CCoinsMap::const_iterator it = FetchCoins(txid);
146  // We're using vtx.empty() instead of IsPruned here for performance reasons,
147  // as we only care about the case where a transaction was replaced entirely
148  // in a reorganization (which wipes vout entirely, as opposed to spending
149  // which just cleans individual outputs).
150  return (it != cacheCoins.end() && !it->second.coins.vout.empty());

165 bool CCoinsViewCache::BatchWrite(CCoinsMap& mapCoins, const uint256& hashBlockIn)
166 {
167  assert(!hasModifier);
168  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
169  if (it->second.flags & CCoinsCacheEntry::DIRTY) { // Ignore non-dirty entries (optimization).
170  CCoinsMap::iterator itUs = cacheCoins.find(it->first);
171  if (itUs == cacheCoins.end()) {
172  if (!it->second.coins.IsPruned()) {
173  // The parent cache does not have an entry, while the child
174  // cache does have (a non-pruned) one. Move the data up, and
175  // mark it as fresh (if the grandparent did have it, we
176  // would have pulled it in at first GetCoins).
177  assert(it->second.flags & CCoinsCacheEntry::FRESH);
178  CCoinsCacheEntry& entry = cacheCoins[it->first];
179  entry.coins.swap(it->second.coins);
180  entry.flags = CCoinsCacheEntry::DIRTY | CCoinsCacheEntry::FRESH;
181  }
182  } else {
183  if ((itUs->second.flags & CCoinsCacheEntry::FRESH) && it->second.coins.IsPruned()) {
184  // The grandparent does not have an entry, and the child is
185  // modified and being pruned. This means we can just delete
186  // it from the parent.
187  cacheCoins.erase(itUs);
188  } else {
189  // A normal modification.
190  itUs->second.coins.swap(it->second.coins);
191  itUs->second.flags |= CCoinsCacheEntry::DIRTY;
192  }
193  }
194  }
195  CCoinsMap::iterator itOld = it++;
196  mapCoins.erase(itOld);
197  }
198  hashBlock = hashBlockIn;
199  return true;
200 }

251 double CCoinsViewCache::GetPriority(const CTransaction& tx, int nHeight) const
252 {
253  if (tx.IsCoinBase() || tx.IsCoinStake())
254  return 0.0;
255  double dResult = 0.0;
256  for (const CTxIn& txin: tx.vin) {
257  const CCoins* coins = AccessCoins(txin.prevout.hash);
258  assert(coins);
259  if (!coins->IsAvailable(txin.prevout.n)) continue;
