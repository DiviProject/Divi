109 void AddressBookPage::setModel(AddressTableModel* model)
110 {
111  this->model = model;
112  if (!model)
113  return;
114 
115  proxyModel = new QSortFilterProxyModel(this);
116  proxyModel->setSourceModel(model);
117  proxyModel->setDynamicSortFilter(true);
118  proxyModel->setSortCaseSensitivity(Qt::CaseInsensitive);
119  proxyModel->setFilterCaseSensitivity(Qt::CaseInsensitive);
120  switch (tab) {
121  case ReceivingTab:
122  // Receive filter
123  proxyModel->setFilterRole(AddressTableModel::TypeRole);
124  proxyModel->setFilterFixedString(AddressTableModel::Receive);
125  break;
126  case SendingTab:
127  // Send filter
128  proxyModel->setFilterRole(AddressTableModel::TypeRole);
129  proxyModel->setFilterFixedString(AddressTableModel::Send);
130  break;
131  }
132  ui->tableView->setModel(proxyModel);
133  ui->tableView->sortByColumn(0, Qt::AscendingOrder);
134 
135 // Set column widths
136 #if QT_VERSION < 0x050000
137  ui->tableView->horizontalHeader()->setResizeMode(AddressTableModel::Label, QHeaderView::Stretch);
138  ui->tableView->horizontalHeader()->setResizeMode(AddressTableModel::Address, QHeaderView::ResizeToContents);
139 #else
140  ui->tableView->horizontalHeader()->setSectionResizeMode(AddressTableModel::Label, QHeaderView::Stretch);
141  ui->tableView->horizontalHeader()->setSectionResizeMode(AddressTableModel::Address, QHeaderView::ResizeToContents);
142 #endif
143 
144  connect(ui->tableView->selectionModel(), SIGNAL(selectionChanged(QItemSelection, QItemSelection)),
145  this, SLOT(selectionChanged()));
146 
147  // Select row for newly created address
148  connect(model, SIGNAL(rowsInserted(QModelIndex, int, int)), this, SLOT(selectNewAddress(QModelIndex, int, int)));
149 
150  selectionChanged();
151 }

176  EditAddressDialog::EditSendingAddress :
177  EditAddressDialog::EditReceivingAddress,
178  this);
179  dlg.setModel(model);
180  QModelIndex origIndex = proxyModel->mapToSource(indexes.at(0));
181  dlg.loadRow(origIndex.row());

242 void AddressBookPage::done(int retval)
243 {
244  QTableView* table = ui->tableView;
245  if (!table->selectionModel() || !table->model())
246  return;
247 
248  // Figure out which address was selected, and return it
249  QModelIndexList indexes = table->selectionModel()->selectedRows(AddressTableModel::Address);
250 
251  foreach (QModelIndex index, indexes) {
252  QVariant address = table->model()->data(index);
253  returnValue = address.toString();
254  }
255 
256  if (returnValue.isEmpty()) {
257  // If no address entry selected, return rejected
258  retval = Rejected;
259  }
260 
261  QDialog::done(retval);
262 }
