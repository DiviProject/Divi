28 //------------------------------------------------------------------------------
29 // Tinyformat: A minimal type safe printf replacement
30 //
31 // tinyformat.h is a type safe printf replacement library in a single C++
32 // header file. Design goals include:
33 //
34 // * Type safety and extensibility for user defined types.
35 // * C99 printf() compatibility, to the extent possible using std::ostream
36 // * Simplicity and minimalism. A single header file to include and distribute
37 // with your projects.
38 // * Augment rather than replace the standard stream formatting mechanism
39 // * C++98 support, with optional C++11 niceties
40 //
41 //
42 // Main interface example usage
43 // ----------------------------
44 //
45 // To print a date to std::cout:
46 //
47 // std::string weekday = "Wednesday";
48 // const char* month = "July";
49 // size_t day = 27;
50 // long hour = 14;
51 // int min = 44;
52 //
53 // tfm::printf("%s, %s %d, %.2d:%.2d\n", weekday, month, day, hour, min);
54 //
55 // The strange types here emphasize the type safety of the interface; it is
56 // possible to print a std::string using the "%s" conversion, and a
57 // size_t using the "%d" conversion. A similar result could be achieved
58 // using either of the tfm::format() functions. One prints on a user provided
59 // stream:
60 //
61 // tfm::format(std::cerr, "%s, %s %d, %.2d:%.2d\n",
62 // weekday, month, day, hour, min);
63 //
64 // The other returns a std::string:
65 //
66 // std::string date = tfm::format("%s, %s %d, %.2d:%.2d\n",
67 // weekday, month, day, hour, min);
68 // std::cout << date;
69 //
70 // These are the three primary interface functions.
71 //
72 //
73 // User defined format functions
74 // -----------------------------
75 //
76 // Simulating variadic templates in C++98 is pretty painful since it requires
77 // writing out the same function for each desired number of arguments. To make
78 // this bearable tinyformat comes with a set of macros which are used
79 // internally to generate the API, but which may also be used in user code.
80 //
81 // The three macros TINYFORMAT_ARGTYPES(n), TINYFORMAT_VARARGS(n) and
82 // TINYFORMAT_PASSARGS(n) will generate a list of n argument types,
83 // type/name pairs and argument names respectively when called with an integer
84 // n between 1 and 16. We can use these to define a macro which generates the
85 // desired user defined function with n arguments. To generate all 16 user
86 // defined function bodies, use the macro TINYFORMAT_FOREACH_ARGNUM. For an
87 // example, see the implementation of printf() at the end of the source file.
88 //
89 //
90 // Additional API information
91 // --------------------------
92 //
93 // Error handling: Define TINYFORMAT_ERROR to customize the error handling for
94 // format strings which are unsupported or have the wrong number of format
95 // specifiers (calls assert() by default).
96 //
97 // User defined types: Uses operator<< for user defined types by default.
98 // Overload formatValue() for more control.

157 // Test whether type T1 is convertible to type T2
158 template <typename T1, typename T2>
159 struct is_convertible {
160 private:
161  // two types of different size
162  struct fail {
163  char dummy[2];
164  };
165  struct succeed {
166  char dummy;
167  };
168  // Try to convert a T1 to a T2 by plugging into tryConvert
169  static fail tryConvert(...);
170  static succeed tryConvert(const T2&);
171  static const T1& makeT1();
172 
173 public:
174 #ifdef _MSC_VER
175 // Disable spurious loss of precision warnings in tryConvert(makeT1())
176 #pragma warning(push)
177 #pragma warning(disable : 4244)
178 #pragma warning(disable : 4267)
179 #endif
180  // Standard trick: the (...) version of tryConvert will be chosen from
181  // the overload set only if the version taking a T2 doesn't match.
182  // Then we compare the sizes of the return types to check which
183  // function matched. Very neat, in a disgusting kind of way :)
184  static const bool value =
185  sizeof(tryConvert(makeT1())) == sizeof(succeed);
186 #ifdef _MSC_VER
187 #pragma warning(pop)
188 #endif
189 };

270 // Format a value into a stream. Called from format() for all types by default.
271 //
272 // Users may override this for their own types. When this function is called,
273 // the stream flags will have been modified according to the format string.
274 // The format specification is provided in the range [fmtBegin, fmtEnd).
275 //
276 // By default, formatValue() uses the usual stream insertion operator
277 // operator<< to format the type T, with special cases for the %c and %p
278 // conversions.
279 template <typename T>
280 inline void formatValue(std::ostream& out, const char* /*fmtBegin*/, const char* fmtEnd, const T& value)
281 {
282 #ifndef TINYFORMAT_ALLOW_WCHAR_STRINGS
283  // Since we don't support printing of wchar_t using "%ls", make it fail at
284  // compile time in preference to printing as a void* at runtime.
285  typedef typename detail::is_wchar<T>::tinyformat_wchar_is_not_supported DummyType;
286  (void)DummyType(); // avoid unused type warning with gcc-4.8
287 #endif
288  // The mess here is to support the %c and %p conversions: if these
289  // conversions are active we try to convert the type to a char or const
290  // void* respectively and format that instead of the value itself. For the
291  // %p conversion it's important to avoid dereferencing the pointer, which
292  // could otherwise lead to a crash when printing a dangling (const char*).
293  const bool canConvertToChar = detail::is_convertible<T, char>::value;
294  const bool canConvertToVoidPtr = detail::is_convertible<T, const void*>::value;
295  if (canConvertToChar && *(fmtEnd - 1) == 'c')
296  detail::formatValueAsType<T, char>::invoke(out, value);
297  else if (canConvertToVoidPtr && *(fmtEnd - 1) == 'p')
298  detail::formatValueAsType<T, const void*>::invoke(out, value);
299 #ifdef TINYFORMAT_OLD_LIBSTDCPLUSPLUS_WORKAROUND
300  else if (detail::formatZeroIntegerWorkaround<T>::invoke(out, value))
301  ;
302 #endif
303  else
304  out << value;
305 }

596 // Accept a value for formatting into the internal stream.
597 template <typename T>
598 TINYFORMAT_NOINLINE // < greatly reduces bloat in optimized builds
599  void
600  FormatIterator::accept(const T& value)
601 {
602  // Parse the format string
603  const char* fmtEnd = 0;
604  if (m_extraFlags == Flag_None && !m_wantWidth && !m_wantPrecision) {
605  m_fmt = printFormatStringLiteral(m_out, m_fmt);
606  fmtEnd = streamStateFromFormat(m_out, m_extraFlags, m_fmt, 0, 0);
607  m_wantWidth = (m_extraFlags & Flag_VariableWidth) != 0;
608  m_wantPrecision = (m_extraFlags & Flag_VariablePrecision) != 0;
609  }
610  // Consume value as variable width and precision specifier if necessary
611  if (m_extraFlags & (Flag_VariableWidth | Flag_VariablePrecision)) {
612  if (m_wantWidth || m_wantPrecision) {
613  int v = convertToInt<T>::invoke(value);
614  if (m_wantWidth) {
615  m_variableWidth = v;
616  m_wantWidth = false;
617  } else if (m_wantPrecision) {
618  m_variablePrecision = v;
619  m_wantPrecision = false;
620  }
621  return;
622  }
623  // If we get here, we've set both the variable precision and width as
624  // required and we need to rerun the stream state setup to insert these.
625  fmtEnd = streamStateFromFormat(m_out, m_extraFlags, m_fmt,
626  m_variableWidth, m_variablePrecision);
627  }
628 
629  // Format the value into the stream.
630  if (!(m_extraFlags & (Flag_SpacePadPositive | Flag_TruncateToPrecision)))
631  formatValue(m_out, m_fmt, fmtEnd, value);
632  else {
633  // The following are special cases where there's no direct
634  // correspondence between stream formatting and the printf() behaviour.
635  // Instead, we simulate the behaviour crudely by formatting into a
636  // temporary string stream and munging the resulting string.
637  std::ostringstream tmpStream;
638  tmpStream.copyfmt(m_out);
639  if (m_extraFlags & Flag_SpacePadPositive)
640  tmpStream.setf(std::ios::showpos);
641  // formatCStringTruncate is required for truncating conversions like
642  // "%.4s" where at most 4 characters of the c-string should be read.
643  // If we didn't include this special case, we might read off the end.
644  if (!((m_extraFlags & Flag_TruncateToPrecision) &&
645  formatCStringTruncate(tmpStream, value, m_out.precision()))) {
646  // Not a truncated c-string; just format normally.
647  formatValue(tmpStream, m_fmt, fmtEnd, value);
648  }
649  std::string result = tmpStream.str(); // allocates... yuck.
650  if (m_extraFlags & Flag_SpacePadPositive) {
651  for (size_t i = 0, iend = result.size(); i < iend; ++i)
652  if (result[i] == '+')
653  result[i] = ' ';
654  }
655  if ((m_extraFlags & Flag_TruncateToPrecision) &&
656  (int)result.size() > (int)m_out.precision())
657  m_out.write(result.c_str(), m_out.precision());
658  else
659  m_out << result;
660  }
661  m_extraFlags = Flag_None;
662  m_fmt = fmtEnd;
663 }

666 // Parse a format string and set the stream state accordingly.
667 //
668 // The format mini-language recognized here is meant to be the one from C99,
669 // with the form "%[flags][width][.precision][length]type".
670 //
671 // Formatting options which can't be natively represented using the ostream
672 // state are returned in the extraFlags parameter which is a bitwise
673 // combination of values from the ExtraFormatFlags enum.
674 inline const char* FormatIterator::streamStateFromFormat(std::ostream& out,
675  unsigned int& extraFlags,
676  const char* fmtStart,
677  int variableWidth,
678  int variablePrecision)
679 {
680  if (*fmtStart != '%') {
681  TINYFORMAT_ERROR("tinyformat: Not enough conversion specifiers in format string");
682  return fmtStart;
683  }
684  // Reset stream state to defaults.
685  out.width(0);
686  out.precision(6);
687  out.fill(' ');
688  // Reset most flags; ignore irrelevant unitbuf & skipws.
689  out.unsetf(std::ios::adjustfield | std::ios::basefield |
690  std::ios::floatfield | std::ios::showbase | std::ios::boolalpha |
691  std::ios::showpoint | std::ios::showpos | std::ios::uppercase);
692  extraFlags = Flag_None;
693  bool precisionSet = false;
694  bool widthSet = false;
695  const char* c = fmtStart + 1;
696  // 1) Parse flags
697  for (;; ++c) {
698  switch (*c) {
699  case '#':
700  out.setf(std::ios::showpoint | std::ios::showbase);
701  continue;
702  case '0':
703  // overridden by left alignment ('-' flag)
704  if (!(out.flags() & std::ios::left)) {
705  // Use internal padding so that numeric values are
706  // formatted correctly, eg -00010 rather than 000-10
707  out.fill('0');
708  out.setf(std::ios::internal, std::ios::adjustfield);
709  }
710  continue;
711  case '-':
712  out.fill(' ');
713  out.setf(std::ios::left, std::ios::adjustfield);
714  continue;
715  case ' ':
716  // overridden by show positive sign, '+' flag.
717  if (!(out.flags() & std::ios::showpos))
718  extraFlags |= Flag_SpacePadPositive;
719  continue;
720  case '+':
721  out.setf(std::ios::showpos);
722  extraFlags &= ~Flag_SpacePadPositive;
723  continue;
724  }
725  break;
726  }
727  // 2) Parse width
728  if (*c >= '0' && *c <= '9') {
729  widthSet = true;
730  out.width(parseIntAndAdvance(c));
731  }
732  if (*c == '*') {
733  widthSet = true;
734  if (variableWidth < 0) {
735  // negative widths correspond to '-' flag set
736  out.fill(' ');
737  out.setf(std::ios::left, std::ios::adjustfield);
738  variableWidth = -variableWidth;
739  }
740  out.width(variableWidth);
741  extraFlags |= Flag_VariableWidth;
742  ++c;
743  }
744  // 3) Parse precision
745  if (*c == '.') {
746  ++c;
747  int precision = 0;
748  if (*c == '*') {
749  ++c;
750  extraFlags |= Flag_VariablePrecision;
751  precision = variablePrecision;
752  } else {
753  if (*c >= '0' && *c <= '9')
754  precision = parseIntAndAdvance(c);
755  else if (*c == '-') // negative precisions ignored, treated as zero.
756  parseIntAndAdvance(++c);
757  }
758  out.precision(precision);
759  precisionSet = true;
760  }
761  // 4) Ignore any C99 length modifier
762  while (*c == 'l' || *c == 'h' || *c == 'L' ||
763  *c == 'j' || *c == 'z' || *c == 't')
764  ++c;
765  // 5) We're up to the conversion specifier character.
766  // Set stream flags based on conversion specifier (thanks to the
767  // boost::format class for forging the way here).
768  bool intConversion = false;
769  switch (*c) {
770  case 'u':
771  case 'd':
772  case 'i':
773  out.setf(std::ios::dec, std::ios::basefield);
774  intConversion = true;
775  break;
776  case 'o':
777  out.setf(std::ios::oct, std::ios::basefield);
778  intConversion = true;
779  break;
780  case 'X':
781  out.setf(std::ios::uppercase);
782  case 'x':
783  case 'p':
784  out.setf(std::ios::hex, std::ios::basefield);
785  intConversion = true;
786  break;
787  case 'E':
788  out.setf(std::ios::uppercase);
789  case 'e':
790  out.setf(std::ios::scientific, std::ios::floatfield);
791  out.setf(std::ios::dec, std::ios::basefield);
792  break;
793  case 'F':
794  out.setf(std::ios::uppercase);
795  case 'f':
796  out.setf(std::ios::fixed, std::ios::floatfield);
797  break;
798  case 'G':
799  out.setf(std::ios::uppercase);
800  case 'g':
801  out.setf(std::ios::dec, std::ios::basefield);
802  // As in boost::format, let stream decide float format.
803  out.flags(out.flags() & ~std::ios::floatfield);
804  break;
805  case 'a':
806  case 'A':
807  TINYFORMAT_ERROR("tinyformat: the %a and %A conversion specs "
808  "are not supported");
809  break;
810  case 'c':
811  // Handled as special case inside formatValue()
812  break;
813  case 's':
814  if (precisionSet)
815  extraFlags |= Flag_TruncateToPrecision;
816  // Make %s print booleans as "true" and "false"
817  out.setf(std::ios::boolalpha);
818  break;
819  case 'n':
820  // Not supported - will cause problems!
821  TINYFORMAT_ERROR("tinyformat: %n conversion spec not supported");
822  break;
823  case '\0':
824  TINYFORMAT_ERROR("tinyformat: Conversion spec incorrectly "
825  "terminated by end of string");
826  return c;
827  }
828  if (intConversion && precisionSet && !widthSet) {
829  // "precision" for integers gives the minimum number of digits (to be
830  // padded with zeros on the left). This isn't really supported by the
831  // iostreams, but we can approximately simulate it with the width if
832  // the width isn't otherwise used.
833  out.width(out.precision());
834  out.setf(std::ios::internal, std::ios::adjustfield);
835  out.fill('0');
836  }
837  return c + 1;
838 }
