126 // find 'best' local address for a particular peer
127 bool GetLocal(CService& addr, const CNetAddr* paddrPeer)
128 {
129  if (!fListen)
130  return false;
131 
132  int nBestScore = -1;
133  int nBestReachability = -1;
134  {
135  LOCK(cs_mapLocalHost);
136  for (map<CNetAddr, LocalServiceInfo>::iterator it = mapLocalHost.begin(); it != mapLocalHost.end(); it++) {
137  int nScore = (*it).second.nScore;
138  int nReachability = (*it).first.GetReachabilityFrom(paddrPeer);
139  if (nReachability > nBestReachability || (nReachability == nBestReachability && nScore > nBestScore)) {
140  addr = CService((*it).first, (*it).second.nPort);
141  nBestReachability = nReachability;
142  nBestScore = nScore;
143  }
144  }
145  }
146  return nBestScore >= 0;
147 }

165 bool RecvLine(SOCKET hSocket, string& strLine)
166 {
167  strLine = "";
168  while (true) {
169  char c;
170  int nBytes = recv(hSocket, &c, 1, 0);
171  if (nBytes > 0) {
172  if (c == '\n')
173  continue;
174  if (c == '\r')
175  return true;
176  strLine += c;
177  if (strLine.size() >= 9000)
178  return true;
179  } else if (nBytes <= 0) {
180  boost::this_thread::interruption_point();
181  if (nBytes < 0) {
182  int nErr = WSAGetLastError();
183  if (nErr == WSAEMSGSIZE)
184  continue;
185  if (nErr == WSAEWOULDBLOCK || nErr == WSAEINTR || nErr == WSAEINPROGRESS) {
186  MilliSleep(10);
187  continue;
188  }
189  }
190  if (!strLine.empty())
191  return true;
192  if (nBytes == 0) {
193  // socket closed
194  LogPrint("net", "socket closed\n");
195  return false;
196  } else {
197  // socket error
198  int nErr = WSAGetLastError();
199  LogPrint("net", "recv failed: %s\n", NetworkErrorString(nErr));
200  return false;
201  }
202  }
203  }
204 }

388 CNode* ConnectNode(CAddress addrConnect, const char* pszDest, bool obfuScationMaster)
389 {
390  if (pszDest == NULL) {
391  // we clean masternode connections in CMasternodeMan::ProcessMasternodeConnections()
392  // so should be safe to skip this and connect to local Hot MN on CActiveMasternode::ManageStatus()
393  if (IsLocal(addrConnect) && !obfuScationMaster)
394  return NULL;
395 
396  // Look for an existing connection
397  CNode* pnode = FindNode((CService)addrConnect);
398  if (pnode) {
399  pnode->fObfuScationMaster = obfuScationMaster;
400 
401  pnode->AddRef();
402  return pnode;
403  }
404  }
405 
407  LogPrint("net", "trying connection %s lastseen=%.1fhrs\n",
408  pszDest ? pszDest : addrConnect.ToString(),
409  pszDest ? 0.0 : (double)(GetAdjustedTime() - addrConnect.nTime) / 3600.0);
410 
411  // Connect
412  SOCKET hSocket;
413  bool proxyConnectionFailed = false;
414  if (pszDest ? ConnectSocketByName(addrConnect, hSocket, pszDest, Params().GetDefaultPort(), nConnectTimeout, &proxyConnectionFailed) :
415  ConnectSocket(addrConnect, hSocket, nConnectTimeout, &proxyConnectionFailed)) {
416  if (!IsSelectableSocket(hSocket)) {
417  LogPrintf("Cannot create connection: non-selectable socket created (fd >= FD_SETSIZE ?)\n");
418  CloseSocket(hSocket);
419  return NULL;
420  }
421 
422  addrman.Attempt(addrConnect);
423 
424  // Add node
425  CNode* pnode = new CNode(hSocket, addrConnect, pszDest ? pszDest : "", false);
426  pnode->AddRef();
427 
428  {
429  LOCK(cs_vNodes);
430  vNodes.push_back(pnode);
431  }
432 
433  pnode->nTimeConnected = GetTime();
434  if (obfuScationMaster) pnode->fObfuScationMaster = true;
435 
436  return pnode;
437  } else if (!proxyConnectionFailed) {
438  // If connecting to the node failed, and failure is not caused by a problem connecting to
439  // the proxy, mark this as an attempt.
440  addrman.Attempt(addrConnect);
441  }
442 
443  return NULL;
444 }

583 bool CNode::ReceiveMsgBytes(const char* pch, unsigned int nBytes)
584 {
585  while (nBytes > 0) {
586  // get current incomplete message, or create a new one
587  if (vRecvMsg.empty() ||
588  vRecvMsg.back().complete())
589  vRecvMsg.push_back(CNetMessage(SER_NETWORK, nRecvVersion));
590 
591  CNetMessage& msg = vRecvMsg.back();
592 
593  // absorb network data
594  int handled;
595  if (!msg.in_data)
596  handled = msg.readHeader(pch, nBytes);
597  else
598  handled = msg.readData(pch, nBytes);
599 
600  if (handled < 0)
601  return false;
602 
603  if (msg.in_data && msg.hdr.nMessageSize > MAX_PROTOCOL_MESSAGE_LENGTH) {
604  LogPrint("net", "Oversized message from peer=%i, disconnecting", GetId());
605  return false;
606  }
607 
608  pch += handled;
609  nBytes -= handled;
610 
611  if (msg.complete()) {
612  msg.nTime = GetTimeMicros();
613  messageHandlerCondition.notify_one();
614  }
615  }
616 
617  return true;
618 }

620 int CNetMessage::readHeader(const char* pch, unsigned int nBytes)
621 {
622  // copy data to temporary parsing buffer
623  unsigned int nRemaining = 24 - nHdrPos;
624  unsigned int nCopy = std::min(nRemaining, nBytes);
625 
626  memcpy(&hdrbuf[nHdrPos], pch, nCopy);
627  nHdrPos += nCopy;
628 
629  // if header incomplete, exit
630  if (nHdrPos < 24)
631  return nCopy;
632 
633  // deserialize to CMessageHeader
634  try {
635  hdrbuf >> hdr;
636  } catch (const std::exception&) {
637  return -1;
638  }
639 
640  // reject messages larger than MAX_SIZE
641  if (hdr.nMessageSize > MAX_SIZE)
642  return -1;
643 
644  // switch state to reading message data
645  in_data = true;
646 
647  return nCopy;
648 }

650 int CNetMessage::readData(const char* pch, unsigned int nBytes)
651 {
652  unsigned int nRemaining = hdr.nMessageSize - nDataPos;
653  unsigned int nCopy = std::min(nRemaining, nBytes);
654 
655  if (vRecv.size() < nDataPos + nCopy) {
656  // Allocate up to 256 KiB ahead, but never more than the total message size.
657  vRecv.resize(std::min(hdr.nMessageSize, nDataPos + nCopy + 256 * 1024));
658  }
659 
660  memcpy(&vRecv[nDataPos], pch, nCopy);
661  nDataPos += nCopy;
662 
663  return nCopy;

667 // requires LOCK(cs_vSend)
668 void SocketSendData(CNode* pnode)
669 {
670  std::deque<CSerializeData>::iterator it = pnode->vSendMsg.begin();
671 
672  while (it != pnode->vSendMsg.end()) {
673  const CSerializeData& data = *it;
674  assert(data.size() > pnode->nSendOffset);
675  int nBytes = send(pnode->hSocket, &data[pnode->nSendOffset], data.size() - pnode->nSendOffset, MSG_NOSIGNAL | MSG_DONTWAIT);
676  if (nBytes > 0) {
677  pnode->nLastSend = GetTime();
678  pnode->nSendBytes += nBytes;
679  pnode->nSendOffset += nBytes;
680  pnode->RecordBytesSent(nBytes);
681  if (pnode->nSendOffset == data.size()) {
682  pnode->nSendOffset = 0;
683  pnode->nSendSize -= data.size();
684  it++;
685  } else {
686  // could not send full message; stop sending more
687  break;
688  }
689  } else {
690  if (nBytes < 0) {
691  // error
692  int nErr = WSAGetLastError();
693  if (nErr != WSAEWOULDBLOCK && nErr != WSAEMSGSIZE && nErr != WSAEINTR && nErr != WSAEINPROGRESS) {
694  LogPrintf("socket send error %s\n", NetworkErrorString(nErr));
695  pnode->CloseSocketDisconnect();
696  }
697  }
698  // couldn't send anything at all
699  break;
700  }
701  }
702 
703  if (it == pnode->vSendMsg.end()) {
704  assert(pnode->nSendOffset == 0);
705  assert(pnode->nSendSize == 0);
706  }
707  pnode->vSendMsg.erase(pnode->vSendMsg.begin(), it);

712 void ThreadSocketHandler()
713 {
714  unsigned int nPrevNodeCount = 0;
715  while (true) {
716  //
717  // Disconnect nodes
718  //
719  {
720  LOCK(cs_vNodes);
721  // Disconnect unused nodes
722  vector<CNode*> vNodesCopy = vNodes;
723  BOOST_FOREACH (CNode* pnode, vNodesCopy) {
724  if (pnode->fDisconnect ||
725  (pnode->GetRefCount() <= 0 && pnode->vRecvMsg.empty() && pnode->nSendSize == 0 && pnode->ssSend.empty())) {
726  // remove from vNodes
727  vNodes.erase(remove(vNodes.begin(), vNodes.end(), pnode), vNodes.end());
728 
729  // release outbound grant (if any)
730  pnode->grantOutbound.Release();
731 
732  // close socket and cleanup
733  pnode->CloseSocketDisconnect();
734 
735  // hold in disconnected pool until all refs are released
736  if (pnode->fNetworkNode || pnode->fInbound)
737  pnode->Release();

739  }
740  }
741  }
742  {
743  // Delete disconnected nodes
744  list<CNode*> vNodesDisconnectedCopy = vNodesDisconnected;
745  BOOST_FOREACH (CNode* pnode, vNodesDisconnectedCopy) {
746  // wait until threads are done using it
747  if (pnode->GetRefCount() <= 0) {
748  bool fDelete = false;
749  {
750  TRY_LOCK(pnode->cs_vSend, lockSend);
751  if (lockSend) {
752  TRY_LOCK(pnode->cs_vRecvMsg, lockRecv);
753  if (lockRecv) {
754  TRY_LOCK(pnode->cs_inventory, lockInv);
755  if (lockInv)
756  fDelete = true;
757  }
758  }
759  }
760  if (fDelete) {
761  vNodesDisconnected.remove(pnode);
762  delete pnode;
763  }
764  }
765  }
766  }
767  size_t vNodesSize;
768  {
769  LOCK(cs_vNodes);
770  vNodesSize = vNodes.size();
771  }
772  if(vNodesSize != nPrevNodeCount) {
773  nPrevNodeCount = vNodesSize;
774  uiInterface.NotifyNumConnectionsChanged(nPrevNodeCount);
775  }
776 
777  //
778  // Find which sockets have data to receive
779  //
780  struct timeval timeout;
781  timeout.tv_sec = 0;
782  timeout.tv_usec = 50000; // frequency to poll pnode->vSend
783 
784  fd_set fdsetRecv;
785  fd_set fdsetSend;
786  fd_set fdsetError;
787  FD_ZERO(&fdsetRecv);
788  FD_ZERO(&fdsetSend);
789  FD_ZERO(&fdsetError);
790  SOCKET hSocketMax = 0;
791  bool have_fds = false;
792 
793  BOOST_FOREACH (const ListenSocket& hListenSocket, vhListenSocket) {
794  FD_SET(hListenSocket.socket, &fdsetRecv);
795  hSocketMax = max(hSocketMax, hListenSocket.socket);
796  have_fds = true;
797  }
798 
799  {
800  LOCK(cs_vNodes);
801  BOOST_FOREACH (CNode* pnode, vNodes) {
802  if (pnode->hSocket == INVALID_SOCKET)
803  continue;
804  FD_SET(pnode->hSocket, &fdsetError);
805  hSocketMax = max(hSocketMax, pnode->hSocket);
806  have_fds = true;
807 
808  // Implement the following logic:
809  // * If there is data to send, select() for sending data. As this only
810  // happens when optimistic write failed, we choose to first drain the
811  // write buffer in this case before receiving more. This avoids
812  // needlessly queueing received data, if the remote peer is not themselves
813  // receiving data. This means properly utilizing TCP flow control signalling.
814  // * Otherwise, if there is no (complete) message in the receive buffer,
815  // or there is space left in the buffer, select() for receiving data.
816  // * (if neither of the above applies, there is certainly one message
817  // in the receiver buffer ready to be processed).
818  // Together, that means that at least one of the following is always possible,
819  // so we don't deadlock:
820  // * We send some data.
821  // * We wait for data to be received (and disconnect after timeout).
822  // * We process a message in the buffer (message handler thread).
823  {
824  TRY_LOCK(pnode->cs_vSend, lockSend);
825  if (lockSend && !pnode->vSendMsg.empty()) {
826  FD_SET(pnode->hSocket, &fdsetSend);
827  continue;
828  }
829  }
830  {
831  TRY_LOCK(pnode->cs_vRecvMsg, lockRecv);
832  if (lockRecv && (pnode->vRecvMsg.empty() || !pnode->vRecvMsg.front().complete() ||
833  pnode->GetTotalRecvSize() <= ReceiveFloodSize()))
834  FD_SET(pnode->hSocket, &fdsetRecv);
835  }
836  }
837  }
838 
839  int nSelect = select(have_fds ? hSocketMax + 1 : 0,
840  &fdsetRecv, &fdsetSend, &fdsetError, &timeout);
841  boost::this_thread::interruption_point();
842 
843  if (nSelect == SOCKET_ERROR) {
844  if (have_fds) {
845  int nErr = WSAGetLastError();
846  LogPrintf("socket select error %s\n", NetworkErrorString(nErr));
847  for (unsigned int i = 0; i <= hSocketMax; i++)
848  FD_SET(i, &fdsetRecv);
849  }
850  FD_ZERO(&fdsetSend);
851  FD_ZERO(&fdsetError);
852  MilliSleep(timeout.tv_usec / 1000);
853  }

856  // Accept new connections
857  //

874  if (pnode->fInbound)

881  LogPrintf("socket error accept failed: %s\n", NetworkErrorString(nErr));
882  } else if (!IsSelectableSocket(hSocket)) {
883  LogPrintf("connection from %s dropped: non-selectable socket\n", addr.ToString());
884  CloseSocket(hSocket);
885  } else if (nInbound >= nMaxConnections - MAX_OUTBOUND_CONNECTIONS) {
886  LogPrint("net", "connection from %s dropped (full)\n", addr.ToString());
887  CloseSocket(hSocket);
888  } else if (CNode::IsBanned(addr) && !whitelisted) {
889  LogPrintf("connection from %s dropped (banned)\n", addr.ToString());
890  CloseSocket(hSocket);
891  } else {
892  CNode* pnode = new CNode(hSocket, addr, "", true);
893  pnode->AddRef();
894  pnode->fWhitelisted = whitelisted;
895 
896  {
897  LOCK(cs_vNodes);
898  vNodes.push_back(pnode);
899  }
900  }
901  }
902  }

905  // Service each socket
906  //

908  {
909  LOCK(cs_vNodes);
910  vNodesCopy = vNodes;
911  BOOST_FOREACH (CNode* pnode, vNodesCopy)
912  pnode->AddRef();
913  }
914  BOOST_FOREACH (CNode* pnode, vNodesCopy) {
915  boost::this_thread::interruption_point();

918  // Receive
919  //
920  if (pnode->hSocket == INVALID_SOCKET)
921  continue;
922  if (FD_ISSET(pnode->hSocket, &fdsetRecv) || FD_ISSET(pnode->hSocket, &fdsetError)) {

934  pnode->RecordBytesRecv(nBytes);

936  // socket closed gracefully

938  LogPrint("net", "socket closed\n");

971  pnode->fDisconnect = true;
972  } else if (nTime - pnode->nLastSend > TIMEOUT_INTERVAL) {
973  LogPrintf("socket sending timeout: %is\n", nTime - pnode->nLastSend);
974  pnode->fDisconnect = true;
975  } else if (nTime - pnode->nLastRecv > (pnode->nVersion > BIP0031_VERSION ? TIMEOUT_INTERVAL : 90 * 60)) {
976  LogPrintf("socket receive timeout: %is\n", nTime - pnode->nLastRecv);
977  pnode->fDisconnect = true;

979  LogPrintf("ping timeout: %fs\n", 0.000001 * (GetTimeMicros() - pnode->nPingUsecStart));
980  pnode->fDisconnect = true;
981  }
982  }
983  }
984  {
985  LOCK(cs_vNodes);
986  BOOST_FOREACH (CNode* pnode, vNodesCopy)
987  pnode->Release();
988  }
