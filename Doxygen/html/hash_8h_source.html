121 inline std::string Hash(std::string input)
122 {
123  unsigned char hash[SHA256_DIGEST_LENGTH];
124  SHA256_CTX sha256;
125  SHA256_Init(&sha256);
126  SHA256_Update(&sha256, input.c_str(), input.size());
127  SHA256_Final(hash, &sha256);
128  stringstream ss;
129  for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
130  ss << hex << setw(2) << setfill('0') << (int)hash[i];
131  }
132  return ss.str();
133 }
