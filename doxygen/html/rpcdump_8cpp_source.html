40 int64_t static DecodeDumpTime(const std::string& str)
41 {
42  static const boost::posix_time::ptime epoch = boost::posix_time::from_time_t(0);
43  static const std::locale loc(std::locale::classic(),
44  new boost::posix_time::time_input_facet("%Y-%m-%dT%H:%M:%SZ"));
45  std::istringstream iss(str);
46  iss.imbue(loc);
47  boost::posix_time::ptime ptime(boost::date_time::not_a_date_time);
48  iss >> ptime;
49  if (ptime.is_not_a_date_time())
50  return 0;
51  return (ptime - epoch).total_seconds();
52 }

54 std::string static EncodeDumpString(const std::string& str)
55 {
56  std::stringstream ret;
57  BOOST_FOREACH (unsigned char c, str) {
58  if (c <= 32 || c >= 128 || c == '%') {
59  ret << '%' << HexStr(&c, &c + 1);
60  } else {
61  ret << c;
62  }
63  }
64  return ret.str();
65 }

67 std::string DecodeDumpString(const std::string& str)
68 {
69  std::stringstream ret;
70  for (unsigned int pos = 0; pos < str.length(); pos++) {
71  unsigned char c = str[pos];
72  if (c == '%' && pos + 2 < str.length()) {
73  c = (((str[pos + 1] >> 6) * 9 + ((str[pos + 1] - '0') & 15)) << 4) |
74  ((str[pos + 2] >> 6) * 9 + ((str[pos + 2] - '0') & 15));
75  pos += 2;
76  }
77  ret << c;
78  }
79  return ret.str();
80 }

211 Value importwallet(const Array& params, bool fHelp)
212 {
213  if (fHelp || params.size() != 1)
214  throw runtime_error(
215  "importwallet \"filename\"\n"
216  "\nImports keys from a wallet dump file (see dumpwallet).\n"
217  "\nArguments:\n"
218  "1. \"filename\" (string, required) The wallet file\n"
219  "\nExamples:\n"
220  "\nDump the wallet\n" +
221  HelpExampleCli("dumpwallet", "\"test\"") +
222  "\nImport the wallet\n" + HelpExampleCli("importwallet", "\"test\"") +
223  "\nImport using the json rpc call\n" + HelpExampleRpc("importwallet", "\"test\""));
224 
225  EnsureWalletIsUnlocked();
226 
227  ifstream file;
228  file.open(params[0].get_str().c_str(), std::ios::in | std::ios::ate);
229  if (!file.is_open())
230  throw JSONRPCError(RPC_INVALID_PARAMETER, "Cannot open wallet dump file");
231 
232  int64_t nTimeBegin = chainActive.Tip()->GetBlockTime();
233 
234  bool fGood = true;
235 
236  int64_t nFilesize = std::max((int64_t)1, (int64_t)file.tellg());
237  file.seekg(0, file.beg);
238 
239  pwalletMain->ShowProgress(_("Importing..."), 0); // show progress dialog in GUI
240  while (file.good()) {
241  pwalletMain->ShowProgress("", std::max(1, std::min(99, (int)(((double)file.tellg() / (double)nFilesize) * 100))));
242  std::string line;
243  std::getline(file, line);
244  if (line.empty() || line[0] == '#')
245  continue;
246 
247  std::vector<std::string> vstr;
248  boost::split(vstr, line, boost::is_any_of(" "));
249  if (vstr.size() < 2)
250  continue;
251  CBitcoinSecret vchSecret;
252  if (!vchSecret.SetString(vstr[0]))
253  continue;
254  CKey key = vchSecret.GetKey();
255  CPubKey pubkey = key.GetPubKey();
256  assert(key.VerifyPubKey(pubkey));
257  CKeyID keyid = pubkey.GetID();
258  if (pwalletMain->HaveKey(keyid)) {
259  LogPrintf("Skipping import of %s (key already present)\n", CBitcoinAddress(keyid).ToString());
260  continue;
261  }
262  int64_t nTime = DecodeDumpTime(vstr[1]);
263  std::string strLabel;
264  bool fLabel = true;
265  for (unsigned int nStr = 2; nStr < vstr.size(); nStr++) {
266  if (boost::algorithm::starts_with(vstr[nStr], "#"))
267  break;
268  if (vstr[nStr] == "change=1")
269  fLabel = false;
270  if (vstr[nStr] == "reserve=1")
271  fLabel = false;
272  if (boost::algorithm::starts_with(vstr[nStr], "label=")) {
273  strLabel = DecodeDumpString(vstr[nStr].substr(6));
274  fLabel = true;
275  }
276  }
277  LogPrintf("Importing %s...\n", CBitcoinAddress(keyid).ToString());
278  if (!pwalletMain->AddKeyPubKey(key, pubkey)) {
279  fGood = false;
280  continue;
281  }
282  pwalletMain->mapKeyMetadata[keyid].nCreateTime = nTime;
283  if (fLabel)
284  pwalletMain->SetAddressBook(keyid, strLabel, "receive");
285  nTimeBegin = std::min(nTimeBegin, nTime);
286  }
287  file.close();
288  pwalletMain->ShowProgress("", 100); // hide progress dialog in GUI
289 
290  CBlockIndex* pindex = chainActive.Tip();
291  while (pindex && pindex->pprev && pindex->GetBlockTime() > nTimeBegin - 7200)
292  pindex = pindex->pprev;
293 
294  if (!pwalletMain->nTimeFirstKey || nTimeBegin < pwalletMain->nTimeFirstKey)
295  pwalletMain->nTimeFirstKey = nTimeBegin;
296 
297  LogPrintf("Rescanning last %i blocks\n", chainActive.Height() - pindex->nHeight + 1);
298  pwalletMain->ScanForWalletTransactions(pindex);
299  pwalletMain->MarkDirty();
300 
301  if (!fGood)
302  throw JSONRPCError(RPC_WALLET_ERROR, "Error adding some keys to wallet");
303 
304  return Value::null;
305 }

350  ofstream file;

363  vKeyBirth.push_back(std::make_pair(it->second, it->first));

370  file << strprintf("# * Created on %s\n", EncodeDumpTime(GetTime()));

372  file << strprintf("# mined on %s\n", EncodeDumpTime(chainActive.Tip()->GetBlockTime()));
373  file << "\n";
374  for (std::vector<std::pair<int64_t, CKeyID> >::const_iterator it = vKeyBirth.begin(); it != vKeyBirth.end(); it++) {
375  const CKeyID& keyid = it->second;
376  std::string strTime = EncodeDumpTime(it->first);
377  std::string strAddr = CBitcoinAddress(keyid).ToString();
378  CKey key;
379  if (pwalletMain->GetKey(keyid, key)) {
380  if (pwalletMain->mapAddressBook.count(keyid)) {
381  file << strprintf("%s %s label=%s # addr=%s\n", CBitcoinSecret(key).ToString(), strTime, EncodeDumpString(pwalletMain->mapAddressBook[keyid].name), strAddr);
382  } else if (setKeyPool.count(keyid)) {
383  file << strprintf("%s %s reserve=1 # addr=%s\n", CBitcoinSecret(key).ToString(), strTime, strAddr);
384  } else {
385  file << strprintf("%s %s change=1 # addr=%s\n", CBitcoinSecret(key).ToString(), strTime, strAddr);
386  }
387  }
388  }
389  file << "\n";
390  file << "# End of dump\n";
391  file.close();
392  return Value::null;
393 }
