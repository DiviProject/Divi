99 bool getIdealSpends(
100  const CAmount nValueTarget,
101  const std::list<CZerocoinMint>& listMints,
102  const std::map<CoinDenomination, CAmount> mapOfDenomsHeld,
103  std::map<CoinDenomination, CAmount>& mapOfDenomsUsed)
104 {
105  CAmount nRemainingValue = nValueTarget;
106  // Initialize
107  for (const auto& denom : zerocoinDenomList)
108  mapOfDenomsUsed.insert(std::pair<CoinDenomination, CAmount>(denom, 0));
109 
110  // Start with the Highest Denomination coin and grab coins as long as the remaining amount is greater than the
111  // current denomination value
112  for (auto& coin : reverse_iterate(zerocoinDenomList)) {
113  for (const CZerocoinMint mint : listMints) {
114  if (mint.IsUsed()) continue;
115  if (nRemainingValue >= ZerocoinDenominationToAmount(coin) && coin == mint.GetDenomination()) {
116  mapOfDenomsUsed.at(coin)++;
117  nRemainingValue -= mint.GetDenominationAsAmount();
118  }
119  if (nRemainingValue < ZerocoinDenominationToAmount(coin)) break;
120  }
121  }
122  return (nRemainingValue == 0);
123 }

127 // -------------------------------------------------------------------------------------------------------

131  CAmount& nCoinsSpentValue)
132 {

136  do {
137  for (const CZerocoinMint mint : listMints) {
138  if (mint.IsUsed()) continue;
139  if (coin == mint.GetDenomination() && mapOfDenomsUsed.at(coin)) {
140  vSelectedMints.push_back(mint);
141  nCoinsSpentValue += ZerocoinDenominationToAmount(coin);
142  mapOfDenomsUsed.at(coin)--;
143  }
144  }
145  } while (mapOfDenomsUsed.at(coin));
146  }
147  return vSelectedMints;

219 int minimizeChange(
220  int nMaxNumberOfSpends,
221  int nChangeCount,
222  const CoinDenomination nextToMaxDenom,
223  const CAmount nValueTarget,
224  const std::map<CoinDenomination, CAmount>& mapOfDenomsHeld,
225  std::map<CoinDenomination, CAmount>& mapOfDenomsUsed)
226 {
227  // Now find out if possible without using 1 coin such that we have more spends but less change
228  // First get set of coins close to value but still less than value (since not exact)
229  CAmount nRemainingValue = nValueTarget;
230  CAmount AmountUsed = 0;
231  int nCoinCount = 0;
232 
233  // Re-clear this
234  std::map<CoinDenomination, CAmount> savedMapOfDenomsUsed = mapOfDenomsUsed;
235  for (const auto& denom : zerocoinDenomList)
236  mapOfDenomsUsed.at(denom) = 0;
237 
238  // Find the amount this is less than total but uses up higher denoms first,
239  // starting at the denom that is not greater than the overall total
240  for (const auto& denom : reverse_iterate(zerocoinDenomList)) {
241  if (denom <= nextToMaxDenom) {
242  CAmount nValue = ZerocoinDenominationToAmount(denom);
243  do {
244  if ((nRemainingValue > nValue) && (mapOfDenomsUsed.at(denom) < mapOfDenomsHeld.at(denom))) {
245  mapOfDenomsUsed.at(denom)++;
246  nRemainingValue -= nValue;
247  AmountUsed += nValue;
248  nCoinCount++;
249  }
250  } while ((nRemainingValue > nValue) && (mapOfDenomsUsed.at(denom) < mapOfDenomsHeld.at(denom)));
251  }
252  }
253 
254  // Now work way back up from the bottom filling in with the denom that we have that is just
255  // bigger than the remaining amount
256  // Shouldn't need more than one coin here?
257  for (const auto& denom : zerocoinDenomList) {
258  CAmount nValue = ZerocoinDenominationToAmount(denom);
259  if ((nValue > nRemainingValue) && (mapOfDenomsUsed.at(denom) < mapOfDenomsHeld.at(denom))) {
260  mapOfDenomsUsed.at(denom)++;
261  nRemainingValue -= nValue;
262  AmountUsed += nValue;
263  nCoinCount++;
264  }
265  if (nRemainingValue < 0) break;
266  }
267 
268  // This can still result in a case where you've used an extra spend than needed.
269  // e.g Spend of 26, while having 1*5 + 4*10
270  // First stage may be 2*10+5 (i.e < 26)
271  // Second stage can be 3*10+5 (no more fives, so add a 10)
272  // So 5 is no longer needed and will become change also
273 
274  CAmount nAltChangeAmount = AmountUsed - nValueTarget;
275  std::map<CoinDenomination, CAmount> mapAltChange = getChange(nAltChangeAmount);
276 
277  // Check if there is overlap between change and spend denominations
278  // And if so, remove those that overlap
279  for (const auto& denom : zerocoinDenomList) {
280  do {
281  if (mapAltChange.at(denom) && mapOfDenomsUsed.at(denom)) {
282  mapOfDenomsUsed.at(denom)--;
283  mapAltChange.at(denom)--;
284  nCoinCount--;
285  CAmount nValue = ZerocoinDenominationToAmount(denom);
286  AmountUsed -= nValue;
287  }
288  } while (mapAltChange.at(denom) && mapOfDenomsUsed.at(denom));
289  }
290 
291  // Still possible to have wrong mix. So meet exact amount found above - with least number of coins
292  mapOfDenomsUsed = getSpendCoins(AmountUsed, mapOfDenomsHeld);
293  nCoinCount = getNumberOfCoinsUsed(mapOfDenomsUsed);
294 
295  // Re-calculate change
296  nAltChangeAmount = AmountUsed - nValueTarget;
297  mapAltChange = getChange(nAltChangeAmount);
298  int AltChangeCount = getNumberOfCoinsUsed(mapAltChange);
299 
300  // Alternative method yields less mints and is less than MaxNumberOfSpends if true
301  if ((AltChangeCount < nChangeCount) && (nCoinCount <= nMaxNumberOfSpends)) {
302  return AltChangeCount;
303  } else {
304  // if we don't meet above go back to what we started with
305  mapOfDenomsUsed = savedMapOfDenomsUsed;
306  return nChangeCount;
307  }
308 }

311 // -------------------------------------------------------------------------------------------------------
312 // Couldn't meet amount exactly, will need to generate change
313 // returning with a 0 means either too many spends or no change
314 // Latter should never happen since we should only get here if exact is not possible
315 // -------------------------------------------------------------------------------------------------------
316 int calculateChange(
317  int nMaxNumberOfSpends,
318  bool fMinimizeChange,
319  const CAmount nValueTarget,
320  const std::map<CoinDenomination, CAmount>& mapOfDenomsHeld,
321  std::map<CoinDenomination, CAmount>& mapOfDenomsUsed)
322 {
323  CoinDenomination minDenomOverTarget = ZQ_ERROR;
324  // Initialize
325  mapOfDenomsUsed.clear();
326  for (const auto& denom : zerocoinDenomList)
327  mapOfDenomsUsed.insert(std::pair<CoinDenomination, CAmount>(denom, 0));
328 
329  for (const auto& denom : zerocoinDenomList) {
330  if (nValueTarget < ZerocoinDenominationToAmount(denom) && mapOfDenomsHeld.at(denom)) {
331  minDenomOverTarget = denom;
332  break;
333  }
334  }
335  // OK so if != ZQ_ERROR we have a solution using 1 coin
336  if (minDenomOverTarget != ZQ_ERROR) {
337  mapOfDenomsUsed.at(minDenomOverTarget) = 1;
338 
339  // Now find out # of coins in change
340  CAmount nChangeAmount = ZerocoinDenominationToAmount(minDenomOverTarget) - nValueTarget;
341  std::map<CoinDenomination, CAmount> mapChange = getChange(nChangeAmount);
342  int nChangeCount = getNumberOfCoinsUsed(mapChange);
343 
344  if (fMinimizeChange) {
345  CoinDenomination nextToMaxDenom = getNextLowerDenomHeld(minDenomOverTarget, mapOfDenomsHeld);
346  int newChangeCount = minimizeChange(nMaxNumberOfSpends, nChangeCount,
347  nextToMaxDenom, nValueTarget,
348  mapOfDenomsHeld, mapOfDenomsUsed);
349 
350  // Alternative method yields less mints and is less than MaxNumberOfSpends if true
351  if (newChangeCount < nChangeCount) return newChangeCount;
352 
353  // Reclear
354  for (const auto& denom : zerocoinDenomList)
355  mapOfDenomsUsed.at(denom) = 0;
356  // Then reset as before previous clearing
357  mapOfDenomsUsed.at(minDenomOverTarget) = 1;
358  }
359 
360  return nChangeCount;
361 
362  } else {
363  // Try to meet a different way
364  for (const auto& denom : zerocoinDenomList)
365  mapOfDenomsUsed.at(denom) = 0;
366  CAmount nRemainingValue = nValueTarget;
367  int nCoinCount = 0;
368  CAmount AmountUsed = 0;
369  for (const auto& denom : reverse_iterate(zerocoinDenomList)) {
370  CAmount nValue = ZerocoinDenominationToAmount(denom);
371  do {
372  if (mapOfDenomsHeld.at(denom) && nRemainingValue > 0) {
373  mapOfDenomsUsed.at(denom)++;
374  AmountUsed += nValue;
375  nRemainingValue -= nValue;
376  nCoinCount++;
377  }
378  } while ((nRemainingValue > 0) && (mapOfDenomsUsed.at(denom) < mapOfDenomsHeld.at(denom)));
379  if (nRemainingValue < 0) break;
380  }
381 
382  CAmount nChangeAmount = AmountUsed - nValueTarget;
383  std::map<CoinDenomination, CAmount> mapChange = getChange(nChangeAmount);
384  int nMaxChangeCount = getNumberOfCoinsUsed(mapChange);
385 
386  // Instead get max Denom held
387  CoinDenomination maxDenomHeld = getMaxDenomHeld(mapOfDenomsHeld);
388 
389  // Assign for size (only)
390  std::map<CoinDenomination, CAmount> mapOfMinDenomsUsed = mapOfDenomsUsed;
391 
392  int nChangeCount = minimizeChange(nMaxNumberOfSpends, nMaxChangeCount,
393  maxDenomHeld, nValueTarget,
394  mapOfDenomsHeld, mapOfMinDenomsUsed);
395 
396  int nNumSpends = getNumberOfCoinsUsed(mapOfMinDenomsUsed);
397 
398  if (!fMinimizeChange && (nCoinCount < nNumSpends)) {
399  return nMaxChangeCount;
400  }
401 
402  mapOfDenomsUsed = mapOfMinDenomsUsed;
403  return nChangeCount;
404  }
405 }
