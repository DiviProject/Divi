116 void ReceiveCoinsDialog::on_receiveButton_clicked()
117 {
118  if (!model || !model->getOptionsModel() || !model->getAddressTableModel() || !model->getRecentRequestsTableModel())
119  return;
120 
121  QString address;
122  QString label = ui->reqLabel->text();
123  if (ui->reuseAddress->isChecked()) {
124  /* Choose existing receiving address */
125  AddressBookPage dlg(AddressBookPage::ForSelection, AddressBookPage::ReceivingTab, this);
126  dlg.setModel(model->getAddressTableModel());
127  if (dlg.exec()) {
128  address = dlg.getReturnValue();
129  if (label.isEmpty()) /* If no label provided, use the previously used label */
130  {
131  label = model->getAddressTableModel()->labelForAddress(address);
132  }
133  } else {
134  return;
135  }
136  } else {
137  /* Generate new receiving address */
138  address = model->getAddressTableModel()->addRow(AddressTableModel::Receive, label, "");
139  }
140  SendCoinsRecipient info(address, label,
141  ui->reqAmount->value(), ui->reqMessage->text());
142  ReceiveRequestDialog* dialog = new ReceiveRequestDialog(this);
143  dialog->setAttribute(Qt::WA_DeleteOnClose);
144  dialog->setModel(model->getOptionsModel());
145  dialog->setInfo(info);
146  dialog->show();
147  clear();
148 
149  /* Store request for later reference */
150  model->getRecentRequestsTableModel()->addNewRequest(info);
151 }
