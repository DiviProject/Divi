169 unsigned int CScript::GetSigOpCount(bool fAccurate) const
170 {
171  unsigned int n = 0;
172  const_iterator pc = begin();
173  opcodetype lastOpcode = OP_INVALIDOPCODE;
174  while (pc < end())
175  {
176  opcodetype opcode;
177  if (!GetOp(pc, opcode))
178  break;
179  if (opcode == OP_CHECKSIG || opcode == OP_CHECKSIGVERIFY)
180  n++;
181  else if (opcode == OP_CHECKMULTISIG || opcode == OP_CHECKMULTISIGVERIFY)
182  {
183  if (fAccurate && lastOpcode >= OP_1 && lastOpcode <= OP_16)
184  n += DecodeOP_N(lastOpcode);
185  else
186  n += 20;
187  }
188  lastOpcode = opcode;
189  }
190  return n;
191 }

193 unsigned int CScript::GetSigOpCount(const CScript& scriptSig) const
194 {
195  if (!IsPayToScriptHash())
196  return GetSigOpCount(true);
197 
198  // This is a pay-to-script-hash scriptPubKey;
199  // get the last item that the scriptSig
200  // pushes onto the stack:
201  const_iterator pc = scriptSig.begin();
202  vector<unsigned char> data;
203  while (pc < scriptSig.end())
204  {
205  opcodetype opcode;
206  if (!scriptSig.GetOp(pc, opcode, data))
207  return 0;
208  if (opcode > OP_16)
209  return 0;
210  }
211 
213  CScript subscript(data.begin(), data.end());
214  return subscript.GetSigOpCount(true);
215 }

217 bool CScript::IsNormalPaymentScript() const
218 {
219  if(this->size() != 25) return false;
220 
221  std::string str;

227  GetOp(pc, opcode);

230  else if(i == 1 && opcode != OP_HASH160) return false;

263 bool CScript::IsPushOnly(const_iterator pc) const
264 {
265  while (pc < end())
266  {
267  opcodetype opcode;
268  if (!GetOp(pc, opcode))
269  return false;
270  // Note that IsPushOnly() *does* consider OP_RESERVED to be a
271  // push-type opcode, however execution of OP_RESERVED fails, so
272  // it's not relevant to P2SH/BIP62 as the scriptSig would fail prior to
273  // the P2SH special validation code being executed.
274  if (opcode > OP_16)
275  return false;
276  }
277  return true;
278 }
