295 void CWalletDB::ListAccountCreditDebit(const string& strAccount, list<CAccountingEntry>& entries)
296 {
297  bool fAllAccounts = (strAccount == "*");
298 
299  Dbc* pcursor = GetCursor();
300  if (!pcursor)
301  throw runtime_error("CWalletDB::ListAccountCreditDebit() : cannot create DB cursor");
302  unsigned int fFlags = DB_SET_RANGE;
303  while (true) {
304  // Read next record
305  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
306  if (fFlags == DB_SET_RANGE)
307  ssKey << std::make_pair(std::string("acentry"), std::make_pair((fAllAccounts ? string("") : strAccount), uint64_t(0)));
308  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
309  int ret = ReadAtCursor(pcursor, ssKey, ssValue, fFlags);
310  fFlags = DB_NEXT;
311  if (ret == DB_NOTFOUND)
312  break;
313  else if (ret != 0) {
314  pcursor->close();
315  throw runtime_error("CWalletDB::ListAccountCreditDebit() : error scanning DB");
316  }
317 
318  // Unserialize
319  string strType;
320  ssKey >> strType;
321  if (strType != "acentry")
322  break;
323  CAccountingEntry acentry;
324  ssKey >> acentry.strAccount;
325  if (!fAllAccounts && acentry.strAccount != strAccount)
326  break;
327 
328  ssValue >> acentry;
329  ssKey >> acentry.nEntryNo;
330  entries.push_back(acentry);
331  }
332 
333  pcursor->close();
334 }

336 DBErrors CWalletDB::ReorderTransactions(CWallet* pwallet)
337 {
338  LOCK(pwallet->cs_wallet);
339  // Old wallets didn't have any defined order for transactions
340  // Probably a bad idea to change the output of this
341 
342  // First: get all CWalletTx and CAccountingEntry into a sorted-by-time multimap.
343  typedef pair<CWalletTx*, CAccountingEntry*> TxPair;
344  typedef multimap<int64_t, TxPair> TxItems;
345  TxItems txByTime;
346 
347  for (map<uint256, CWalletTx>::iterator it = pwallet->mapWallet.begin(); it != pwallet->mapWallet.end(); ++it) {
348  CWalletTx* wtx = &((*it).second);
349  txByTime.insert(make_pair(wtx->nTimeReceived, TxPair(wtx, (CAccountingEntry*)0)));
350  }
351  list<CAccountingEntry> acentries;
352  ListAccountCreditDebit("", acentries);
353  BOOST_FOREACH (CAccountingEntry& entry, acentries) {
354  txByTime.insert(make_pair(entry.nTime, TxPair((CWalletTx*)0, &entry)));
355  }
356 
357  int64_t& nOrderPosNext = pwallet->nOrderPosNext;
358  nOrderPosNext = 0;
359  std::vector<int64_t> nOrderPosOffsets;
360  for (TxItems::iterator it = txByTime.begin(); it != txByTime.end(); ++it) {
361  CWalletTx* const pwtx = (*it).second.first;
362  CAccountingEntry* const pacentry = (*it).second.second;
363  int64_t& nOrderPos = (pwtx != 0) ? pwtx->nOrderPos : pacentry->nOrderPos;
364 
365  if (nOrderPos == -1) {
366  nOrderPos = nOrderPosNext++;
367  nOrderPosOffsets.push_back(nOrderPos);
368 
369  if (pwtx) {
370  if (!WriteTx(pwtx->GetHash(), *pwtx))
371  return DB_LOAD_FAIL;
372  } else if (!WriteAccountingEntry(pacentry->nEntryNo, *pacentry))
373  return DB_LOAD_FAIL;
374  } else {
375  int64_t nOrderPosOff = 0;
376  BOOST_FOREACH (const int64_t& nOffsetStart, nOrderPosOffsets) {
377  if (nOrderPos >= nOffsetStart)
378  ++nOrderPosOff;
379  }
380  nOrderPos += nOrderPosOff;
381  nOrderPosNext = std::max(nOrderPosNext, nOrderPos + 1);
382 
383  if (!nOrderPosOff)
384  continue;
385 
386  // Since we're changing the order, write it back
387  if (pwtx) {
388  if (!WriteTx(pwtx->GetHash(), *pwtx))
389  return DB_LOAD_FAIL;
390  } else if (!WriteAccountingEntry(pacentry->nEntryNo, *pacentry))
391  return DB_LOAD_FAIL;
392  }
393  }
394  WriteOrderPosNext(nOrderPosNext);
395 
396  return DB_LOAD_OK;
397 }

419 bool ReadKeyValue(CWallet* pwallet, CDataStream& ssKey, CDataStream& ssValue, CWalletScanState& wss, string& strType, string& strErr)
420 {
421  try {
422  // Unserialize
423  // Taking advantage of the fact that pair serialization
424  // is just the two items serialized one after the other
425  ssKey >> strType;
426  if (strType == "name") {
427  string strAddress;
428  ssKey >> strAddress;
429  ssValue >> pwallet->mapAddressBook[CBitcoinAddress(strAddress).Get()].name;
430  } else if (strType == "purpose") {
431  string strAddress;
432  ssKey >> strAddress;
433  ssValue >> pwallet->mapAddressBook[CBitcoinAddress(strAddress).Get()].purpose;
434  } else if (strType == "tx") {
435  uint256 hash;
436  ssKey >> hash;
437  CWalletTx wtx;
438  ssValue >> wtx;
439  CValidationState state;
440  // false because there is no reason to go through the zerocoin checks for our own wallet
441  if (!(CheckTransaction(wtx, false, false, state) && (wtx.GetHash() == hash) && state.IsValid()))
442  return false;
443 
444  // Undo serialize changes in 31600
445  if (31404 <= wtx.fTimeReceivedIsTxTime && wtx.fTimeReceivedIsTxTime <= 31703) {
446  if (!ssValue.empty()) {
447  char fTmp;
448  char fUnused;
449  ssValue >> fTmp >> fUnused >> wtx.strFromAccount;
450  strErr = strprintf("LoadWallet() upgrading tx ver=%d %d '%s' %s",
451  wtx.fTimeReceivedIsTxTime, fTmp, wtx.strFromAccount, hash.ToString());
452  wtx.fTimeReceivedIsTxTime = fTmp;
453  } else {
454  strErr = strprintf("LoadWallet() repairing tx ver=%d %s", wtx.fTimeReceivedIsTxTime, hash.ToString());
455  wtx.fTimeReceivedIsTxTime = 0;
456  }
457  wss.vWalletUpgrade.push_back(hash);
458  }
459 
460  if (wtx.nOrderPos == -1)
461  wss.fAnyUnordered = true;
462 
463  pwallet->AddToWallet(wtx, true);
464  } else if (strType == "acentry") {
465  string strAccount;
466  ssKey >> strAccount;
467  uint64_t nNumber;
468  ssKey >> nNumber;
469  if (nNumber > nAccountingEntryNumber)
470  nAccountingEntryNumber = nNumber;
471 
472  if (!wss.fAnyUnordered) {
473  CAccountingEntry acentry;
474  ssValue >> acentry;
475  if (acentry.nOrderPos == -1)
476  wss.fAnyUnordered = true;
477  }
478  } else if (strType == "watchs") {
479  CScript script;
480  ssKey >> script;
481  char fYes;
482  ssValue >> fYes;
483  if (fYes == '1')
484  pwallet->LoadWatchOnly(script);
485 
486  // Watch-only addresses have no birthday information for now,
487  // so set the wallet birthday to the beginning of time.
488  pwallet->nTimeFirstKey = 1;
489  } else if (strType == "multisig") {
490  CScript script;
491  ssKey >> script;
492  char fYes;
493  ssValue >> fYes;
494  if (fYes == '1')
495  pwallet->LoadMultiSig(script);
496 
497  // MultiSig addresses have no birthday information for now,
498  // so set the wallet birthday to the beginning of time.
499  pwallet->nTimeFirstKey = 1;
500  } else if (strType == "key" || strType == "wkey") {
501  CPubKey vchPubKey;
502  ssKey >> vchPubKey;
503  if (!vchPubKey.IsValid()) {
504  strErr = "Error reading wallet database: CPubKey corrupt";
505  return false;
506  }
507  CKey key;
508  CPrivKey pkey;
509  uint256 hash = 0;
510 
511  if (strType == "key") {
512  wss.nKeys++;
513  ssValue >> pkey;
514  } else {
515  CWalletKey wkey;
516  ssValue >> wkey;
517  pkey = wkey.vchPrivKey;
518  }
519 
520  // Old wallets store keys as "key" [pubkey] => [privkey]
521  // ... which was slow for wallets with lots of keys, because the public key is re-derived from the private key
522  // using EC operations as a checksum.
523  // Newer wallets store keys as "key"[pubkey] => [privkey][hash(pubkey,privkey)], which is much faster while
524  // remaining backwards-compatible.
525  try {
526  ssValue >> hash;
527  } catch (...) {
528  }
529 
530  bool fSkipCheck = false;
531 
532  if (hash != 0) {
533  // hash pubkey/privkey to accelerate wallet load
534  std::vector<unsigned char> vchKey;
535  vchKey.reserve(vchPubKey.size() + pkey.size());
536  vchKey.insert(vchKey.end(), vchPubKey.begin(), vchPubKey.end());
537  vchKey.insert(vchKey.end(), pkey.begin(), pkey.end());
538 
539  if (Hash(vchKey.begin(), vchKey.end()) != hash) {
540  strErr = "Error reading wallet database: CPubKey/CPrivKey corrupt";
541  return false;
542  }
543 
544  fSkipCheck = true;
545  }
546 
547  if (!key.Load(pkey, vchPubKey, fSkipCheck)) {
548  strErr = "Error reading wallet database: CPrivKey corrupt";
549  return false;
550  }
551  if (!pwallet->LoadKey(key, vchPubKey)) {
552  strErr = "Error reading wallet database: LoadKey failed";
553  return false;
554  }
555  } else if (strType == "mkey") {
556  unsigned int nID;
557  ssKey >> nID;
558  CMasterKey kMasterKey;
559  ssValue >> kMasterKey;
560  if (pwallet->mapMasterKeys.count(nID) != 0) {
561  strErr = strprintf("Error reading wallet database: duplicate CMasterKey id %u", nID);
562  return false;
563  }
564  pwallet->mapMasterKeys[nID] = kMasterKey;
565  if (pwallet->nMasterKeyMaxID < nID)
566  pwallet->nMasterKeyMaxID = nID;
567  } else if (strType == "ckey") {
568  vector<unsigned char> vchPubKey;
569  ssKey >> vchPubKey;
570  vector<unsigned char> vchPrivKey;
571  ssValue >> vchPrivKey;
572  wss.nCKeys++;
573 
574  if (!pwallet->LoadCryptedKey(vchPubKey, vchPrivKey)) {
575  strErr = "Error reading wallet database: LoadCryptedKey failed";
576  return false;
577  }
578  wss.fIsEncrypted = true;
579  } else if (strType == "keymeta") {
580  CPubKey vchPubKey;
581  ssKey >> vchPubKey;
582  CKeyMetadata keyMeta;
583  ssValue >> keyMeta;
584  wss.nKeyMeta++;
585 
586  pwallet->LoadKeyMetadata(vchPubKey, keyMeta);
587 
588  // find earliest key creation time, as wallet birthday
589  if (!pwallet->nTimeFirstKey ||
590  (keyMeta.nCreateTime < pwallet->nTimeFirstKey))
591  pwallet->nTimeFirstKey = keyMeta.nCreateTime;
592  } else if (strType == "defaultkey") {
593  ssValue >> pwallet->vchDefaultKey;
594  } else if (strType == "pool") {
595  int64_t nIndex;
596  ssKey >> nIndex;
597  CKeyPool keypool;
598  ssValue >> keypool;
599  pwallet->setKeyPool.insert(nIndex);
600 
601  // If no metadata exists yet, create a default with the pool key's
602  // creation time. Note that this may be overwritten by actually
603  // stored metadata for that key later, which is fine.
604  CKeyID keyid = keypool.vchPubKey.GetID();
605  if (pwallet->mapKeyMetadata.count(keyid) == 0)
606  pwallet->mapKeyMetadata[keyid] = CKeyMetadata(keypool.nTime);
607  } else if (strType == "version") {
608  ssValue >> wss.nFileVersion;
609  if (wss.nFileVersion == 10300)
610  wss.nFileVersion = 300;
611  } else if (strType == "cscript") {
612  uint160 hash;
613  ssKey >> hash;
614  CScript script;
615  ssValue >> script;
616  if (!pwallet->LoadCScript(script)) {
617  strErr = "Error reading wallet database: LoadCScript failed";
618  return false;
619  }
620  } else if (strType == "orderposnext") {
621  ssValue >> pwallet->nOrderPosNext;
622  } else if (strType == "stakeSplitThreshold") //presstab HyperStake
623  {
624  ssValue >> pwallet->nStakeSplitThreshold;
625  } else if (strType == "multisend") //presstab HyperStake
626  {
627  unsigned int i;
628  ssKey >> i;
629  std::pair<std::string, int> pMultiSend;
630  ssValue >> pMultiSend;
631  if (CBitcoinAddress(pMultiSend.first).IsValid()) {
632  pwallet->vMultiSend.push_back(pMultiSend);
633  }
634  } else if (strType == "msettingsv2") //presstab HyperStake
635  {

673 DBErrors CWalletDB::LoadWallet(CWallet* pwallet)
674 {
675  pwallet->vchDefaultKey = CPubKey();
676  CWalletScanState wss;
677  bool fNoncriticalErrors = false;
678  DBErrors result = DB_LOAD_OK;
679 
680  try {
681  LOCK(pwallet->cs_wallet);
682  int nMinVersion = 0;
683  if (Read((string) "minversion", nMinVersion)) {
684  if (nMinVersion > CLIENT_VERSION)
685  return DB_TOO_NEW;
686  pwallet->LoadMinVersion(nMinVersion);
687  }
688 
689  // Get cursor
690  Dbc* pcursor = GetCursor();
691  if (!pcursor) {
692  LogPrintf("Error getting wallet database cursor\n");
693  return DB_CORRUPT;
694  }
695 
696  while (true) {
697  // Read next record
698  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
699  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
700  int ret = ReadAtCursor(pcursor, ssKey, ssValue);
701  if (ret == DB_NOTFOUND)
702  break;
703  else if (ret != 0) {
704  LogPrintf("Error reading next record from wallet database\n");
705  return DB_CORRUPT;
706  }
707 
708  // Try to be tolerant of single corrupt records:
709  string strType, strErr;
710  if (!ReadKeyValue(pwallet, ssKey, ssValue, wss, strType, strErr)) {
711  // losing keys is considered a catastrophic error, anything else
712  // we assume the user can live with:
713  if (IsKeyType(strType))
714  result = DB_CORRUPT;
715  else {
716  // Leave other errors alone, if we try to fix them we might make things worse.
717  fNoncriticalErrors = true; // ... but do warn the user there is something wrong.
718  if (strType == "tx")
719  // Rescan if there is a bad transaction record:
720  SoftSetBoolArg("-rescan", true);
721  }
722  }
723  if (!strErr.empty())
724  LogPrintf("%s\n", strErr);
725  }
726  pcursor->close();
727  } catch (boost::thread_interrupted) {
728  throw;
729  } catch (...) {
730  result = DB_CORRUPT;
731  }
732 
733  if (fNoncriticalErrors && result == DB_LOAD_OK)
734  result = DB_NONCRITICAL_ERROR;
735 
736  // Any wallet corruption at all: skip any rewriting or
737  // upgrading, we don't want to make it worse.
738  if (result != DB_LOAD_OK)
739  return result;
740 
741  LogPrintf("nFileVersion = %d\n", wss.nFileVersion);
742 
743  LogPrintf("Keys: %u plaintext, %u encrypted, %u w/ metadata, %u total\n",
744  wss.nKeys, wss.nCKeys, wss.nKeyMeta, wss.nKeys + wss.nCKeys);
745 
746  // nTimeFirstKey is only reliable if all keys have metadata
747  if ((wss.nKeys + wss.nCKeys) != wss.nKeyMeta)
748  pwallet->nTimeFirstKey = 1; // 0 would be considered 'no value'
749 
750  BOOST_FOREACH (uint256 hash, wss.vWalletUpgrade)
751  WriteTx(hash, pwallet->mapWallet[hash]);
752 
753  // Rewrite encrypted wallets of versions 0.4.0 and 0.5.0rc:
754  if (wss.fIsEncrypted && (wss.nFileVersion == 40000 || wss.nFileVersion == 50000))
755  return DB_NEED_REWRITE;
756 
757  if (wss.nFileVersion < CLIENT_VERSION) // Update
758  WriteVersion(CLIENT_VERSION);
759 
760  if (wss.fAnyUnordered)
761  result = ReorderTransactions(pwallet);
762 
763  return result;
764 }

960  }

963  bool allOK = dbenv.Salvage(newFilename, true, salvagedData);

974  "main", // Logical db name
975  DB_BTREE, // Database type

977  0);

980  return false;

983  CWalletScanState wss;
984

987  if (fOnlyKeys) {

1004  fSuccess = false;
1005  }
