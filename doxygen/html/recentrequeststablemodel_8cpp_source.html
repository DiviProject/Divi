51 QVariant RecentRequestsTableModel::data(const QModelIndex& index, int role) const
52 {
53  if (!index.isValid() || index.row() >= list.length())
54  return QVariant();
55 
56  const RecentRequestEntry* rec = &list[index.row()];
57 
58  if (role == Qt::DisplayRole || role == Qt::EditRole) {
59  switch (index.column()) {
60  case Date:
61  return GUIUtil::dateTimeStr(rec->date);
62  case Label:
63  if (rec->recipient.label.isEmpty() && role == Qt::DisplayRole) {
64  return tr("(no label)");
65  } else {
66  return rec->recipient.label;
67  }
68  case Message:
69  if (rec->recipient.message.isEmpty() && role == Qt::DisplayRole) {
70  return tr("(no message)");
71  } else {
72  return rec->recipient.message;
73  }
74  case Amount:
75  if (rec->recipient.amount == 0 && role == Qt::DisplayRole)
76  return tr("(no amount)");
77  else if (role == Qt::EditRole)
78  return BitcoinUnits::format(walletModel->getOptionsModel()->getDisplayUnit(), rec->recipient.amount, false, BitcoinUnits::separatorNever);
79  else
80  return BitcoinUnits::format(walletModel->getOptionsModel()->getDisplayUnit(), rec->recipient.amount);
81  }
82  } else if (role == Qt::TextAlignmentRole) {
83  if (index.column() == Amount)
84  return (int)(Qt::AlignRight | Qt::AlignVCenter);
85  }
86  return QVariant();
87 }

128 bool RecentRequestsTableModel::removeRows(int row, int count, const QModelIndex& parent)
129 {
130  Q_UNUSED(parent);
131 
132  if (count > 0 && row >= 0 && (row + count) <= list.size()) {
133  const RecentRequestEntry* rec;
134  for (int i = 0; i < count; ++i) {
135  rec = &list[row + i];
136  if (!walletModel->saveReceiveRequest(rec->recipient.address.toStdString(), rec->id, ""))
137  return false;
138  }
139 
140  beginRemoveRows(parent, row, row + count - 1);
141  list.erase(list.begin() + row, list.begin() + row + count);
142  endRemoveRows();
143  return true;
144  } else {
145  return false;
146  }

208 bool RecentRequestEntryLessThan::operator()(RecentRequestEntry& left, RecentRequestEntry& right) const
209 {
210  RecentRequestEntry* pLeft = &left;
211  RecentRequestEntry* pRight = &right;
212  if (order == Qt::DescendingOrder)
213  std::swap(pLeft, pRight);
214 
215  switch (column) {
216  case RecentRequestsTableModel::Date:
217  return pLeft->date.toTime_t() < pRight->date.toTime_t();
218  case RecentRequestsTableModel::Label:
219  return pLeft->recipient.label < pRight->recipient.label;
220  case RecentRequestsTableModel::Message:
221  return pLeft->recipient.message < pRight->recipient.message;
222  case RecentRequestsTableModel::Amount:
223  return pLeft->recipient.amount < pRight->recipient.amount;
224  default:
225  return pLeft->id < pRight->id;
226  }
