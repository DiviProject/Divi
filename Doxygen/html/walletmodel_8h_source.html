44  // If from an insecure payment request, this is used for storing
45  // the addresses, e.g. address-A<br />address-B<br />address-C.

67  template <typename Stream, typename Operation>
68  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion)
69  {
70  std::string sAddress = address.toStdString();
71  std::string sLabel = label.toStdString();
72  std::string sMessage = message.toStdString();
73  std::string sPaymentRequest;
74  if (!ser_action.ForRead() && paymentRequest.IsInitialized())
75  paymentRequest.SerializeToString(&sPaymentRequest);
76  std::string sAuthenticatedMerchant = authenticatedMerchant.toStdString();
77 
78  READWRITE(this->nVersion);
79  nVersion = this->nVersion;
80  READWRITE(sAddress);
81  READWRITE(sLabel);
82  READWRITE(amount);
83  READWRITE(sMessage);
84  READWRITE(sPaymentRequest);
85  READWRITE(sAuthenticatedMerchant);
86 
87  if (ser_action.ForRead()) {
88  address = QString::fromStdString(sAddress);
89  label = QString::fromStdString(sLabel);
90  message = QString::fromStdString(sMessage);
91  if (!sPaymentRequest.empty())
92  paymentRequest.parse(QByteArray::fromRawData(sPaymentRequest.data(), sPaymentRequest.size()));
93  authenticatedMerchant = QString::fromStdString(sAuthenticatedMerchant);
94  }
95  }
