93  // If setting doesn't exist create it with defaults.

178 // read QSettings values and return them
179 QVariant OptionsModel::data(const QModelIndex& index, int role) const
180 {
181  if (role == Qt::EditRole) {
182  QSettings settings;
183  switch (index.row()) {
184  case StartAtStartup:
185  return GUIUtil::GetStartOnSystemStartup();
186  case MinimizeToTray:
187  return fMinimizeToTray;
188  case MapPortUPnP:
189 #ifdef USE_UPNP
190  return settings.value("fUseUPnP");
191 #else
192  return false;
193 #endif
194  case MinimizeOnClose:
195  return fMinimizeOnClose;
196 
197  // default proxy
198  case ProxyUse:
199  return settings.value("fUseProxy", false);
200  case ProxyIP: {
201  // contains IP at index 0 and port at index 1
202  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
203  return strlIpPort.at(0);
204  }
205  case ProxyPort: {
206  // contains IP at index 0 and port at index 1
207  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
208  return strlIpPort.at(1);
209  }
210 
211 #ifdef ENABLE_WALLET
212  case SpendZeroConfChange:
213  return settings.value("bSpendZeroConfChange");
214  case ShowMasternodesTab:
215  return settings.value("fShowMasternodesTab");
216 #endif
217  case DisplayUnit:
218  return nDisplayUnit;
219  case ThirdPartyTxUrls:
220  return strThirdPartyTxUrls;
221  case Digits:
222  return settings.value("digits");
223  case Theme:
224  return settings.value("theme");
225  case Language:
226  return settings.value("language");
227  case CoinControlFeatures:
228  return fCoinControlFeatures;
229  case DatabaseCache:
230  return settings.value("nDatabaseCache");
231  case ThreadsScriptVerif:
232  return settings.value("nThreadsScriptVerif");
233  case ZeromintPercentage:
234  return QVariant(nZeromintPercentage);
235  case ZeromintPrefDenom:
236  return QVariant(nPreferredDenom);
237  case AnonymizeDiviAmount:
238  return QVariant(nAnonymizeDiviAmount);
239  case Listen:
240  return settings.value("fListen");
241  default:
242  return QVariant();
243  }
244  }
245  return QVariant();
246 }

249 bool OptionsModel::setData(const QModelIndex& index, const QVariant& value, int role)
250 {
251  bool successful = true; /* set to false on parse error */
252  if (role == Qt::EditRole) {
253  QSettings settings;
254  switch (index.row()) {
255  case StartAtStartup:
256  successful = GUIUtil::SetStartOnSystemStartup(value.toBool());
257  break;
258  case MinimizeToTray:
259  fMinimizeToTray = value.toBool();
260  settings.setValue("fMinimizeToTray", fMinimizeToTray);
261  break;
262  case MapPortUPnP: // core option - can be changed on-the-fly
263  settings.setValue("fUseUPnP", value.toBool());
264  MapPort(value.toBool());
265  break;
266  case MinimizeOnClose:
267  fMinimizeOnClose = value.toBool();
268  settings.setValue("fMinimizeOnClose", fMinimizeOnClose);
269  break;
270 
271  // default proxy
272  case ProxyUse:
273  if (settings.value("fUseProxy") != value) {
274  settings.setValue("fUseProxy", value.toBool());
275  setRestartRequired(true);
276  }
277  break;
278  case ProxyIP: {
279  // contains current IP at index 0 and current port at index 1
280  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
281  // if that key doesn't exist or has a changed IP
282  if (!settings.contains("addrProxy") || strlIpPort.at(0) != value.toString()) {
283  // construct new value from new IP and current port
284  QString strNewValue = value.toString() + ":" + strlIpPort.at(1);
285  settings.setValue("addrProxy", strNewValue);
286  setRestartRequired(true);
287  }
288  } break;
289  case ProxyPort: {
290  // contains current IP at index 0 and current port at index 1
291  QStringList strlIpPort = settings.value("addrProxy").toString().split(":", QString::SkipEmptyParts);
292  // if that key doesn't exist or has a changed port
293  if (!settings.contains("addrProxy") || strlIpPort.at(1) != value.toString()) {
294  // construct new value from current IP and new port
295  QString strNewValue = strlIpPort.at(0) + ":" + value.toString();
296  settings.setValue("addrProxy", strNewValue);
297  setRestartRequired(true);
298  }
299  } break;
300 #ifdef ENABLE_WALLET
301  case SpendZeroConfChange:
302  if (settings.value("bSpendZeroConfChange") != value) {
303  settings.setValue("bSpendZeroConfChange", value);
304  setRestartRequired(true);
305  }
306  break;
307  case ShowMasternodesTab:
308  if (settings.value("fShowMasternodesTab") != value) {
309  settings.setValue("fShowMasternodesTab", value);
310  setRestartRequired(true);
311  }
312  break;
313 #endif
314  case DisplayUnit:
315  setDisplayUnit(value);
316  break;
317  case ThirdPartyTxUrls:
318  if (strThirdPartyTxUrls != value.toString()) {
319  strThirdPartyTxUrls = value.toString();
320  settings.setValue("strThirdPartyTxUrls", strThirdPartyTxUrls);
321  setRestartRequired(true);
322  }
323  break;
324  case Digits:
325  if (settings.value("digits") != value) {
326  settings.setValue("digits", value);
327  setRestartRequired(true);
328  }
329  break;
330  case Theme:
331  if (settings.value("theme") != value) {
332  settings.setValue("theme", value);
333  setRestartRequired(true);
334  }
335  break;
336  case Language:
337  if (settings.value("language") != value) {
338  settings.setValue("language", value);
339  setRestartRequired(true);
340  }
341  break;
342  case ZeromintPercentage:
343  nZeromintPercentage = value.toInt();
344  settings.setValue("nZeromintPercentage", nZeromintPercentage);
345  emit zeromintPercentageChanged(nZeromintPercentage);
346  break;
347  case ZeromintPrefDenom:
348  nPreferredDenom = value.toInt();
349  settings.setValue("nPreferredDenom", nPreferredDenom);
350  emit preferredDenomChanged(nPreferredDenom);
351  break;
352 
353  case AnonymizeDiviAmount:
354  nAnonymizeDiviAmount = value.toInt();
355  settings.setValue("nAnonymizeDiviAmount", nAnonymizeDiviAmount);
356  emit anonymizeDiviAmountChanged(nAnonymizeDiviAmount);
357  break;
358  case CoinControlFeatures:
359  fCoinControlFeatures = value.toBool();
360  settings.setValue("fCoinControlFeatures", fCoinControlFeatures);
361  emit coinControlFeaturesChanged(fCoinControlFeatures);
362  break;
363  case DatabaseCache:
364  if (settings.value("nDatabaseCache") != value) {
365  settings.setValue("nDatabaseCache", value);
366  setRestartRequired(true);
367  }
368  break;
369  case ThreadsScriptVerif:
370  if (settings.value("nThreadsScriptVerif") != value) {
371  settings.setValue("nThreadsScriptVerif", value);
372  setRestartRequired(true);
373  }
374  break;
375  case Listen:
376  if (settings.value("fListen") != value) {
377  settings.setValue("fListen", value);
378  setRestartRequired(true);
379  }
380  break;
381  default:
382  break;
383  }
384  }
385 
386  emit dataChanged(index, index);
387 
388  return successful;
389 }

402 bool OptionsModel::getProxySettings(QNetworkProxy& proxy) const
403 {
404  // Directly query current base proxy, because

406  proxyType curProxy;
407  if (GetProxy(NET_IPV4, curProxy)) {

411 
412  return true;
413  } else
414  proxy.setType(QNetworkProxy::NoProxy);
415 
416  return false;
