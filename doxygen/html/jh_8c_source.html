103 #define Sb(x0, x1, x2, x3, c) do { \
104  x3 = ~x3; \
105  x0 ^= (c) & ~x2; \
106  tmp = (c) ^ (x0 & x1); \
107  x0 ^= x2 & x3; \
108  x3 ^= ~x1 & x2; \
109  x1 ^= x0 & x2; \
110  x2 ^= x0 & ~x3; \
111  x0 ^= x1 | x3; \
112  x3 ^= x1 & x2; \
113  x1 ^= tmp & x0; \
114  x2 ^= tmp; \
115  } while (0)
