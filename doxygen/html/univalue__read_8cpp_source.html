13 static const char *hatoui(const char *first, const char *last,
14  unsigned int& out)
15 {
16  unsigned int result = 0;
17  for (; first != last; ++first)
18  {
19  int digit;
20  if (isdigit(*first))
21  digit = *first - '0';
22 
23  else if (*first >= 'a' && *first <= 'f')
24  digit = *first - 'a' + 10;
25 
26  else if (*first >= 'A' && *first <= 'F')
27  digit = *first - 'A' + 10;
28 
29  else
30  break;
31 
32  result = 16 * result + digit;
33  }
34  out = result;
35 
36  return first;
37 }
