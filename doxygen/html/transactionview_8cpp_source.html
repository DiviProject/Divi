251 void TransactionView::chooseDate(int idx)
252 {
253  if (!transactionProxyModel)
254  return;
255  QDate current = QDate::currentDate();
256  dateRangeWidget->setVisible(false);
257  switch (dateWidget->itemData(idx).toInt()) {
258  case All:
259  transactionProxyModel->setDateRange(
260  TransactionFilterProxy::MIN_DATE,
261  TransactionFilterProxy::MAX_DATE);
262  break;
263  case Today:
264  transactionProxyModel->setDateRange(
265  QDateTime(current),
266  TransactionFilterProxy::MAX_DATE);
267  break;
268  case ThisWeek: {
269  // Find last Monday
270  QDate startOfWeek = current.addDays(-(current.dayOfWeek() - 1));
271  transactionProxyModel->setDateRange(
272  QDateTime(startOfWeek),
273  TransactionFilterProxy::MAX_DATE);
274 
275  } break;
276  case ThisMonth:
277  transactionProxyModel->setDateRange(
278  QDateTime(QDate(current.year(), current.month(), 1)),
279  TransactionFilterProxy::MAX_DATE);
280  break;
281  case LastMonth:
282  transactionProxyModel->setDateRange(
283  QDateTime(QDate(current.year(), current.month() - 1, 1)),
284  QDateTime(QDate(current.year(), current.month(), 1)));
285  break;
286  case ThisYear:
287  transactionProxyModel->setDateRange(
288  QDateTime(QDate(current.year(), 1, 1)),
289  TransactionFilterProxy::MAX_DATE);
290  break;
291  case Range:
292  dateRangeWidget->setVisible(true);
293  dateRangeChanged();
294  break;
295  }
296  // Persist settings
297  if (dateWidget->itemData(idx).toInt() != Range) {
298  QSettings settings;
299  settings.setValue("transactionDate", idx);
300  }
301 }

330 {

332  return;

334 
335  if (model) {

415 void TransactionView::editLabel()
416 {
417  if (!transactionView->selectionModel() || !model)
418  return;
419  QModelIndexList selection = transactionView->selectionModel()->selectedRows();
420  if (!selection.isEmpty()) {
421  AddressTableModel* addressBook = model->getAddressTableModel();
422  if (!addressBook)
423  return;
424  QString address = selection.at(0).data(TransactionTableModel::AddressRole).toString();
425  if (address.isEmpty()) {
426  // If this transaction has no associated address, exit
427  return;
428  }
429  // Is address in address book? Address book can miss address when a transaction is
430  // sent from outside the UI.
431  int idx = addressBook->lookupAddress(address);
432  if (idx != -1) {
433  // Edit sending / receiving address
434  QModelIndex modelIdx = addressBook->index(idx, 0, QModelIndex());
435  // Determine type of address, launch appropriate editor dialog type
436  QString type = modelIdx.data(AddressTableModel::TypeRole).toString();
437 
438  EditAddressDialog dlg(
439  type == AddressTableModel::Receive ? EditAddressDialog::EditReceivingAddress : EditAddressDialog::EditSendingAddress, this);
440  dlg.setModel(addressBook);
441  dlg.loadRow(idx);
442  dlg.exec();
443  } else {
444  // Add sending address
445  EditAddressDialog dlg(EditAddressDialog::NewSendingAddress,
446  this);
447  dlg.setModel(addressBook);
448  dlg.setAddress(address);
449  dlg.exec();
450  }
451  }
452 }
