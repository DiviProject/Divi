38 uint256 CPartialMerkleTree::CalcHash(int height, unsigned int pos, const std::vector<uint256>& vTxid)
39 {
40  if (height == 0) {
41  // hash at height 0 is the txids themself
42  return vTxid[pos];
43  } else {
44  // calculate left hash
45  uint256 left = CalcHash(height - 1, pos * 2, vTxid), right;
46  // calculate right hash if not beyond the end of the array - copy left hash otherwise1
47  if (pos * 2 + 1 < CalcTreeWidth(height - 1))
48  right = CalcHash(height - 1, pos * 2 + 1, vTxid);
49  else
50  right = left;
51  // combine subhashes
52  return Hash(BEGIN(left), END(left), BEGIN(right), END(right));
53  }
54 }

56 void CPartialMerkleTree::TraverseAndBuild(int height, unsigned int pos, const std::vector<uint256>& vTxid, const std::vector<bool>& vMatch)
57 {
58  // determine whether this node is the parent of at least one matched txid
59  bool fParentOfMatch = false;
60  for (unsigned int p = pos << height; p < (pos + 1) << height && p < nTransactions; p++)
61  fParentOfMatch |= vMatch[p];
62  // store as flag bit
63  vBits.push_back(fParentOfMatch);
64  if (height == 0 || !fParentOfMatch) {
65  // if at height 0, or nothing interesting below, store hash and stop
66  vHash.push_back(CalcHash(height, pos, vTxid));
67  } else {
68  // otherwise, don't store any hash, but descend into the subtrees
69  TraverseAndBuild(height - 1, pos * 2, vTxid, vMatch);
70  if (pos * 2 + 1 < CalcTreeWidth(height - 1))
71  TraverseAndBuild(height - 1, pos * 2 + 1, vTxid, vMatch);
72  }
73 }

75 uint256 CPartialMerkleTree::TraverseAndExtract(int height, unsigned int pos, unsigned int& nBitsUsed, unsigned int& nHashUsed, std::vector<uint256>& vMatch)
76 {
77  if (nBitsUsed >= vBits.size()) {
78  // overflowed the bits array - failure
79  fBad = true;
80  return 0;
81  }
82  bool fParentOfMatch = vBits[nBitsUsed++];
83  if (height == 0 || !fParentOfMatch) {
84  // if at height 0, or nothing interesting below, use stored hash and do not descend
85  if (nHashUsed >= vHash.size()) {
86  // overflowed the hash array - failure
87  fBad = true;
88  return 0;
89  }
90  const uint256& hash = vHash[nHashUsed++];
91  if (height == 0 && fParentOfMatch) // in case of height 0, we have a matched txid
92  vMatch.push_back(hash);
93  return hash;
94  } else {
95  // otherwise, descend into the subtrees to extract matched txids and hashes
96  uint256 left = TraverseAndExtract(height - 1, pos * 2, nBitsUsed, nHashUsed, vMatch), right;
97  if (pos * 2 + 1 < CalcTreeWidth(height - 1))
98  right = TraverseAndExtract(height - 1, pos * 2 + 1, nBitsUsed, nHashUsed, vMatch);
99  else
100  right = left;
101  // and combine them before returning
102  return Hash(BEGIN(left), END(left), BEGIN(right), END(right));
103  }
104 }

123 uint256 CPartialMerkleTree::ExtractMatches(std::vector<uint256>& vMatch)
124 {
125  vMatch.clear();
126  // An empty set will not work
127  if (nTransactions == 0)
128  return 0;
129  // check for excessively high numbers of transactions
130  if (nTransactions > MAX_BLOCK_SIZE_CURRENT / 60) // 60 is the lower bound for the size of a serialized CTransaction
131  return 0;
132  // there can never be more hashes provided than one for every txid
133  if (vHash.size() > nTransactions)
134  return 0;
135  // there must be at least one bit per node in the partial tree, and at least one node per hash
136  if (vBits.size() < vHash.size())
137  return 0;
138  // calculate height of tree
139  int nHeight = 0;
140  while (CalcTreeWidth(nHeight) > 1)
141  nHeight++;
142  // traverse the partial tree
143  unsigned int nBitsUsed = 0, nHashUsed = 0;
144  uint256 hashMerkleRoot = TraverseAndExtract(nHeight, 0, nBitsUsed, nHashUsed, vMatch);
145  // verify that no problems occured during the tree traversal
146  if (fBad)
147  return 0;
148  // verify that all bits were consumed (except for the padding caused by serializing it as a byte sequence)
149  if ((nBitsUsed + 7) / 8 != (vBits.size() + 7) / 8)
150  return 0;
151  // verify that all hashes were consumed
152  if (nHashUsed != vHash.size())
153  return 0;
154  return hashMerkleRoot;
155 }
