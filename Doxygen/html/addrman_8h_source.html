276  void Serialize(Stream& s, int nType, int nVersionDummy) const
277  {
278  LOCK(cs);
279 
280  unsigned char nVersion = 1;
281  s << nVersion;
282  s << ((unsigned char)32);
283  s << nKey;
284  s << nNew;
285  s << nTried;
286 
287  int nUBuckets = ADDRMAN_NEW_BUCKET_COUNT ^ (1 << 30);
288  s << nUBuckets;
289  std::map<int, int> mapUnkIds;
290  int nIds = 0;
291  for (std::map<int, CAddrInfo>::const_iterator it = mapInfo.begin(); it != mapInfo.end(); it++) {
292  mapUnkIds[(*it).first] = nIds;
293  const CAddrInfo& info = (*it).second;
294  if (info.nRefCount) {
295  assert(nIds != nNew); // this means nNew was wrong, oh ow
296  s << info;
297  nIds++;
298  }
299  }
300  nIds = 0;
301  for (std::map<int, CAddrInfo>::const_iterator it = mapInfo.begin(); it != mapInfo.end(); it++) {
302  const CAddrInfo& info = (*it).second;
303  if (info.fInTried) {
304  assert(nIds != nTried); // this means nTried was wrong, oh ow
305  s << info;
306  nIds++;
307  }
308  }
309  for (int bucket = 0; bucket < ADDRMAN_NEW_BUCKET_COUNT; bucket++) {
310  int nSize = 0;
311  for (int i = 0; i < ADDRMAN_BUCKET_SIZE; i++) {
312  if (vvNew[bucket][i] != -1)
313  nSize++;
314  }
315  s << nSize;
316  for (int i = 0; i < ADDRMAN_BUCKET_SIZE; i++) {
317  if (vvNew[bucket][i] != -1) {
318  int nIndex = mapUnkIds[vvNew[bucket][i]];
319  s << nIndex;
320  }
321  }
322  }

325  template <typename Stream>
326  void Unserialize(Stream& s, int nType, int nVersionDummy)
327  {
328  LOCK(cs);
329 
330  Clear();
331 
332  unsigned char nVersion;
333  s >> nVersion;
334  unsigned char nKeySize;
335  s >> nKeySize;
336  if (nKeySize != 32) throw std::ios_base::failure("Incorrect keysize in addrman deserialization");
337  s >> nKey;
338  s >> nNew;
339  s >> nTried;
340  int nUBuckets = 0;
341  s >> nUBuckets;
342  if (nVersion != 0) {
343  nUBuckets ^= (1 << 30);
344  }
345 
346  // Deserialize entries from the new table.
347  for (int n = 0; n < nNew; n++) {
348  CAddrInfo& info = mapInfo[n];
349  s >> info;
350  mapAddr[info] = n;
351  info.nRandomPos = vRandom.size();
352  vRandom.push_back(n);
353  if (nVersion != 1 || nUBuckets != ADDRMAN_NEW_BUCKET_COUNT) {
354  // In case the new table data cannot be used (nVersion unknown, or bucket count wrong),
355  // immediately try to give them a reference based on their primary source address.
356  int nUBucket = info.GetNewBucket(nKey);
357  int nUBucketPos = info.GetBucketPosition(nKey, true, nUBucket);
358  if (vvNew[nUBucket][nUBucketPos] == -1) {
359  vvNew[nUBucket][nUBucketPos] = n;
360  info.nRefCount++;
361  }
362  }
363  }
364  nIdCount = nNew;
365 
366  // Deserialize entries from the tried table.
367  int nLost = 0;
368  for (int n = 0; n < nTried; n++) {
369  CAddrInfo info;
370  s >> info;
371  int nKBucket = info.GetTriedBucket(nKey);
372  int nKBucketPos = info.GetBucketPosition(nKey, false, nKBucket);
373  if (vvTried[nKBucket][nKBucketPos] == -1) {
374  info.nRandomPos = vRandom.size();
375  info.fInTried = true;
376  vRandom.push_back(nIdCount);
377  mapInfo[nIdCount] = info;
378  mapAddr[info] = nIdCount;
379  vvTried[nKBucket][nKBucketPos] = nIdCount;
380  nIdCount++;
381  } else {
382  nLost++;
383  }
384  }
385  nTried -= nLost;
386 
387  // Deserialize positions in the new table (if possible).
388  for (int bucket = 0; bucket < nUBuckets; bucket++) {
389  int nSize = 0;
390  s >> nSize;
391  for (int n = 0; n < nSize; n++) {
392  int nIndex = 0;
393  s >> nIndex;
394  if (nIndex >= 0 && nIndex < nNew) {
395  CAddrInfo& info = mapInfo[nIndex];
396  int nUBucketPos = info.GetBucketPosition(nKey, true, bucket);
397  if (nVersion == 1 && nUBuckets == ADDRMAN_NEW_BUCKET_COUNT && vvNew[bucket][nUBucketPos] == -1 && info.nRefCount < ADDRMAN_NEW_BUCKETS_PER_ADDRESS) {
398  info.nRefCount++;
399  vvNew[bucket][nUBucketPos] = nIndex;
400  }
401  }
402  }
403  }
404 
405  // Prune new entries with refcount 0 (as a result of collisions).
406  int nLostUnk = 0;
407  for (std::map<int, CAddrInfo>::const_iterator it = mapInfo.begin(); it != mapInfo.end();) {
408  if (it->second.fInTried == false && it->second.nRefCount == 0) {
409  std::map<int, CAddrInfo>::const_iterator itCopy = it++;
410  Delete(itCopy->first);
411  nLostUnk++;
412  } else {
413  it++;
414  }
415  }
416  if (nLost + nLostUnk > 0) {
417  LogPrint("addrman", "addrman lost %i new and %i tried addresses due to collisions\n", nLostUnk, nLost);
418  }
419 
420  Check();
421  }
