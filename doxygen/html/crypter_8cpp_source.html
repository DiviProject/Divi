49 bool CCrypter::Encrypt(const CKeyingMaterial& vchPlaintext, std::vector<unsigned char>& vchCiphertext)
50 {
51  if (!fKeySet)
52  return false;
53 
54  // max ciphertext len for a n bytes of plaintext is
55  // n + AES_BLOCK_SIZE - 1 bytes
56  int nLen = vchPlaintext.size();
57  int nCLen = nLen + AES_BLOCK_SIZE, nFLen = 0;
58  vchCiphertext = std::vector<unsigned char>(nCLen);
59 
60  EVP_CIPHER_CTX ctx;
61 
62  bool fOk = true;
63 
64  EVP_CIPHER_CTX_init(&ctx);
65  if (fOk) fOk = EVP_EncryptInit_ex(&ctx, EVP_aes_256_cbc(), NULL, chKey, chIV) != 0;
66  if (fOk) fOk = EVP_EncryptUpdate(&ctx, &vchCiphertext[0], &nCLen, &vchPlaintext[0], nLen) != 0;
67  if (fOk) fOk = EVP_EncryptFinal_ex(&ctx, (&vchCiphertext[0]) + nCLen, &nFLen) != 0;
68  EVP_CIPHER_CTX_cleanup(&ctx);
69 
70  if (!fOk) return false;
71 
72  vchCiphertext.resize(nCLen + nFLen);

116 bool EncryptAES256(const SecureString& sKey, const SecureString& sPlaintext, const std::string& sIV, std::string& sCiphertext)
117 {
118  // max ciphertext len for a n bytes of plaintext is
119  // n + AES_BLOCK_SIZE - 1 bytes
120  int nLen = sPlaintext.size();
121  int nCLen = nLen + AES_BLOCK_SIZE;
122  int nFLen = 0;
123 
124  // Verify key sizes
125  if (sKey.size() != 32 || sIV.size() != AES_BLOCK_SIZE) {
126  LogPrintf("crypter EncryptAES256 - Invalid key or block size: Key: %d sIV:%d\n", sKey.size(), sIV.size());
127  return false;
128  }
129 
130  // Prepare output buffer
131  sCiphertext.resize(nCLen);
132 
133  // Perform the encryption
134  EVP_CIPHER_CTX ctx;
135 
136  bool fOk = true;
137 
138  EVP_CIPHER_CTX_init(&ctx);
139  if (fOk) fOk = EVP_EncryptInit_ex(&ctx, EVP_aes_256_cbc(), NULL, (const unsigned char*)&sKey[0], (const unsigned char*)&sIV[0]);
140  if (fOk) fOk = EVP_EncryptUpdate(&ctx, (unsigned char*)&sCiphertext[0], &nCLen, (const unsigned char*)&sPlaintext[0], nLen);
141  if (fOk) fOk = EVP_EncryptFinal_ex(&ctx, (unsigned char*)(&sCiphertext[0]) + nCLen, &nFLen);
142  EVP_CIPHER_CTX_cleanup(&ctx);
143 
144  if (!fOk) return false;
145 
146  sCiphertext.resize(nCLen + nFLen);

217 bool CCryptoKeyStore::Unlock(const CKeyingMaterial& vMasterKeyIn)
218 {
219  {
220  LOCK(cs_KeyStore);
221  if (!SetCrypted())
222  return false;
223 
224  bool keyPass = false;
225  bool keyFail = false;
226  CryptedKeyMap::const_iterator mi = mapCryptedKeys.begin();
227  for (; mi != mapCryptedKeys.end(); ++mi) {
228  const CPubKey& vchPubKey = (*mi).second.first;
229  const std::vector<unsigned char>& vchCryptedSecret = (*mi).second.second;
230  CKeyingMaterial vchSecret;
231  if (!DecryptSecret(vMasterKeyIn, vchCryptedSecret, vchPubKey.GetHash(), vchSecret)) {
232  keyFail = true;
233  break;
234  }
235  if (vchSecret.size() != 32) {
236  keyFail = true;
237  break;
238  }
239  CKey key;
240  key.Set(vchSecret.begin(), vchSecret.end(), vchPubKey.IsCompressed());
241  if (key.GetPubKey() != vchPubKey) {
242  keyFail = true;
243  break;
244  }
245  keyPass = true;
246  if (fDecryptionThoroughlyChecked)
247  break;
248  }
249  if (keyPass && keyFail) {
250  LogPrintf("The wallet is probably corrupted: Some keys decrypt but not all.");
251  assert(false);
252  }
253  if (keyFail || !keyPass)
254  return false;
255  vMasterKey = vMasterKeyIn;
256  fDecryptionThoroughlyChecked = true;
257  }
258  NotifyStatusChanged(this);
259  return true;
260 }
