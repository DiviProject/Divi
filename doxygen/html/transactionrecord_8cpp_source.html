33 QList<TransactionRecord> TransactionRecord::decomposeTransaction(const CWallet* wallet, const CWalletTx& wtx)
34 {
35  QList<TransactionRecord> parts;
36  int64_t nTime = wtx.GetComputedTxTime();
37  CAmount nCredit = wtx.GetCredit(ISMINE_ALL);
38  CAmount nDebit = wtx.GetDebit(ISMINE_ALL);
39  CAmount nNet = nCredit - nDebit;
40  uint256 hash = wtx.GetHash();
41  std::map<std::string, std::string> mapValue = wtx.mapValue;
42 
43  if (wtx.IsCoinStake()) {
44  TransactionRecord sub(hash, nTime);
45  CTxDestination address;
46  if (!ExtractDestination(wtx.vout[1].scriptPubKey, address))
47  return parts;
48 
49  if (!IsMine(*wallet, address)) {
50  //if the address is not yours then it means you have a tx sent to you in someone elses coinstake tx
51  for (unsigned int i = 1; i < wtx.vout.size(); i++) {
52  CTxDestination outAddress;
53  if (ExtractDestination(wtx.vout[i].scriptPubKey, outAddress)) {
54  if (IsMine(*wallet, outAddress)) {
55  isminetype mine = wallet->IsMine(wtx.vout[i]);
56  sub.involvesWatchAddress = mine & ISMINE_WATCH_ONLY;
57  sub.type = TransactionRecord::MNReward;
58  sub.address = CBitcoinAddress(outAddress).ToString();
59  sub.credit = wtx.vout[i].nValue;
60  }
61  }
62  }
63  } else {
64  //stake reward
65  isminetype mine = wallet->IsMine(wtx.vout[1]);
66  sub.involvesWatchAddress = mine & ISMINE_WATCH_ONLY;
67  sub.type = TransactionRecord::StakeMint;
68  sub.address = CBitcoinAddress(address).ToString();
69  sub.credit = nNet;
70  }
71  parts.append(sub);
72  } else if (wtx.IsZerocoinSpend()) {
73  // a zerocoin spend that was created by this wallet
74  libzerocoin::CoinSpend zcspend = TxInToZerocoinSpend(wtx.vin[0]);
75  bool fSpendFromMe = wallet->IsMyZerocoinSpend(zcspend.getCoinSerialNumber());
76 
77  //zerocoin spend outputs
78  bool fFeeAssigned = false;
79  for (const CTxOut txout : wtx.vout) {
80  // change that was reminted as zerocoins
81  if (txout.IsZerocoinMint()) {
82  // do not display record if this isn't from our wallet
83  if (!fSpendFromMe)
84  continue;
85 
86  TransactionRecord sub(hash, nTime);
87  sub.type = TransactionRecord::ZerocoinSpend_Change_zDiv;
88  sub.address = mapValue["zerocoinmint"];
89  sub.debit = -txout.nValue;
90  if (!fFeeAssigned) {
91  sub.debit -= (wtx.GetZerocoinSpent() - wtx.GetValueOut());
92  fFeeAssigned = true;
93  }
94  sub.idx = parts.size();
95  parts.append(sub);
96  continue;
97  }
98 
99  string strAddress = "";
100  CTxDestination address;
101  if (ExtractDestination(txout.scriptPubKey, address))
102  strAddress = CBitcoinAddress(address).ToString();
103 
104  // a zerocoinspend that was sent to an address held by this wallet
105  isminetype mine = wallet->IsMine(txout);
106  if (mine) {
107  TransactionRecord sub(hash, nTime);
108  sub.type = (fSpendFromMe ? TransactionRecord::ZerocoinSpend_FromMe : TransactionRecord::RecvFromZerocoinSpend);
109  sub.debit = txout.nValue;
110  sub.address = mapValue["recvzerocoinspend"];
111  if (strAddress != "")
112  sub.address = strAddress;
113  sub.idx = parts.size();
114  parts.append(sub);
115  continue;
116  }
117 
118  // spend is not from us, so do not display the spend side of the record
119  if (!fSpendFromMe)
120  continue;
121 
122  // zerocoin spend that was sent to someone else
123  TransactionRecord sub(hash, nTime);
124  sub.debit = -txout.nValue;
125  sub.type = TransactionRecord::ZerocoinSpend;
126  sub.address = mapValue["zerocoinspend"];
127  if (strAddress != "")
128  sub.address = strAddress;
129  sub.idx = parts.size();
130  parts.append(sub);
131  }
132  } else if (nNet > 0 || wtx.IsCoinBase()) {
133  //
134  // Credit
135  //
136  BOOST_FOREACH (const CTxOut& txout, wtx.vout) {
137  isminetype mine = wallet->IsMine(txout);
138  if (mine) {
139  TransactionRecord sub(hash, nTime);
140  CTxDestination address;
141  sub.idx = parts.size(); // sequence number
142  sub.credit = txout.nValue;
143  sub.involvesWatchAddress = mine & ISMINE_WATCH_ONLY;
144  if (ExtractDestination(txout.scriptPubKey, address) && IsMine(*wallet, address)) {
145  // Received by DIVI Address
146  sub.type = TransactionRecord::RecvWithAddress;
147  sub.address = CBitcoinAddress(address).ToString();
148  } else {
149  // Received by IP connection (deprecated features), or a multisignature or other non-simple transaction
150  sub.type = TransactionRecord::RecvFromOther;
151  sub.address = mapValue["from"];
152  }
153  if (wtx.IsCoinBase()) {
154  // Generated
155  sub.type = TransactionRecord::Generated;
156  }
157 
158  parts.append(sub);
159  }
160  }
161  } else {
162  bool fAllFromMeDenom = true;
163  int nFromMe = 0;
164  bool involvesWatchAddress = false;
165  isminetype fAllFromMe = ISMINE_SPENDABLE;
166  BOOST_FOREACH (const CTxIn& txin, wtx.vin) {
167  if (wallet->IsMine(txin)) {
168  fAllFromMeDenom = fAllFromMeDenom && wallet->IsDenominated(txin);
169  nFromMe++;
170  }
171  isminetype mine = wallet->IsMine(txin);
172  if (mine & ISMINE_WATCH_ONLY) involvesWatchAddress = true;
173  if (fAllFromMe > mine) fAllFromMe = mine;
174  }
175 
176  isminetype fAllToMe = ISMINE_SPENDABLE;
177  bool fAllToMeDenom = true;
178  int nToMe = 0;
179  BOOST_FOREACH (const CTxOut& txout, wtx.vout) {
180  if (wallet->IsMine(txout)) {
181  fAllToMeDenom = fAllToMeDenom && wallet->IsDenominatedAmount(txout.nValue);
182  nToMe++;
183  }
184  isminetype mine = wallet->IsMine(txout);
185  if (mine & ISMINE_WATCH_ONLY) involvesWatchAddress = true;
186  if (fAllToMe > mine) fAllToMe = mine;
187  }
188 
189  if (fAllFromMeDenom && fAllToMeDenom && nFromMe * nToMe) {
190  parts.append(TransactionRecord(hash, nTime, TransactionRecord::ObfuscationDenominate, "", -nDebit, nCredit));
191  parts.last().involvesWatchAddress = false; // maybe pass to TransactionRecord as constructor argument
192  } else if (fAllFromMe && fAllToMe) {
193  // Payment to self
194  // TODO: this section still not accurate but covers most cases,
195  // might need some additional work however
196 
197  TransactionRecord sub(hash, nTime);
198  // Payment to self by default
199  sub.type = TransactionRecord::SendToSelf;
200  sub.address = "";
201 
202  if (mapValue["DS"] == "1") {
203  sub.type = TransactionRecord::Obfuscated;
204  CTxDestination address;
205  if (ExtractDestination(wtx.vout[0].scriptPubKey, address)) {
206  // Sent to DIVI Address
207  sub.address = CBitcoinAddress(address).ToString();
208  } else {
209  // Sent to IP, or other non-address transaction like OP_EVAL
210  sub.address = mapValue["to"];
211  }
212  } else {
213  for (unsigned int nOut = 0; nOut < wtx.vout.size(); nOut++) {
214  const CTxOut& txout = wtx.vout[nOut];
215  sub.idx = parts.size();
216 
217  if (wallet->IsCollateralAmount(txout.nValue)) sub.type = TransactionRecord::ObfuscationMakeCollaterals;
218  if (wallet->IsDenominatedAmount(txout.nValue)) sub.type = TransactionRecord::ObfuscationCreateDenominations;
219  if (nDebit - wtx.GetValueOut() == OBFUSCATION_COLLATERAL) sub.type = TransactionRecord::ObfuscationCollateralPayment;
220  }
221  }
222 
223  CAmount nChange = wtx.GetChange();
224 
225  sub.debit = -(nDebit - nChange);
226  sub.credit = nCredit - nChange;
227  parts.append(sub);
228  parts.last().involvesWatchAddress = involvesWatchAddress; // maybe pass to TransactionRecord as constructor argument
229  } else if (fAllFromMe) {
230  //
231  // Debit
232  //
233  CAmount nTxFee = nDebit - wtx.GetValueOut();
234 
235  for (unsigned int nOut = 0; nOut < wtx.vout.size(); nOut++) {
236  const CTxOut& txout = wtx.vout[nOut];
237  TransactionRecord sub(hash, nTime);
238  sub.idx = parts.size();
239  sub.involvesWatchAddress = involvesWatchAddress;
240 
241  if (wallet->IsMine(txout)) {
242  // Ignore parts sent to self, as this is usually the change
243  // from a transaction sent back to our own address.
244  continue;
245  }
246 
247  CTxDestination address;
248  if (ExtractDestination(txout.scriptPubKey, address)) {
249  // Sent to DIVI Address
250  sub.type = TransactionRecord::SendToAddress;
251  sub.address = CBitcoinAddress(address).ToString();
252  } else if (txout.IsZerocoinMint()){
253  sub.type = TransactionRecord::ZerocoinMint;
254  sub.address = mapValue["zerocoinmint"];
255  } else {
256  // Sent to IP, or other non-address transaction like OP_EVAL
257  sub.type = TransactionRecord::SendToOther;
258  sub.address = mapValue["to"];
259  }
260 
261  if (mapValue["DS"] == "1") {
262  sub.type = TransactionRecord::Obfuscated;
263  }
264 
265  CAmount nValue = txout.nValue;
266  /* Add fee to first output */
267  if (nTxFee > 0) {
268  nValue += nTxFee;
269  nTxFee = 0;
270  }
271  sub.debit = -nValue;
272 
273  parts.append(sub);
274  }
275  } else {
276  //
277  // Mixed debit transaction, can't break down payees
278  //
279  parts.append(TransactionRecord(hash, nTime, TransactionRecord::Other, "", nNet, 0));
280  parts.last().involvesWatchAddress = involvesWatchAddress;
281  }
282  }
283 
284  return parts;
285 }

287 void TransactionRecord::updateStatus(const CWalletTx& wtx)
288 {
289  AssertLockHeld(cs_main);
290  // Determine transaction status
291 
292  // Find the block the tx is in
293  CBlockIndex* pindex = NULL;
294  BlockMap::iterator mi = mapBlockIndex.find(wtx.hashBlock);
295  if (mi != mapBlockIndex.end())
296  pindex = (*mi).second;
297 
298  // Sort order, unrecorded transactions sort to the top
299  status.sortKey = strprintf("%010d-%01d-%010u-%03d",
300  (pindex ? pindex->nHeight : std::numeric_limits<int>::max()),
301  (wtx.IsCoinBase() ? 1 : 0),
302  wtx.nTimeReceived,
303  idx);
304  status.countsForBalance = wtx.IsTrusted() && !(wtx.GetBlocksToMaturity() > 0);
305  status.depth = wtx.GetDepthInMainChain();
306  status.cur_num_blocks = chainActive.Height();
307  status.cur_num_ix_locks = nCompleteTXLocks;
308 
309  if (!IsFinalTx(wtx, chainActive.Height() + 1)) {
310  if (wtx.nLockTime < LOCKTIME_THRESHOLD) {
311  status.status = TransactionStatus::OpenUntilBlock;
312  status.open_for = wtx.nLockTime - chainActive.Height();
313  } else {
314  status.status = TransactionStatus::OpenUntilDate;
315  status.open_for = wtx.nLockTime;
316  }
317  }
318  // For generated transactions, determine maturity
319  else if (type == TransactionRecord::Generated || type == TransactionRecord::StakeMint || type == TransactionRecord::MNReward) {
320  if (wtx.GetBlocksToMaturity() > 0) {
321  status.status = TransactionStatus::Immature;
322 
323  if (wtx.IsInMainChain()) {
324  status.matures_in = wtx.GetBlocksToMaturity();
325 
326  // Check if the block was requested by anyone
327  if (GetAdjustedTime() - wtx.nTimeReceived > 2 * 60 && wtx.GetRequestCount() == 0)
328  status.status = TransactionStatus::MaturesWarning;
329  } else {
330  status.status = TransactionStatus::NotAccepted;
331  }
332  } else {
333  status.status = TransactionStatus::Confirmed;
334  }
335  } else {
336  if (status.depth < 0) {
337  status.status = TransactionStatus::Conflicted;
338  } else if (GetAdjustedTime() - wtx.nTimeReceived > 2 * 60 && wtx.GetRequestCount() == 0) {
339  status.status = TransactionStatus::Offline;
340  } else if (status.depth == 0) {
341  status.status = TransactionStatus::Unconfirmed;
342  } else if (status.depth < RecommendedNumConfirmations) {
343  status.status = TransactionStatus::Confirming;
344  } else {
345  status.status = TransactionStatus::Confirmed;
346  }
347  }
348 }
