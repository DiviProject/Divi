111 bool SendCoinsEntry::validate()
112 {
113  if (!model)
114  return false;
115 
116  // Check input validity
117  bool retval = true;
118 
119  // Skip checks for payment request
120  if (recipient.paymentRequest.IsInitialized())
121  return retval;
122 
123  if (!model->validateAddress(ui->payTo->text())) {
124  ui->payTo->setValid(false);
125  retval = false;
126  }
127 
128  if (!ui->payAmount->validate()) {
129  retval = false;
130  }
131 
132  // Sending a zero amount is invalid
133  if (ui->payAmount->value(0) <= 0) {
134  ui->payAmount->setValid(false);
135  retval = false;
136  }
137 
138  // Reject dust outputs:
139  if (retval && GUIUtil::isDust(ui->payTo->text(), ui->payAmount->value())) {
140  ui->payAmount->setValid(false);
141  retval = false;
142  }
143 
144  return retval;
145 }

173 void SendCoinsEntry::setValue(const SendCoinsRecipient& value)
174 {
175  recipient = value;
176 
177  if (recipient.paymentRequest.IsInitialized()) // payment request
178  {
179  if (recipient.authenticatedMerchant.isEmpty()) // insecure
180  {
181  ui->payTo_is->setText(recipient.address);
182  ui->memoTextLabel_is->setText(recipient.message);
183  ui->payAmount_is->setValue(recipient.amount);
184  ui->payAmount_is->setReadOnly(true);
185  setCurrentWidget(ui->SendCoins_InsecurePaymentRequest);
186  } else // secure
187  {
188  ui->payTo_s->setText(recipient.authenticatedMerchant);
189  ui->memoTextLabel_s->setText(recipient.message);
190  ui->payAmount_s->setValue(recipient.amount);
191  ui->payAmount_s->setReadOnly(true);
192  setCurrentWidget(ui->SendCoins_SecurePaymentRequest);
193  }
194  } else // normal payment
195  {
196  // message
197  ui->messageTextLabel->setText(recipient.message);
198  ui->messageTextLabel->setVisible(!recipient.message.isEmpty());
199  ui->messageLabel->setVisible(!recipient.message.isEmpty());
200 
201  ui->addAsLabel->clear();
202  ui->payTo->setText(recipient.address); // this may set a label from addressbook
203  if (!recipient.label.isEmpty()) // if a label had been set from the addressbook, dont overwrite with an empty label
204  ui->addAsLabel->setText(recipient.label);
205  ui->payAmount->setValue(recipient.amount);
206  }
207 }
