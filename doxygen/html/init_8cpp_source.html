105 //
106 // Thread management and startup/shutdown:
107 //
108 // The network-processing threads are all part of a thread group
109 // created by AppInit() or the Qt main() function.
110 //
111 // A clean exit happens when StartShutdown() or the SIGTERM
112 // signal handler sets fRequestShutdown, which triggers
113 // the DetectShutdownThread(), which interrupts the main thread group.
114 // DetectShutdownThread() then exits, which causes AppInit() to
115 // continue (it .joins the shutdown thread).
116 // Shutdown() is then
117 // called to clean up database connections, and stop other
118 // threads that should only be stopped after the main network-processing
119 // threads have exited.
120 //
121 // Note that if running -daemon the parent process returns from AppInit2
122 // before adding any threads to the threadGroup, so .join_all() returns
123 // immediately and the parent exits from main().
124 //
125 // Shutdown for Qt is very similar, only it uses a QTimer to detect
126 // fRequestShutdown getting set, and then does the normal Qt
127 // shutdown thing.
128 //
129

145  bool GetCoins(const uint256& txid, CCoins& coins) const
146  {
147  try {
148  return CCoinsViewBacked::GetCoins(txid, coins);

150  uiInterface.ThreadSafeMessageBox(_("Error reading from database, shutting down."), "", CClientUIInterface::MSG_ERROR);
151  LogPrintf("Error reading from database: %s\n", e.what());
152  // Starting the shutdown sequence and returning false to the caller would be
153  // interpreted as 'entry not found' (as opposed to unable to read data), and
154  // could lead to invalid interpration. Just exit immediately, as we can't
155  // continue anyway, and all writes should be atomic.
156  abort();
157  }
158  }
159  // Writes do not need similar protection, as failure to write is handled by the caller.
160 };

166 void PrepareShutdown()
167 {
168  fRequestShutdown = true; // Needed when we shutdown the wallet
169  fRestartRequested = true; // Needed when we restart the wallet
170  LogPrintf("%s: In progress...\n", __func__);
171  static CCriticalSection cs_Shutdown;
172  TRY_LOCK(cs_Shutdown, lockShutdown);
173  if (!lockShutdown)
174  return;
175 
180  RenameThread("divi-shutoff");
181  mempool.AddTransactionsUpdated(1);
182  StopRPCThreads();
183 #ifdef ENABLE_WALLET
184  if (pwalletMain)
185  bitdb.Flush(false);
186  GenerateBitcoins(false, NULL, 0);
187 #endif
188  StopNode();
189  InterruptTorControl();
190  StopTorControl();
191  DumpMasternodes();
192  DumpBudgets();
193  DumpMasternodePayments();
194  UnregisterNodeSignals(GetNodeSignals());
195 
196  if (fFeeEstimatesInitialized) {
197  boost::filesystem::path est_path = GetDataDir() / FEE_ESTIMATES_FILENAME;
198  CAutoFile est_fileout(fopen(est_path.string().c_str(), "wb"), SER_DISK, CLIENT_VERSION);
199  if (!est_fileout.IsNull())
200  mempool.WriteFeeEstimates(est_fileout);
201  else
202  LogPrintf("%s: Failed to write fee estimates to %s\n", __func__, est_path.string());
203  fFeeEstimatesInitialized = false;
204  }
205 
206  {
207  LOCK(cs_main);
208  if (pcoinsTip != NULL) {
209  FlushStateToDisk();
210 
211  //record that client took the proper shutdown procedure
212  pblocktree->WriteFlag("shutdown", true);
213  }
214  delete pcoinsTip;
215  pcoinsTip = NULL;
216  delete pcoinscatcher;
217  pcoinscatcher = NULL;
218  delete pcoinsdbview;
219  pcoinsdbview = NULL;
220  delete pblocktree;
221  pblocktree = NULL;
222  delete zerocoinDB;
223  zerocoinDB = NULL;
224  delete pSporkDB;
225  pSporkDB = NULL;
226  }
227 #ifdef ENABLE_WALLET
228  if (pwalletMain)
229  bitdb.Flush(true);
230 #endif
231 
232 #if ENABLE_ZMQ
233  if (pzmqNotificationInterface) {
234  UnregisterValidationInterface(pzmqNotificationInterface);
235  delete pzmqNotificationInterface;
236  pzmqNotificationInterface = NULL;
237  }
238 #endif
239 
240 #ifndef WIN32
241  boost::filesystem::remove(GetPidFile());
242 #endif
243  UnregisterAllValidationInterfaces();
244 }

295 bool static Bind(const CService& addr, unsigned int flags)
296 {
297  if (!(flags & BF_EXPLICIT) && IsLimited(addr))
298  return false;
299  std::string strError;
300  if (!BindListenPort(addr, strError, (flags & BF_WHITELIST) != 0)) {
301  if (flags & BF_REPORT_ERROR)
302  return InitError(strError);
303  return false;
304  }
305  return true;
306 }
