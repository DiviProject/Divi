91 void AskPassphraseDialog::accept()
92 {
93  SecureString oldpass, newpass1, newpass2;
94  if (!model)
95  return;
96  oldpass.reserve(MAX_PASSPHRASE_SIZE);
97  newpass1.reserve(MAX_PASSPHRASE_SIZE);
98  newpass2.reserve(MAX_PASSPHRASE_SIZE);
99  // TODO: get rid of this .c_str() by implementing SecureString::operator=(std::string)
100  // Alternately, find a way to make this input mlock()'d to begin with.
101  oldpass.assign(ui->passEdit1->text().toStdString().c_str());
102  newpass1.assign(ui->passEdit2->text().toStdString().c_str());
103  newpass2.assign(ui->passEdit3->text().toStdString().c_str());
104 
105  switch (mode) {
106  case Encrypt: {
107  if (newpass1.empty() || newpass2.empty()) {
108  // Cannot encrypt with empty passphrase
109  break;
110  }
111  QMessageBox::StandardButton retval = QMessageBox::question(this, tr("Confirm wallet encryption"),
112  tr("Warning: If you encrypt your wallet and lose your passphrase, you will <b>LOSE ALL OF YOUR DIV</b>!") + "<br><br>" + tr("Are you sure you wish to encrypt your wallet?"),
113  QMessageBox::Yes | QMessageBox::Cancel,
114  QMessageBox::Cancel);
115  if (retval == QMessageBox::Yes) {
116  if (newpass1 == newpass2) {
117  if (model->setWalletEncrypted(true, newpass1)) {
118  QMessageBox::warning(this, tr("Wallet encrypted"),
119  "<qt>" +
120  tr("DIVI will close now to finish the encryption process. "
121  "Remember that encrypting your wallet cannot fully protect "
122  "your DIVs from being stolen by malware infecting your computer.") +
123  "<br><br><b>" +
124  tr("IMPORTANT: Any previous backups you have made of your wallet file "
125  "should be replaced with the newly generated, encrypted wallet file. "
126  "For security reasons, previous backups of the unencrypted wallet file "
127  "will become useless as soon as you start using the new, encrypted wallet.") +
128  "</b></qt>");
129  QApplication::quit();
130  } else {
131  QMessageBox::critical(this, tr("Wallet encryption failed"),
132  tr("Wallet encryption failed due to an internal error. Your wallet was not encrypted."));
133  }
134  QDialog::accept(); // Success
135  } else {
136  QMessageBox::critical(this, tr("Wallet encryption failed"),
137  tr("The supplied passphrases do not match."));
138  }
139  } else {
140  QDialog::reject(); // Cancelled
141  }
142  } break;
143  case UnlockAnonymize:
144  case Unlock:
145  if (!model->setWalletLocked(false, oldpass, ui->anonymizationCheckBox->isChecked())) {
146  QMessageBox::critical(this, tr("Wallet unlock failed"),
147  tr("The passphrase entered for the wallet decryption was incorrect."));
148  } else {
149  QDialog::accept(); // Success
150  }
151  break;
152  case Decrypt:
153  if (!model->setWalletEncrypted(false, oldpass)) {
154  QMessageBox::critical(this, tr("Wallet decryption failed"),
155  tr("The passphrase entered for the wallet decryption was incorrect."));
156  } else {
157  QDialog::accept(); // Success
158  }
159  break;
160  case ChangePass:
161  if (newpass1 == newpass2) {
162  if (model->changePassphrase(oldpass, newpass1)) {
163  QMessageBox::information(this, tr("Wallet encrypted"),
164  tr("Wallet passphrase was successfully changed."));
165  QDialog::accept(); // Success
166  } else {
167  QMessageBox::critical(this, tr("Wallet encryption failed"),
168  tr("The passphrase entered for the wallet decryption was incorrect."));
169  }
170  } else {
171  QMessageBox::critical(this, tr("Wallet encryption failed"),
172  tr("The supplied passphrases do not match."));
173  }
174  break;
175  }
176 }

215 bool AskPassphraseDialog::eventFilter(QObject* object, QEvent* event)
216 {
217  /* Detect Caps Lock.
218  * There is no good OS-independent way to check a key state in Qt, but we
219  * can detect Caps Lock by checking for the following condition:
220  * Shift key is down and the result is a lower case character, or
221  * Shift key is not down and the result is an upper case character.
222  */
223  if (event->type() == QEvent::KeyPress) {
224  QKeyEvent* ke = static_cast<QKeyEvent*>(event);
225  QString str = ke->text();
226  if (str.length() != 0) {
227  const QChar* psz = str.unicode();
228  bool fShift = (ke->modifiers() & Qt::ShiftModifier) != 0;
229  if ((fShift && *psz >= 'a' && *psz <= 'z') || (!fShift && *psz >= 'A' && *psz <= 'Z')) {
230  fCapsLock = true;
231  ui->capsLabel->setText(tr("Warning: The Caps Lock key is on!"));
232  } else if (psz->isLetter()) {
233  fCapsLock = false;
234  ui->capsLabel->clear();
235  }
236  }
237  }
238  return QDialog::eventFilter(object, event);
239 }
