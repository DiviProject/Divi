536 void CoinControlDialog::updateLabels(WalletModel* model, QDialog* dialog)
537 {
538  if (!model)
539  return;
540 
541  // nPayAmount
542  CAmount nPayAmount = 0;
543  bool fDust = false;
544  CMutableTransaction txDummy;
545  foreach (const CAmount& amount, CoinControlDialog::payAmounts) {
546  nPayAmount += amount;
547 
548  if (amount > 0) {
549  CTxOut txout(amount, (CScript)vector<unsigned char>(24, 0));
550  txDummy.vout.push_back(txout);
551  if (txout.IsDust(::minRelayTxFee))
552  fDust = true;
553  }
554  }
555 
556  QString sPriorityLabel = tr("none");
557  CAmount nAmount = 0;
558  CAmount nPayFee = 0;
559  CAmount nAfterFee = 0;
560  CAmount nChange = 0;
561  unsigned int nBytes = 0;
562  unsigned int nBytesInputs = 0;
563  double dPriority = 0;
564  double dPriorityInputs = 0;
565  unsigned int nQuantity = 0;
566  int nQuantityUncompressed = 0;
567  bool fAllowFree = false;
568 
569  vector<COutPoint> vCoinControl;
570  vector<COutput> vOutputs;
571  coinControl->ListSelected(vCoinControl);
572  model->getOutputs(vCoinControl, vOutputs);
573 
574  BOOST_FOREACH (const COutput& out, vOutputs) {
575  // unselect already spent, very unlikely scenario, this could happen
576  // when selected are spent elsewhere, like rpc or another computer
577  uint256 txhash = out.tx->GetHash();
578  COutPoint outpt(txhash, out.i);
579  if (model->isSpent(outpt)) {
580  coinControl->UnSelect(outpt);
581  continue;
582  }
583 
584  // Quantity
585  nQuantity++;
586 
587  // Amount
588  nAmount += out.tx->vout[out.i].nValue;
589 
590  // Priority
591  dPriorityInputs += (double)out.tx->vout[out.i].nValue * (out.nDepth + 1);
592 
593  // Bytes
594  CTxDestination address;
595  if (ExtractDestination(out.tx->vout[out.i].scriptPubKey, address)) {
596  CPubKey pubkey;
597  CKeyID* keyid = boost::get<CKeyID>(&address);
598  if (keyid && model->getPubKey(*keyid, pubkey)) {
599  nBytesInputs += (pubkey.IsCompressed() ? 148 : 180);
