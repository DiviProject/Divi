425  uint64_t nSrcPos; // how many bytes have been read from source
426  uint64_t nReadPos; // how many bytes have been read from this
427  uint64_t nReadLimit; // up to which position we're allowed to read
428  uint64_t nRewind; // how many bytes we guarantee to rewind
429  std::vector<char> vchBuf; // the buffer

431 protected:
432  // read data from the source to fill the buffer
433  bool Fill()
434  {
435  unsigned int pos = nSrcPos % vchBuf.size();
436  unsigned int readNow = vchBuf.size() - pos;
437  unsigned int nAvail = vchBuf.size() - (nSrcPos - nReadPos) - nRewind;
438  if (nAvail < readNow)
439  readNow = nAvail;
440  if (readNow == 0)
441  return false;
442  size_t read = fread((void*)&vchBuf[pos], 1, readNow, src);
443  if (read == 0) {
444  throw std::ios_base::failure(feof(src) ? "CBufferedFile::Fill : end of file" : "CBufferedFile::Fill : fread failed");
445  } else {
446  nSrcPos += read;
447  return true;
448  }
449  }

478  // read a number of bytes
479  CBufferedFile& read(char* pch, size_t nSize)
480  {
481  if (nSize + nReadPos > nReadLimit)
482  throw std::ios_base::failure("Read attempted past buffer limit");
483  if (nSize + nRewind > vchBuf.size())
484  throw std::ios_base::failure("Read larger than buffer size");
485  while (nSize > 0) {
486  if (nReadPos == nSrcPos)
487  Fill();
488  unsigned int pos = nReadPos % vchBuf.size();
489  size_t nNow = nSize;
490  if (nNow + pos > vchBuf.size())
491  nNow = vchBuf.size() - pos;
492  if (nNow + nReadPos > nSrcPos)
493  nNow = nSrcPos - nReadPos;
494  memcpy(pch, &vchBuf[pos], nNow);
495  nReadPos += nNow;
496  pch += nNow;
497  nSize -= nNow;
498  }
499  return (*this);
500  }
