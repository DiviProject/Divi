38 struct TxLessThan {
39  bool operator()(const TransactionRecord& a, const TransactionRecord& b) const
40  {
41  return a.hash < b.hash;
42  }
43  bool operator()(const TransactionRecord& a, const uint256& b) const

54 class TransactionTablePriv
55 {
56 public:
57  TransactionTablePriv(CWallet* wallet, TransactionTableModel* parent) : wallet(wallet),
58  parent(parent)
59  {
60  }
61 
62  CWallet* wallet;
63  TransactionTableModel* parent;
64 
65  /* Local cache of wallet.
66  * As it is in the same order as the CWallet, by definition
67  * this is sorted by sha256.
68  */
69  QList<TransactionRecord> cachedWallet;
70 
71  /* Query entire wallet anew from core.
72  */
73  void refreshWallet()
74  {
75  qDebug() << "TransactionTablePriv::refreshWallet";
76  cachedWallet.clear();
77  {
78  LOCK2(cs_main, wallet->cs_wallet);
79  for (std::map<uint256, CWalletTx>::iterator it = wallet->mapWallet.begin(); it != wallet->mapWallet.end(); ++it) {
80  if (TransactionRecord::showTransaction(it->second))
81  cachedWallet.append(TransactionRecord::decomposeTransaction(wallet, it->second));
82  }
83  }
84  }
85 
86  /* Update our model of the wallet incrementally, to synchronize our model of the wallet
87  with that of the core.
88 
89  Call with transaction that was added, removed or changed.
90  */
91  void updateWallet(const uint256& hash, int status, bool showTransaction)
92  {
93  qDebug() << "TransactionTablePriv::updateWallet : " + QString::fromStdString(hash.ToString()) + " " + QString::number(status);
94 
95  // Find bounds of this transaction in model
96  QList<TransactionRecord>::iterator lower = qLowerBound(
97  cachedWallet.begin(), cachedWallet.end(), hash, TxLessThan());
98  QList<TransactionRecord>::iterator upper = qUpperBound(
99  cachedWallet.begin(), cachedWallet.end(), hash, TxLessThan());
100  int lowerIndex = (lower - cachedWallet.begin());
101  int upperIndex = (upper - cachedWallet.begin());
102  bool inModel = (lower != upper);
103 
104  if (status == CT_UPDATED) {
105  if (showTransaction && !inModel)
106  status = CT_NEW; /* Not in model, but want to show, treat as new */
107  if (!showTransaction && inModel)
108  status = CT_DELETED; /* In model, but want to hide, treat as deleted */
109  }
110 
111  qDebug() << " inModel=" + QString::number(inModel) +
112  " Index=" + QString::number(lowerIndex) + "-" + QString::number(upperIndex) +
113  " showTransaction=" + QString::number(showTransaction) + " derivedStatus=" + QString::number(status);
114 
115  switch (status) {
116  case CT_NEW:
117  if (inModel) {
118  qWarning() << "TransactionTablePriv::updateWallet : Warning: Got CT_NEW, but transaction is already in model";
119  break;
120  }
121  if (showTransaction) {
122  LOCK2(cs_main, wallet->cs_wallet);
123  // Find transaction in wallet
124  std::map<uint256, CWalletTx>::iterator mi = wallet->mapWallet.find(hash);
125  if (mi == wallet->mapWallet.end()) {
126  qWarning() << "TransactionTablePriv::updateWallet : Warning: Got CT_NEW, but transaction is not in wallet";
127  break;
128  }
129  // Added -- insert at the right position
130  QList<TransactionRecord> toInsert =
131  TransactionRecord::decomposeTransaction(wallet, mi->second);
132  if (!toInsert.isEmpty()) /* only if something to insert */
133  {
134  parent->beginInsertRows(QModelIndex(), lowerIndex, lowerIndex + toInsert.size() - 1);
135  int insert_idx = lowerIndex;
136  foreach (const TransactionRecord& rec, toInsert) {
137  cachedWallet.insert(insert_idx, rec);
138  insert_idx += 1;
139  }
140  parent->endInsertRows();
141  }
142  }
143  break;
144  case CT_DELETED:
145  if (!inModel) {
146  qWarning() << "TransactionTablePriv::updateWallet : Warning: Got CT_DELETED, but transaction is not in model";
147  break;
148  }
149  // Removed -- remove entire transaction from table
150  parent->beginRemoveRows(QModelIndex(), lowerIndex, upperIndex - 1);
151  cachedWallet.erase(lower, upper);
152  parent->endRemoveRows();
153  break;
154  case CT_UPDATED:
155  // Miscellaneous updates -- nothing to do, status update will take care of this, and is only computed for
156  // visible transactions.
157  break;
158  }
159  }
160 
161  int size()
162  {
163  return cachedWallet.size();
164  }
165 
166  TransactionRecord* index(int idx)
167  {
168  if (idx >= 0 && idx < cachedWallet.size()) {
169  TransactionRecord* rec = &cachedWallet[idx];
170 
171  // Get required locks upfront. This avoids the GUI from getting
172  // stuck if the core is holding the locks for a longer time - for
173  // example, during a wallet rescan.
174  //
175  // If a status update is needed (blocks came in since last check),
176  // update the status of this transaction from the wallet. Otherwise,
177  // simply re-use the cached status.
178  TRY_LOCK(cs_main, lockMain);
179  if (lockMain) {
180  TRY_LOCK(wallet->cs_wallet, lockWallet);
181  if (lockWallet && rec->statusUpdateNeeded()) {
182  std::map<uint256, CWalletTx>::iterator mi = wallet->mapWallet.find(rec->hash);
183 
184  if (mi != wallet->mapWallet.end()) {
185  rec->updateStatus(mi->second);
186  }
187  }
188  }
189  return rec;
190  }
191  return 0;
192  }
193 
194  QString describe(TransactionRecord* rec, int unit)
195  {
196  {
197  LOCK2(cs_main, wallet->cs_wallet);
198  std::map<uint256, CWalletTx>::iterator mi = wallet->mapWallet.find(rec->hash);
199  if (mi != wallet->mapWallet.end()) {
200  return TransactionDesc::toHTML(wallet, mi->second, rec, unit);
201  }
202  }
203  return QString();
204  }

264 QString TransactionTableModel::formatTxStatus(const TransactionRecord* wtx) const
265 {
266  QString status;
267 
268  switch (wtx->status.status) {
269  case TransactionStatus::OpenUntilBlock:
270  status = tr("Open for %n more block(s)", "", wtx->status.open_for);
271  break;
272  case TransactionStatus::OpenUntilDate:
273  status = tr("Open until %1").arg(GUIUtil::dateTimeStr(wtx->status.open_for));
274  break;
275  case TransactionStatus::Offline:
276  status = tr("Offline");
277  break;
278  case TransactionStatus::Unconfirmed:
279  status = tr("Unconfirmed");
280  break;
281  case TransactionStatus::Confirming:
282  status = tr("Confirming (%1 of %2 recommended confirmations)").arg(wtx->status.depth).arg(TransactionRecord::RecommendedNumConfirmations);
283  break;
284  case TransactionStatus::Confirmed:
285  status = tr("Confirmed (%1 confirmations)").arg(wtx->status.depth);
286  break;
287  case TransactionStatus::Conflicted:
288  status = tr("Conflicted");
289  break;
290  case TransactionStatus::Immature:
291  status = tr("Immature (%1 confirmations, will be available after %2)").arg(wtx->status.depth).arg(wtx->status.depth + wtx->status.matures_in);
292  break;
293  case TransactionStatus::MaturesWarning:
294  status = tr("This block was not received by any other nodes and will probably not be accepted!");
295  break;
296  case TransactionStatus::NotAccepted:
297  status = tr("Orphan Block - Generated but not accepted. This does not impact your holdings.");
298  break;
299  }
300 
301  return status;
302 }

519 QVariant TransactionTableModel::data(const QModelIndex& index, int role) const
520 {
521  if (!index.isValid())
522  return QVariant();
523  TransactionRecord* rec = static_cast<TransactionRecord*>(index.internalPointer());
524 
525  switch (role) {
526  case Qt::DecorationRole:
527  switch (index.column()) {
528  case Status:
529  return txStatusDecoration(rec);
530  case Watchonly:
531  return txWatchonlyDecoration(rec);
532  case ToAddress:
533  return txAddressDecoration(rec);
534  }
535  break;
536  case Qt::DisplayRole:
537  switch (index.column()) {
538  case Date:
539  return formatTxDate(rec);
540  case Type:
541  return formatTxType(rec);
542  case ToAddress:
543  return formatTxToAddress(rec, false);
544  case Amount:
545  return formatTxAmount(rec, true, BitcoinUnits::separatorAlways);
546  }
547  break;
548  case Qt::EditRole:
549  // Edit role is used for sorting, so return the unformatted values
550  switch (index.column()) {
551  case Status:
552  return QString::fromStdString(rec->status.sortKey);
553  case Date:
554  return rec->time;
555  case Type:
556  return formatTxType(rec);
557  case Watchonly:
558  return (rec->involvesWatchAddress ? 1 : 0);
559  case ToAddress:
560  return formatTxToAddress(rec, true);
561  case Amount:
562  return qint64(rec->credit + rec->debit);
563  }
564  break;
565  case Qt::ToolTipRole:
566  return formatTooltip(rec);
567  case Qt::TextAlignmentRole:
568  return column_alignments[index.column()];
569  case Qt::ForegroundRole:
570  // Non-confirmed (but not immature) as transactions are grey
571  if (!rec->status.countsForBalance && rec->status.status != TransactionStatus::Immature) {
572  return COLOR_UNCONFIRMED;
573  }
574  if (index.column() == Amount && (rec->credit + rec->debit) < 0) {
575  return COLOR_NEGATIVE;
576  }
577  if (index.column() == ToAddress) {
578  return addressColor(rec);
579  }
580 
581  // To avoid overriding above conditional formats a default text color for this QTableView is not defined in stylesheet,
582  // so we must always return a color here
583  return COLOR_BLACK;
584  case TypeRole:
585  return rec->type;
586  case DateRole:
587  return QDateTime::fromTime_t(static_cast<uint>(rec->time));
588  case WatchonlyRole:
589  return rec->involvesWatchAddress;
590  case WatchonlyDecorationRole:
591  return txWatchonlyDecoration(rec);
592  case LongDescriptionRole:
593  return priv->describe(rec, walletModel->getOptionsModel()->getDisplayUnit());
594  case AddressRole:
595  return QString::fromStdString(rec->address);
596  case LabelRole:
597  return walletModel->getAddressTableModel()->labelForAddress(QString::fromStdString(rec->address));
598  case AmountRole:
599  return qint64(rec->credit + rec->debit);
600  case TxIDRole:
601  return rec->getTxID();
602  case TxHashRole:
603  return QString::fromStdString(rec->hash.ToString());
604  case ConfirmedRole:
605  return rec->status.countsForBalance;
606  case FormattedAmountRole:
607  // Used for copy/export, so don't include separators
608  return formatTxAmount(rec, false, BitcoinUnits::separatorNever);
609  case StatusRole:
610  return rec->status.status;
611  }
612  return QVariant();
613 }

661 public:

663  TransactionNotification(uint256 hash, ChangeType status, bool showTransaction) : hash(hash), status(status), showTransaction(showTransaction) {}

665  void invoke(QObject* ttm)
666  {
667  QString strHash = QString::fromStdString(hash.GetHex());
668  qDebug() << "NotifyTransactionChanged : " + strHash + " status= " + QString::number(status);
669  QMetaObject::invokeMethod(ttm, "updateTransaction", Qt::QueuedConnection,
670  Q_ARG(QString, strHash),

673  }

684 static void NotifyTransactionChanged(TransactionTableModel* ttm, CWallet* wallet, const uint256& hash, ChangeType status)
685 {
686  // Find transaction in wallet
687  std::map<uint256, CWalletTx>::iterator mi = wallet->mapWallet.find(hash);
688  // Determine whether to show transaction or not (determine this here so that no relocking is needed in GUI thread)
689  bool inWallet = mi != wallet->mapWallet.end();
690  bool showTransaction = (inWallet && TransactionRecord::showTransaction(mi->second));
691 
692  TransactionNotification notification(hash, status, showTransaction);
693 
694  if (fQueueNotifications) {
695  vQueueNotifications.push_back(notification);
696  return;
697  }
698  notification.invoke(ttm);
699 }

701 static void ShowProgress(TransactionTableModel* ttm, const std::string& title, int nProgress)
702 {
703  if (nProgress == 0)
704  fQueueNotifications = true;
705 
706  if (nProgress == 100) {
707  fQueueNotifications = false;
708  if (vQueueNotifications.size() > 10) // prevent balloon spam, show maximum 10 balloons
709  QMetaObject::invokeMethod(ttm, "setProcessingQueuedTransactions", Qt::QueuedConnection, Q_ARG(bool, true));
710  for (unsigned int i = 0; i < vQueueNotifications.size(); ++i) {
711  if (vQueueNotifications.size() - i <= 10)
712  QMetaObject::invokeMethod(ttm, "setProcessingQueuedTransactions", Qt::QueuedConnection, Q_ARG(bool, false));
713 
714  vQueueNotifications[i].invoke(ttm);
715  }
716  std::vector<TransactionNotification>().swap(vQueueNotifications); // clear
717  }
718 }
