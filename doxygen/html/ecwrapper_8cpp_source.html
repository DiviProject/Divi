185 bool CECKey::Verify(const uint256& hash, const std::vector<unsigned char>& vchSig)
186 {
187  if (vchSig.empty())
188  return false;
189 
190  // New versions of OpenSSL will reject non-canonical DER signatures. de/re-serialize first.
191  unsigned char* norm_der = NULL;
192  ECDSA_SIG* norm_sig = ECDSA_SIG_new();
193  const unsigned char* sigptr = &vchSig[0];
194  assert(norm_sig);
195  if (d2i_ECDSA_SIG(&norm_sig, &sigptr, vchSig.size()) == NULL) {
196  /* As of OpenSSL 1.0.0p d2i_ECDSA_SIG frees and nulls the pointer on
197  * error. But OpenSSL's own use of this function redundantly frees the
198  * result. As ECDSA_SIG_free(NULL) is a no-op, and in the absence of a
199  * clear contract for the function behaving the same way is more
200  * conservative.
201  */
202  ECDSA_SIG_free(norm_sig);
203  return false;
204  }
205  int derlen = i2d_ECDSA_SIG(norm_sig, &norm_der);
206  ECDSA_SIG_free(norm_sig);
207  if (derlen <= 0)
208  return false;
209 
210  // -1 = error, 0 = bad sig, 1 = good
211  bool ret = ECDSA_verify(0, (unsigned char*)&hash, sizeof(hash), norm_der, derlen, pkey) == 1;
212  OPENSSL_free(norm_der);
213  return ret;
214 }

216 bool CECKey::Recover(const uint256& hash, const unsigned char* p64, int rec)
217 {
218  if (rec < 0 || rec >= 3)
