116 bool CBloomFilter::IsRelevantAndUpdate(const CTransaction& tx)
117 {
118  bool fFound = false;
119  // Match if the filter contains the hash of tx
120  // for finding tx when they appear in a block
121  if (isFull)
122  return true;
123  if (isEmpty)
124  return false;
125  const uint256& hash = tx.GetHash();
126  if (contains(hash))
127  fFound = true;
128 
129  for (unsigned int i = 0; i < tx.vout.size(); i++) {
130  const CTxOut& txout = tx.vout[i];
131  // Match if the filter contains any arbitrary script data element in any scriptPubKey in tx
132  // If this matches, also add the specific output that was matched.
133  // This means clients don't have to update the filter themselves when a new relevant tx
134  // is discovered in order to find spending transactions, which avoids round-tripping and race conditions.
135  CScript::const_iterator pc = txout.scriptPubKey.begin();
136  vector<unsigned char> data;
137  while (pc < txout.scriptPubKey.end()) {
138  opcodetype opcode;
139  if (!txout.scriptPubKey.GetOp(pc, opcode, data))
140  break;
141  if (data.size() != 0 && contains(data)) {
142  fFound = true;
143  if ((nFlags & BLOOM_UPDATE_MASK) == BLOOM_UPDATE_ALL)
144  insert(COutPoint(hash, i));
145  else if ((nFlags & BLOOM_UPDATE_MASK) == BLOOM_UPDATE_P2PUBKEY_ONLY) {
146  txnouttype type;
147  vector<vector<unsigned char> > vSolutions;
148  if (Solver(txout.scriptPubKey, type, vSolutions) &&
149  (type == TX_PUBKEY || type == TX_MULTISIG))
150  insert(COutPoint(hash, i));
151  }
152  break;
153  }
154  }
155  }
156 
157  if (fFound)
158  return true;
159 
160  BOOST_FOREACH (const CTxIn& txin, tx.vin) {
161  // Match if the filter contains an outpoint tx spends
162  if (contains(txin.prevout))
163  return true;
164 
165  // Match if the filter contains any arbitrary script data element in any scriptSig in tx
166  CScript::const_iterator pc = txin.scriptSig.begin();
167  vector<unsigned char> data;
168  while (pc < txin.scriptSig.end()) {
169  opcodetype opcode;
170  if (!txin.scriptSig.GetOp(pc, opcode, data))
171  break;
172  if (data.size() != 0 && contains(data))
173  return true;
174  }
175  }
176 
177  return false;
178 }
