18 bool NodeLessThan::operator()(const CNodeCombinedStats& left, const CNodeCombinedStats& right) const
19 {
20  const CNodeStats* pLeft = &(left.nodeStats);
21  const CNodeStats* pRight = &(right.nodeStats);
22 
23  if (order == Qt::DescendingOrder)
24  std::swap(pLeft, pRight);
25 
26  switch (column) {
27  case PeerTableModel::Address:
28  return pLeft->addrName.compare(pRight->addrName) < 0;
29  case PeerTableModel::Subversion:
30  return pLeft->cleanSubVer.compare(pRight->cleanSubVer) < 0;
31  case PeerTableModel::Ping:
32  return pLeft->dPingTime < pRight->dPingTime;
33  }
34 
35  return false;
36 }

52  void refreshPeers()
53  {
54  {
55  TRY_LOCK(cs_vNodes, lockNodes);
56  if (!lockNodes) {
57  // skip the refresh if we can't immediately get the lock
58  return;
59  }
60  cachedNodeStats.clear();
61 #if QT_VERSION >= 0x040700
62  cachedNodeStats.reserve(vNodes.size());
63 #endif
64  BOOST_FOREACH (CNode* pnode, vNodes) {
65  CNodeCombinedStats stats;
66  stats.nodeStateStats.nMisbehavior = 0;
67  stats.nodeStateStats.nSyncHeight = -1;
68  stats.fNodeStateStatsAvailable = false;
69  pnode->copyStats(stats.nodeStats);
70  cachedNodeStats.append(stats);
71  }
72  }
73 
74  // Try to retrieve the CNodeStateStats for each node.
75  {
76  TRY_LOCK(cs_main, lockMain);
77  if (lockMain) {
78  BOOST_FOREACH (CNodeCombinedStats& stats, cachedNodeStats)
79  stats.fNodeStateStatsAvailable = GetNodeStateStats(stats.nodeStats.nodeid, stats.nodeStateStats);
80  }
81  }
82 
83  if (sortColumn >= 0)
84  // sort cacheNodeStats (use stable sort to prevent rows jumping around unneceesarily)
85  qStableSort(cachedNodeStats.begin(), cachedNodeStats.end(), NodeLessThan(sortColumn, sortOrder));
86 
87  // build index map
88  mapNodeRows.clear();
89  int row = 0;
90  BOOST_FOREACH (CNodeCombinedStats& stats, cachedNodeStats)
91  mapNodeRows.insert(std::pair<NodeId, int>(stats.nodeStats.nodeid, row++));
92  }
93 
94  int size()
95  {
96  return cachedNodeStats.size();
97  }
98 
99  CNodeCombinedStats* index(int idx)
100  {
101  if (idx >= 0 && idx < cachedNodeStats.size()) {
102  return &cachedNodeStats[idx];
103  } else {
104  return 0;
105  }
106  }
107 };
