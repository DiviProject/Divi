161  bool IsDust(CFeeRate minRelayTxFee) const
162  {
163  // "Dust" is defined in terms of CTransaction::minRelayTxFee, which has units upiv-per-kilobyte.
164  // If you'd pay more than 1/3 in fees to spend something, then we consider it dust.
165  // A typical txout is 34 bytes big, and will need a CTxIn of at least 148 bytes to spend
166  // i.e. total is 148 + 32 = 182 bytes. Default -minrelaytxfee is 10000 upiv per kB
167  // and that means that fee per txout is 182 * 10000 / 1000 = 1820 upiv.
168  // So dust is a txout less than 1820 *3 = 5460 upiv
169  // with default -minrelaytxfee = minRelayTxFee = 10000 upiv per kB.
170  size_t nSize = GetSerializeSize(SER_DISK,0)+148u;
171  return (nValue < 3*minRelayTxFee.GetFee(nSize));
172  }

177  }
