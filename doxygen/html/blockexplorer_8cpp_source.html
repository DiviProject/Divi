493  return true;

498  uint256 hash = uint256S(query.toUtf8().constData());
499

501  BlockMap::iterator iter = mapBlockIndex.find(hash);
502  if (iter != mapBlockIndex.end()) {
503  setBlock(iter->second);
504  return true;
505  }
506 
507  // If the query is neither an integer nor a block hash, assume a transaction hash
508  CTransaction tx;
509  uint256 hashBlock = 0;
510  if (GetTransaction(hash, tx, hashBlock, true)) {
511  setContent(TxToString(hashBlock, tx));

514
