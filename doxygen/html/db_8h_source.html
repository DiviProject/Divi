119  {

122

125  ssKey.reserve(1000);
126  ssKey << key;
127  Dbt datKey(&ssKey[0], ssKey.size());
128 
129  // Read
130  Dbt datValue;
131  datValue.set_flags(DB_DBT_MALLOC);
132  int ret = pdb->get(activeTxn, &datKey, &datValue, 0);
133  memset(datKey.get_data(), 0, datKey.get_size());
134  if (datValue.get_data() == NULL)
135  return false;
136 
137  // Unserialize value
138  try {
139  CDataStream ssValue((char*)datValue.get_data(), (char*)datValue.get_data() + datValue.get_size(), SER_DISK, CLIENT_VERSION);
140  ssValue >> value;
141  } catch (const std::exception&) {
142  return false;
143  }
144 
145  // Clear and free memory
146  memset(datValue.get_data(), 0, datValue.get_size());
147  free(datValue.get_data());

233  int ReadAtCursor(Dbc* pcursor, CDataStream& ssKey, CDataStream& ssValue, unsigned int fFlags = DB_NEXT)
234  {
235  // Read at cursor
236  Dbt datKey;
237  if (fFlags == DB_SET || fFlags == DB_SET_RANGE || fFlags == DB_GET_BOTH || fFlags == DB_GET_BOTH_RANGE) {
238  datKey.set_data(&ssKey[0]);
239  datKey.set_size(ssKey.size());
240  }
241  Dbt datValue;
242  if (fFlags == DB_GET_BOTH || fFlags == DB_GET_BOTH_RANGE) {
243  datValue.set_data(&ssValue[0]);
244  datValue.set_size(ssValue.size());
245  }
246  datKey.set_flags(DB_DBT_MALLOC);
247  datValue.set_flags(DB_DBT_MALLOC);
248  int ret = pcursor->get(&datKey, &datValue, fFlags);
249  if (ret != 0)
250  return ret;
251  else if (datKey.get_data() == NULL || datValue.get_data() == NULL)
252  return 99999;
253 
254  // Convert to streams
255  ssKey.SetType(SER_DISK);
256  ssKey.clear();
257  ssKey.write((char*)datKey.get_data(), datKey.get_size());
258  ssValue.SetType(SER_DISK);
259  ssValue.clear();
260  ssValue.write((char*)datValue.get_data(), datValue.get_size());
261 
262  // Clear and free memory
263  memset(datKey.get_data(), 0, datKey.get_size());
264  memset(datValue.get_data(), 0, datValue.get_size());
265  free(datKey.get_data());
266  free(datValue.get_data());
267  return 0;
268  }
