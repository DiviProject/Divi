546 class CValidationState
547 {
548 private:
549  enum mode_state {
550  MODE_VALID,
551  MODE_INVALID,
552  MODE_ERROR,
553  } mode;
554  int nDoS;
555  std::string strRejectReason;
556  unsigned char chRejectCode;
557  bool corruptionPossible;
558 
559 public:
560  CValidationState() : mode(MODE_VALID), nDoS(0), chRejectCode(0), corruptionPossible(false) {}
561  bool DoS(int level, bool ret = false, unsigned char chRejectCodeIn = 0, std::string strRejectReasonIn = "", bool corruptionIn = false)
562  {
563  chRejectCode = chRejectCodeIn;
564  strRejectReason = strRejectReasonIn;
565  corruptionPossible = corruptionIn;
566  if (mode == MODE_ERROR)
567  return ret;
568  nDoS += level;
569  mode = MODE_INVALID;
570  return ret;
571  }
572  bool Invalid(bool ret = false,
573  unsigned char _chRejectCode = 0,
574  std::string _strRejectReason = "")
575  {
576  return DoS(0, ret, _chRejectCode, _strRejectReason);
577  }
578  bool Error(std::string strRejectReasonIn = "")
579  {
580  if (mode == MODE_VALID)
581  strRejectReason = strRejectReasonIn;
582  mode = MODE_ERROR;
583  return false;
584  }
585  bool Abort(const std::string& msg)
586  {
587  AbortNode(msg);
588  return Error(msg);
589  }
590  bool IsValid() const
591  {
592  return mode == MODE_VALID;
593  }
594  bool IsInvalid() const
595  {
596  return mode == MODE_INVALID;
597  }
598  bool IsError() const
599  {
600  return mode == MODE_ERROR;
601  }
602  bool IsInvalid(int& nDoSOut) const
603  {
604  if (IsInvalid()) {
605  nDoSOut = nDoS;
606  return true;
607  }
608  return false;
609  }
610  bool CorruptionPossible() const
611  {
612  return corruptionPossible;
613  }
614  unsigned char GetRejectCode() const { return chRejectCode; }
615  std::string GetRejectReason() const { return strRejectReason; }
