169  void setint64(int64_t sn)
170  {
171  unsigned char pch[sizeof(sn) + 6];
172  unsigned char* p = pch + 4;
173  bool fNegative;
174  uint64_t n;
175 
176  if (sn < (int64_t)0)
177  {
178  // Since the minimum signed integer cannot be represented as positive so long as its type is signed,
179  // and it's not well-defined what happens if you make it unsigned before negating it,
180  // we instead increment the negative integer by 1, convert it, then increment the (now positive) unsigned integer by 1 to compensate
181  n = -(sn + 1);
182  ++n;
183  fNegative = true;
184  } else {
185  n = sn;
186  fNegative = false;
187  }
188 
189  bool fLeadingZeroes = true;
190  for (int i = 0; i < 8; i++)
191  {
192  unsigned char c = (n >> 56) & 0xff;
193  n <<= 8;
194  if (fLeadingZeroes)
195  {
196  if (c == 0)
197  continue;
198  if (c & 0x80)
199  *p++ = (fNegative ? 0x80 : 0);
200  else if (fNegative)
201  c |= 0x80;
202  fLeadingZeroes = false;
203  }
204  *p++ = c;
205  }
206  unsigned int nSize = p - (pch + 4);
207  pch[0] = (nSize >> 24) & 0xff;
208  pch[1] = (nSize >> 16) & 0xff;
209  pch[2] = (nSize >> 8) & 0xff;
210  pch[3] = (nSize) & 0xff;
211  BN_mpi2bn(pch, p - pch, this);
212  }

272  if (nSize < 4)
273  return 0;

275  BN_bn2mpi(this, &vch[0]);
276  if (vch.size() > 4)
277  vch[4] &= 0x7f;
278  uint256 n = 0;
279  for (unsigned int i = 0, j = vch.size()-1; i < sizeof(n) && j >= 4; i++, j--)

284  void setvch(const std::vector<unsigned char>& vch)
285  {
286  std::vector<unsigned char> vch2(vch.size() + 4);
287  unsigned int nSize = vch.size();
288  // BIGNUM's byte stream format expects 4 bytes of
289  // big endian size data info at the front
290  vch2[0] = (nSize >> 24) & 0xff;
291  vch2[1] = (nSize >> 16) & 0xff;
292  vch2[2] = (nSize >> 8) & 0xff;
293  vch2[3] = (nSize >> 0) & 0xff;
294  // swap data to big endian
295  reverse_copy(vch.begin(), vch.end(), vch2.begin() + 4);
296  BN_mpi2bn(&vch2[0], vch2.size(), this);
297  }

311  // The "compact" format is a representation of a whole
312  // number N using an unsigned 32bit number similar to a
313  // floating point format.
314  // The most significant 8 bits are the unsigned exponent of base 256.
315  // This exponent can be thought of as "number of bytes of N".
316  // The lower 23 bits are the mantissa.
317  // Bit number 24 (0x800000) represents the sign of N.
318  // N = (-1^sign) * mantissa * 256^(exponent-3)
319  //
320  // Satoshi's original implementation used BN_bn2mpi() and BN_mpi2bn().
321  // MPI uses the most significant bit of the first byte as sign.
322  // Thus 0x1234560000 is compact (0x05123456)
323  // and 0xc0de000000 is compact (0x0600c0de)
324  // (0x05c0de00) would be -0x40de000000
325  //
326  // Bitcoin only uses this "compact" format for encoding difficulty
327  // targets, which are unsigned 256bit quantities. Thus, all the
328  // complexities of the sign bit and using base 256 are probably an
329  // implementation accident.
330  //
331  // This implementation directly uses shifts instead of going
332  // through an intermediate MPI representation.
333  CBigNum& SetCompact(unsigned int nCompact)
334  {
335  unsigned int nSize = nCompact >> 24;
336  bool fNegative =(nCompact & 0x00800000) != 0;
337  unsigned int nWord = nCompact & 0x007fffff;
338  if (nSize <= 3)
339  {
340  nWord >>= 8*(3-nSize);
341  BN_set_word(this, nWord);
342  }
343  else
344  {
345  BN_set_word(this, nWord);
346  BN_lshift(this, this, 8*(nSize-3));
347  }
348  BN_set_negative(this, fNegative);
349  return *this;
350  }

352  unsigned int GetCompact() const
353  {
354  unsigned int nSize = BN_num_bytes(this);
355  unsigned int nCompact = 0;
356  if (nSize <= 3)
357  nCompact = BN_get_word(this) << 8*(3-nSize);
358  else
359  {
360  CBigNum bn;
361  BN_rshift(&bn, this, 8*(nSize-3));
362  nCompact = BN_get_word(&bn);
363  }
364  // The 0x00800000 bit denotes the sign.
365  // Thus, if it is already set, divide the mantissa by 256 and increase the exponent.
366  if (nCompact & 0x00800000)
367  {
368  nCompact >>= 8;
369  nSize++;
370  }
371  nCompact |= nSize << 24;
372  nCompact |= (BN_is_negative(this) ? 0x00800000 : 0);
373  return nCompact;
374  }

439  std::string ToString(int nBase=10) const
440  {
441  CAutoBN_CTX pctx;
442  CBigNum bnBase = nBase;
443  CBigNum bn0 = 0;
444  std::string str;
445  CBigNum bn = *this;
446  BN_set_negative(&bn, false);
447  CBigNum dv;
448  CBigNum rem;
449  if (BN_cmp(&bn, &bn0) == 0)
450  return "0";
451  while (BN_cmp(&bn, &bn0) > 0)
452  {
453  if (!BN_div(&dv, &rem, &bn, &bnBase, pctx))
454  throw bignum_error("CBigNum::ToString() : BN_div failed");
455  bn = dv;
456  unsigned int c = rem.getulong();
457  str += "0123456789abcdef"[c];
458  }
459  if (BN_is_negative(this))
460  str += "-";
461  reverse(str.begin(), str.end());
462  return str;
463  }
