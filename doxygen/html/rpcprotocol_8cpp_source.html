139 bool ReadHTTPRequestLine(std::basic_istream<char>& stream, int& proto, string& http_method, string& http_uri)
140 {
141  string str;
142  getline(stream, str);
143 
144  // HTTP request line is space-delimited
145  vector<string> vWords;
146  boost::split(vWords, str, boost::is_any_of(" "));
147  if (vWords.size() < 2)
148  return false;
149 
150  // HTTP methods permitted: GET, POST
151  http_method = vWords[0];
152  if (http_method != "GET" && http_method != "POST")
153  return false;
154 
155  // HTTP URI must be an absolute path, relative to current host
156  http_uri = vWords[1];
157  if (http_uri.size() == 0 || http_uri[0] != '/')
158  return false;
159 
160  // parse proto, if present
161  string strProto = "";
162  if (vWords.size() > 2)
163  strProto = vWords[2];
164 
165  proto = 0;
166  const char* ver = strstr(strProto.c_str(), "HTTP/1.");
167  if (ver != NULL)
168  proto = atoi(ver + 7);
169 
170  return true;
171 }

189 int ReadHTTPHeaders(std::basic_istream<char>& stream, map<string, string>& mapHeadersRet)
190 {
191  int nLen = 0;
192  while (true) {
193  string str;
194  std::getline(stream, str);
195  if (str.empty() || str == "\r")
196  break;
197  string::size_type nColon = str.find(":");
198  if (nColon != string::npos) {
199  string strHeader = str.substr(0, nColon);
200  boost::trim(strHeader);
201  boost::to_lower(strHeader);
202  string strValue = str.substr(nColon + 1);
203  boost::trim(strValue);
204  mapHeadersRet[strHeader] = strValue;
205  if (strHeader == "content-length")
206  nLen = atoi(strValue.c_str());
207  }
208  }
209  return nLen;
210 }

213 int ReadHTTPMessage(std::basic_istream<char>& stream, map<string, string>& mapHeadersRet, string& strMessageRet, int nProto, size_t max_size)
214 {
215  mapHeadersRet.clear();
216  strMessageRet = "";
217 
218  // Read header
219  int nLen = ReadHTTPHeaders(stream, mapHeadersRet);
220  if (nLen < 0 || (size_t)nLen > max_size)
221  return HTTP_INTERNAL_SERVER_ERROR;
222 
223  // Read message
224  if (nLen > 0) {
225  vector<char> vch;
226  size_t ptr = 0;
227  while (ptr < (size_t)nLen) {
228  size_t bytes_to_read = std::min((size_t)nLen - ptr, POST_READ_SIZE);
229  vch.resize(ptr + bytes_to_read);
230  stream.read(&vch[ptr], bytes_to_read);
231  if (!stream) // Connection lost while reading
232  return HTTP_INTERNAL_SERVER_ERROR;
233  ptr += bytes_to_read;
234  }
235  strMessageRet = string(vch.begin(), vch.end());
236  }
237 
238  string sConHdr = mapHeadersRet["connection"];
239 
240  if ((sConHdr != "close") && (sConHdr != "keep-alive")) {
241  if (nProto >= 1)
242  mapHeadersRet["connection"] = "keep-alive";
243  else
244  mapHeadersRet["connection"] = "close";
245  }
246 
247  return HTTP_OK;
248 }
