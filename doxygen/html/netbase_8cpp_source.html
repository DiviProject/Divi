75 void SplitHostPort(std::string in, int& portOut, std::string& hostOut)
76 {
77  size_t colon = in.find_last_of(':');
78  // if a : is found, and it either follows a [...], or no other : is in the string, treat it as port separator
79  bool fHaveColon = colon != in.npos;
80  bool fBracketed = fHaveColon && (in[0] == '[' && in[colon - 1] == ']'); // if there is a colon, and in[0]=='[', colon is not 0, so in[colon-1] is safe
81  bool fMultiColon = fHaveColon && (in.find_last_of(':', colon - 1) != in.npos);
82  if (fHaveColon && (colon == 0 || fBracketed || !fMultiColon)) {
83  int32_t n;
84  if (ParseInt32(in.substr(colon + 1), &n) && n > 0 && n < 0x10000) {
85  in = in.substr(0, colon);
86  portOut = n;
87  }
88  }
89  if (in.size() > 0 && in[0] == '[' && in[in.size() - 1] == ']')
90  hostOut = in.substr(1, in.size() - 2);
91  else
92  hostOut = in;
93 }

129  struct addrinfo aiHint;
130  memset(&aiHint, 0, sizeof(struct addrinfo));
131  aiHint.ai_socktype = SOCK_STREAM;
132  aiHint.ai_protocol = IPPROTO_TCP;
133  aiHint.ai_family = AF_UNSPEC;
134 #ifdef WIN32
135  aiHint.ai_flags = fAllowLookup ? 0 : AI_NUMERICHOST;
136 #else
137  aiHint.ai_flags = fAllowLookup ? AI_ADDRCONFIG : AI_NUMERICHOST;
138 #endif

141 #ifdef HAVE_GETADDRINFO_A
142  struct gaicb gcb, *query = &gcb;

147  if (nErr)

149 
150  do {
151  // Should set the timeout limit to a resonable value to avoid
152  // generating unnecessary checking call during the polling loop,
153  // while it can still response to stop request quick enough.
154  // 2 seconds looks fine in our situation.
155  struct timespec ts = {2, 0};
156  gai_suspend(&query, 1, &ts);
157  boost::this_thread::interruption_point();
158 
159  nErr = gai_error(query);
160  if (0 == nErr)
161  aiRes = query->ar_result;
162  } while (nErr == EAI_INPROGRESS);
163 #else
164  int nErr = getaddrinfo(pszName, NULL, &aiHint, &aiRes);
165 #endif
166  if (nErr)
167  return false;
168 
169  struct addrinfo* aiTrav = aiRes;
170  while (aiTrav != NULL && (nMaxSolutions == 0 || vIP.size() < nMaxSolutions)) {
171  if (aiTrav->ai_family == AF_INET) {
172  assert(aiTrav->ai_addrlen >= sizeof(sockaddr_in));
173  vIP.push_back(CNetAddr(((struct sockaddr_in*)(aiTrav->ai_addr))->sin_addr));
174  }
175 
176  if (aiTrav->ai_family == AF_INET6) {
177  assert(aiTrav->ai_addrlen >= sizeof(sockaddr_in6));
178  vIP.push_back(CNetAddr(((struct sockaddr_in6*)(aiTrav->ai_addr))->sin6_addr));
179  }
180 
181  aiTrav = aiTrav->ai_next;
182  }
183 
184  freeaddrinfo(aiRes);
185 
186  return (vIP.size() > 0);

253 bool static InterruptibleRecv(char* data, size_t len, int timeout, SOCKET& hSocket)
254 {
255  int64_t curTime = GetTimeMillis();
256  int64_t endTime = curTime + timeout;
257  // Maximum time to wait in one select call. It will take up until this time (in millis)
258  // to break off in case of an interruption.
259  const int64_t maxWait = 1000;
260  while (len > 0 && curTime < endTime) {
261  ssize_t ret = recv(hSocket, data, len, 0); // Optimistically try the recv first
262  if (ret > 0) {
263  len -= ret;
264  data += ret;
265  } else if (ret == 0) { // Unexpected disconnection
266  return false;
267  } else { // Other error or blocking
268  int nErr = WSAGetLastError();
269  if (nErr == WSAEINPROGRESS || nErr == WSAEWOULDBLOCK || nErr == WSAEINVAL) {
270  if (!IsSelectableSocket(hSocket)) {
271  return false;
272  }
273  struct timeval tval = MillisToTimeval(std::min(endTime - curTime, maxWait));
274  fd_set fdset;
275  FD_ZERO(&fdset);
276  FD_SET(hSocket, &fdset);
277  int nRet = select(hSocket + 1, &fdset, NULL, NULL, &tval);
278  if (nRet == SOCKET_ERROR) {
279  return false;
280  }
281  } else {
282  return false;
283  }
284  }
285  boost::this_thread::interruption_point();
286  curTime = GetTimeMillis();
287  }
288  return len == 0;
289 }

298 bool static Socks5(string strDest, int port, const ProxyCredentials *auth, SOCKET& hSocket)
299 {
300  LogPrintf("SOCKS5 connecting %s\n", strDest);
301  if (strDest.size() > 255) {
302  CloseSocket(hSocket);
303  return error("Hostname too long");
304  }
305  // Accepted authentication methods
306  std::vector<uint8_t> vSocks5Init;
307  vSocks5Init.push_back(0x05);
308  if (auth) {
309  vSocks5Init.push_back(0x02); // # METHODS
310  vSocks5Init.push_back(0x00); // X'00' NO AUTHENTICATION REQUIRED
311  vSocks5Init.push_back(0x02); // X'02' USERNAME/PASSWORD (RFC1929)
312  } else {
313  vSocks5Init.push_back(0x01); // # METHODS
314  vSocks5Init.push_back(0x00); // X'00' NO AUTHENTICATION REQUIRED
315  }
316  ssize_t ret = send(hSocket, (const char*)begin_ptr(vSocks5Init), vSocks5Init.size(), MSG_NOSIGNAL);
317  if (ret != (ssize_t)vSocks5Init.size()) {
318  CloseSocket(hSocket);
319  return error("Error sending to proxy");
320  }
321  char pchRet1[2];
322  if (!InterruptibleRecv(pchRet1, 2, SOCKS5_RECV_TIMEOUT, hSocket)) {
323  CloseSocket(hSocket);
324  return error("Error reading proxy response");
325  }
326  if (pchRet1[0] != 0x05) {
327  CloseSocket(hSocket);
328  return error("Proxy failed to initialize");
329  }
330  if (pchRet1[1] == 0x02 && auth) {
331  // Perform username/password authentication (as described in RFC1929)
332  std::vector<uint8_t> vAuth;
333  vAuth.push_back(0x01);
334  if (auth->username.size() > 255 || auth->password.size() > 255)
335  return error("Proxy username or password too long");
336  vAuth.push_back(auth->username.size());
337  vAuth.insert(vAuth.end(), auth->username.begin(), auth->username.end());
338  vAuth.push_back(auth->password.size());
339  vAuth.insert(vAuth.end(), auth->password.begin(), auth->password.end());
340  ret = send(hSocket, (const char*)begin_ptr(vAuth), vAuth.size(), MSG_NOSIGNAL);
341  if (ret != (ssize_t)vAuth.size()) {
342  CloseSocket(hSocket);
343  return error("Error sending authentication to proxy");
344  }
345  LogPrint("proxy", "SOCKS5 sending proxy authentication %s:%s\n", auth->username, auth->password);
346  char pchRetA[2];
347  if (!InterruptibleRecv(pchRetA, 2, SOCKS5_RECV_TIMEOUT, hSocket)) {
348  CloseSocket(hSocket);
349  return error("Error reading proxy authentication response");
350  }
351  if (pchRetA[0] != 0x01 || pchRetA[1] != 0x00) {
352  CloseSocket(hSocket);
353  return error("Proxy authentication unsuccesful");
354  }
355  } else if (pchRet1[1] == 0x00) {
356  // Perform no authentication
357  } else {
358  CloseSocket(hSocket);
359  return error("Proxy requested wrong authentication method %02x", pchRet1[1]);
360  }
361  std::vector<uint8_t> vSocks5;
362  vSocks5.push_back(0x05); // VER protocol version
363  vSocks5.push_back(0x01); // CMD CONNECT
364  vSocks5.push_back(0x00); // RSV Reserved
365  vSocks5.push_back(0x03); // ATYP DOMAINNAME
366  vSocks5.push_back(strDest.size()); // Length<=255 is checked at beginning of function
367  vSocks5.insert(vSocks5.end(), strDest.begin(), strDest.end());
368  vSocks5.push_back((port >> 8) & 0xFF);
369  vSocks5.push_back((port >> 0) & 0xFF);
370  ret = send(hSocket, (const char*)begin_ptr(vSocks5), vSocks5.size(), MSG_NOSIGNAL);
371  if (ret != (ssize_t)vSocks5.size()) {
372  CloseSocket(hSocket);
373  return error("Error sending to proxy");
374  }
375  char pchRet2[4];
376  if (!InterruptibleRecv(pchRet2, 4, SOCKS5_RECV_TIMEOUT, hSocket)) {
377  CloseSocket(hSocket);
378  return error("Error reading proxy response");
379  }
380  if (pchRet2[0] != 0x05) {
381  CloseSocket(hSocket);
382  return error("Proxy failed to accept request");
383  }
384  if (pchRet2[1] != 0x00) {
385  CloseSocket(hSocket);
386  switch (pchRet2[1]) {
387  case 0x01:
388  return error("Proxy error: general failure");
389  case 0x02:
390  return error("Proxy error: connection not allowed");
391  case 0x03:
392  return error("Proxy error: network unreachable");
393  case 0x04:
394  return error("Proxy error: host unreachable");
395  case 0x05:
396  return error("Proxy error: connection refused");
397  case 0x06:
398  return error("Proxy error: TTL expired");
399  case 0x07:
400  return error("Proxy error: protocol error");
401  case 0x08:
402  return error("Proxy error: address type not supported");
403  default:
404  return error("Proxy error: unknown");
405  }
406  }
407  if (pchRet2[2] != 0x00) {
408  CloseSocket(hSocket);
409  return error("Error: malformed proxy response");
410  }
411  char pchRet3[256];
412  switch (pchRet2[3]) {
413  case 0x01:
414  ret = InterruptibleRecv(pchRet3, 4, SOCKS5_RECV_TIMEOUT, hSocket);
415  break;
416  case 0x04:
417  ret = InterruptibleRecv(pchRet3, 16, SOCKS5_RECV_TIMEOUT, hSocket);
418  break;
419  case 0x03: {
420  ret = InterruptibleRecv(pchRet3, 1, SOCKS5_RECV_TIMEOUT, hSocket);
421  if (!ret) {
422  CloseSocket(hSocket);
423  return error("Error reading from proxy");
424  }
425  int nRecv = pchRet3[0];
426  ret = InterruptibleRecv(pchRet3, nRecv, SOCKS5_RECV_TIMEOUT, hSocket);
427  break;
428  }
429  default:
430  CloseSocket(hSocket);
431  return error("Error: malformed proxy response");
432  }
433  if (!ret) {
434  CloseSocket(hSocket);
435  return error("Error reading from proxy");
436  }
437  if (!InterruptibleRecv(pchRet3, 2, SOCKS5_RECV_TIMEOUT, hSocket)) {
438  CloseSocket(hSocket);
439  return error("Error reading from proxy");
440  }
441  LogPrintf("SOCKS5 connected %s\n", strDest);
442  return true;
443 }

445 bool static ConnectSocketDirectly(const CService& addrConnect, SOCKET& hSocketRet, int nTimeout)
446 {
447  hSocketRet = INVALID_SOCKET;
448 
449  struct sockaddr_storage sockaddr;
450  socklen_t len = sizeof(sockaddr);
451  if (!addrConnect.GetSockAddr((struct sockaddr*)&sockaddr, &len)) {
452  LogPrintf("Cannot connect to %s: unsupported network\n", addrConnect.ToString());
453  return false;
454  }
455 
456  SOCKET hSocket = socket(((struct sockaddr*)&sockaddr)->sa_family, SOCK_STREAM, IPPROTO_TCP);
457  if (hSocket == INVALID_SOCKET)
458  return false;
459 
460 #ifdef SO_NOSIGPIPE
461  int set = 1;
462  // Different way of disabling SIGPIPE on BSD
463  setsockopt(hSocket, SOL_SOCKET, SO_NOSIGPIPE, (void*)&set, sizeof(int));
464 #endif
465 
466  // Set to non-blocking
467  if (!SetSocketNonBlocking(hSocket, true))
468  return error("ConnectSocketDirectly: Setting socket to non-blocking failed, error %s\n", NetworkErrorString(WSAGetLastError()));
469 
470  if (connect(hSocket, (struct sockaddr*)&sockaddr, len) == SOCKET_ERROR) {
471  int nErr = WSAGetLastError();
472  // WSAEINVAL is here because some legacy version of winsock uses it
473  if (nErr == WSAEINPROGRESS || nErr == WSAEWOULDBLOCK || nErr == WSAEINVAL) {
474  struct timeval timeout = MillisToTimeval(nTimeout);
475  fd_set fdset;
476  FD_ZERO(&fdset);
477  FD_SET(hSocket, &fdset);
478  int nRet = select(hSocket + 1, NULL, &fdset, NULL, &timeout);
479  if (nRet == 0) {
480  LogPrint("net", "connection to %s timeout\n", addrConnect.ToString());
481  CloseSocket(hSocket);
482  return false;
483  }
484  if (nRet == SOCKET_ERROR) {
485  LogPrintf("select() for %s failed: %s\n", addrConnect.ToString(), NetworkErrorString(WSAGetLastError()));
486  CloseSocket(hSocket);
487  return false;
488  }
489  socklen_t nRetSize = sizeof(nRet);
490 #ifdef WIN32
491  if (getsockopt(hSocket, SOL_SOCKET, SO_ERROR, (char*)(&nRet), &nRetSize) == SOCKET_ERROR)
492 #else
493  if (getsockopt(hSocket, SOL_SOCKET, SO_ERROR, &nRet, &nRetSize) == SOCKET_ERROR)
494 #endif
495  {
496  LogPrintf("getsockopt() for %s failed: %s\n", addrConnect.ToString(), NetworkErrorString(WSAGetLastError()));
497  CloseSocket(hSocket);
498  return false;
499  }
500  if (nRet != 0) {
501  LogPrintf("connect() to %s failed after select(): %s\n", addrConnect.ToString(), NetworkErrorString(nRet));
502  CloseSocket(hSocket);
503  return false;
504  }
505  }
506 #ifdef WIN32
507  else if (WSAGetLastError() != WSAEISCONN)
508 #else
509  else
510 #endif
511  {
512  LogPrintf("connect() to %s failed: %s\n", addrConnect.ToString(), NetworkErrorString(WSAGetLastError()));
513  CloseSocket(hSocket);
514  return false;
515  }
516  }
517 
518  hSocketRet = hSocket;
519  return true;
520 }

932 // get canonical identifier of an address' group
933 // no two connections will be attempted to addresses with the same group
934 std::vector<unsigned char> CNetAddr::GetGroup() const
935 {
936  std::vector<unsigned char> vchRet;
937  int nClass = NET_IPV6;
938  int nStartByte = 0;
939  int nBits = 16;
940 
941  // all local addresses belong to the same group
942  if (IsLocal()) {
943  nClass = 255;
944  nBits = 0;
945  }
946 
947  // all unroutable addresses belong to the same group
948  if (!IsRoutable()) {
949  nClass = NET_UNROUTABLE;
950  nBits = 0;
951  }
952  // for IPv4 addresses, '1' + the 16 higher-order bits of the IP
953  // includes mapped IPv4, SIIT translated IPv4, and the well-known prefix
954  else if (IsIPv4() || IsRFC6145() || IsRFC6052()) {
955  nClass = NET_IPV4;
956  nStartByte = 12;
957  }
958  // for 6to4 tunnelled addresses, use the encapsulated IPv4 address
959  else if (IsRFC3964()) {
960  nClass = NET_IPV4;
961  nStartByte = 2;
962  }
963  // for Teredo-tunnelled IPv6 addresses, use the encapsulated IPv4 address
964  else if (IsRFC4380()) {
965  vchRet.push_back(NET_IPV4);
966  vchRet.push_back(GetByte(3) ^ 0xFF);
967  vchRet.push_back(GetByte(2) ^ 0xFF);
968  return vchRet;
969  } else if (IsTor()) {
970  nClass = NET_TOR;
971  nStartByte = 6;
972  nBits = 4;
973  }
974  // for he.net, use /36 groups
975  else if (GetByte(15) == 0x20 && GetByte(14) == 0x01 && GetByte(13) == 0x04 && GetByte(12) == 0x70)
976  nBits = 36;
977  // for the rest of the IPv6 network, use /32 groups
978  else
979  nBits = 32;
980 
981  vchRet.push_back(nClass);
982  while (nBits >= 8) {
983  vchRet.push_back(GetByte(15 - nStartByte));
984  nStartByte++;
985  nBits -= 8;
986  }
987  if (nBits > 0)
988  vchRet.push_back(GetByte(15 - nStartByte) | ((1 << nBits) - 1));
989 
990  return vchRet;
991 }
