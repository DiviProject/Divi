119 template <unsigned int BITS>
120 base_uint<BITS>& base_uint<BITS>::operator/=(const base_uint& b)
121 {
122  base_uint<BITS> div = b; // make a copy, so we can shift.
123  base_uint<BITS> num = *this; // make a copy, so we can subtract.
124  *this = 0; // the quotient.
125  int num_bits = num.bits();
126  int div_bits = div.bits();
127  if (div_bits == 0)
128  throw uint_error("Division by zero");
129  if (div_bits > num_bits) // the result is certainly 0.
130  return *this;
131  int shift = num_bits - div_bits;
132  div <<= shift; // shift so that div and num align.
133  while (shift >= 0) {
134  if (num >= div) {
135  num -= div;
136  pn[shift / 32] |= (1 << (shift & 31)); // set a bit of the result.
137  }
138  div >>= 1; // shift back.
139  shift--;
140  }
141  // num now contains the remainder of the division.
142  return *this;
143 }

197 {

199  if (pn[pos]) {

242 // This implementation directly uses shifts instead of going
243 // through an intermediate MPI representation.
244 arith_uint256& arith_uint256::SetCompact(uint32_t nCompact, bool* pfNegative, bool* pfOverflow)
245 {
246  int nSize = nCompact >> 24;
247  uint32_t nWord = nCompact & 0x007fffff;
248  if (nSize <= 3) {
249  nWord >>= 8 * (3 - nSize);
250  *this = nWord;
251  } else {
252  *this = nWord;
253  *this <<= 8 * (nSize - 3);
254  }
255  if (pfNegative)
256  *pfNegative = nWord != 0 && (nCompact & 0x00800000) != 0;
257  if (pfOverflow)
258  *pfOverflow = nWord != 0 && ((nSize > 34) ||
259  (nWord > 0xff && nSize > 33) ||
260  (nWord > 0xffff && nSize > 32));
261  return *this;
262 }

264 uint32_t arith_uint256::GetCompact(bool fNegative) const
265 {
266  int nSize = (bits() + 7) / 8;
267  uint32_t nCompact = 0;
268  if (nSize <= 3) {
269  nCompact = GetLow64() << 8 * (3 - nSize);
270  } else {
271  arith_uint256 bn = *this >> 8 * (nSize - 3);
272  nCompact = bn.GetLow64();
273  }
274  // The 0x00800000 bit denotes the sign.
275  // Thus, if it is already set, divide the mantissa by 256 and increase the exponent.
276  if (nCompact & 0x00800000) {
277  nCompact >>= 8;
278  nSize++;
279  }
280  assert((nCompact & ~0x007fffff) == 0);
281  assert(nSize < 256);
282  nCompact |= nSize << 24;
283  nCompact |= (fNegative && (nCompact & 0x007fffff) ? 0x00800000 : 0);
284  return nCompact;
285 }
