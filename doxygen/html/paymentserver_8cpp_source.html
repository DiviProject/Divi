113 // Load OpenSSL's list of root certificate authorities
114 //
115 void PaymentServer::LoadRootCAs(X509_STORE* _store)
116 {
117  // Unit tests mostly use this, to pass in fake root CAs:
118  if (_store) {
119  certStore.reset(_store);
120  return;
121  }
122 
123  // Normal execution, use either -rootcertificates or system certs:
124  certStore.reset(X509_STORE_new());
125 
126  // Note: use "-system-" default here so that users can pass -rootcertificates=""
127  // and get 'I don't like X.509 certificates, don't trust anybody' behavior:
128  QString certFile = QString::fromStdString(GetArg("-rootcertificates", "-system-"));
129 
130  if (certFile.isEmpty())
131  return; // Empty store
132 
133  QList<QSslCertificate> certList;
134 
135  if (certFile != "-system-") {
136  certList = QSslCertificate::fromPath(certFile);
137  // Use those certificates when fetching payment requests, too:
138  QSslSocket::setDefaultCaCertificates(certList);
139  } else
140  certList = QSslSocket::systemCaCertificates();
141 
142  int nRootCerts = 0;
143  const QDateTime currentTime = QDateTime::currentDateTime();
144  foreach (const QSslCertificate& cert, certList) {
145  if (currentTime < cert.effectiveDate() || currentTime > cert.expiryDate()) {
146  ReportInvalidCertificate(cert);
147  continue;
148  }
149 #if QT_VERSION >= 0x050000
150  if (cert.isBlacklisted()) {
151  ReportInvalidCertificate(cert);
152  continue;
153  }
154 #endif
155  QByteArray certData = cert.toDer();
156  const unsigned char* data = (const unsigned char*)certData.data();
157 
158  std::unique_ptr<X509, X509Deleter> x509(d2i_X509(0, &data, certData.size()));
159  if (x509 && X509_STORE_add_cert(certStore.get(), x509.get())) {
160  // Note: X509_STORE increases the reference count to the X509 object,
161  // we still have to release our reference to it.
162  ++nRootCerts;
163  } else {
164  ReportInvalidCertificate(cert);
165  continue;
166  }
167  }
168  qWarning() << "PaymentServer::LoadRootCAs : Loaded " << nRootCerts << " root certificates";
169 
170  // Project for another day:
171  // Fetch certificate revocation lists, and add them to certStore.
172  // Issues to consider:
173  // performance (start a thread to fetch in background?)
174  // privacy (fetch through tor/proxy so IP address isn't revealed)
175  // would it be easier to just use a compiled-in blacklist?
176  // or use Qt's blacklist?
177  // "certificate stapling" with server-side caching is more efficient
178 }

181 // Sending to the server is done synchronously, at startup.
182 // If the server isn't already running, startup continues,
183 // and the items in savedPaymentRequest will be handled
184 // when uiReady() is called.

186 // Warning: ipcSendCommandLine() is called early in init,

189 void PaymentServer::ipcParseCommandLine(int argc, char* argv[])
190 {
191  for (int i = 1; i < argc; i++) {
192  QString arg(argv[i]);
193  if (arg.startsWith("-"))
194  continue;
195 
196  // If the divx: URI contains a payment request, we are not able to detect the
197  // network as that would require fetching and parsing the payment request.
198  // That means clicking such an URI which contains a testnet payment request
199  // will start a mainnet instance and throw a "wrong network" error.
200  if (arg.startsWith(BITCOIN_IPC_PREFIX, Qt::CaseInsensitive)) // divx: URI
201  {
202  savedPaymentRequests.append(arg);
203 
204  SendCoinsRecipient r;
205  if (GUIUtil::parseBitcoinURI(arg, &r) && !r.address.isEmpty()) {
206  CBitcoinAddress address(r.address.toStdString());
207 
208  if (address.IsValid(Params(CBaseChainParams::MAIN))) {
209  SelectParams(CBaseChainParams::MAIN);
210  } else if (address.IsValid(Params(CBaseChainParams::TESTNET))) {
211  SelectParams(CBaseChainParams::TESTNET);
212  }
213  }
214  } else if (QFile::exists(arg)) // Filename
215  {
216  savedPaymentRequests.append(arg);
217 
218  PaymentRequestPlus request;
219  if (readPaymentRequestFromFile(arg, request)) {
220  if (request.getDetails().network() == "main") {
221  SelectParams(CBaseChainParams::MAIN);
222  } else if (request.getDetails().network() == "test") {
223  SelectParams(CBaseChainParams::TESTNET);
224  }
225  }
226  } else {
227  // Printing to debug.log is about the best we can do here, the
228  // GUI hasn't started yet so we can't pop up a message box.
229  qWarning() << "PaymentServer::ipcSendCommandLine : Payment request file does not exist: " << arg;
230  }
231  }
232 }
233 
234 //
235 // Sending to the server is done synchronously, at startup.
236 // If the server isn't already running, startup continues,
237 // and the items in savedPaymentRequest will be handled
238 // when uiReady() is called.
239 //
240 bool PaymentServer::ipcSendCommandLine()
241 {
242  bool fResult = false;
243  foreach (const QString& r, savedPaymentRequests) {
244  QLocalSocket* socket = new QLocalSocket();
245  socket->connectToServer(ipcServerName(), QIODevice::WriteOnly);
246  if (!socket->waitForConnected(BITCOIN_IPC_CONNECT_TIMEOUT)) {
247  delete socket;
248  socket = NULL;
249  return false;
250  }
251 
252  QByteArray block;
253  QDataStream out(&block, QIODevice::WriteOnly);
254  out.setVersion(QDataStream::Qt_4_0);
255  out << r;
256  out.device()->seek(0);
257 
258  socket->write(block);
259  socket->flush();
260  socket->waitForBytesWritten(BITCOIN_IPC_CONNECT_TIMEOUT);
261  socket->disconnectFromServer();
262 
263  delete socket;
264  socket = NULL;
265  fResult = true;
266  }
267 
268  return fResult;
269 }

368 void PaymentServer::handleURIOrFile(const QString& s)
369 {
370  if (saveURIs) {
371  savedPaymentRequests.append(s);
372  return;
373  }
374 
375  if (s.startsWith(BITCOIN_IPC_PREFIX, Qt::CaseInsensitive)) // divx: URI
376  {
377 #if QT_VERSION < 0x050000
378  QUrl uri(s);
379 #else
380  QUrlQuery uri((QUrl(s)));
381 #endif
382  if (uri.hasQueryItem("r")) // payment request URI
383  {
384  QByteArray temp;
385  temp.append(uri.queryItemValue("r"));
386  QString decoded = QUrl::fromPercentEncoding(temp);
387  QUrl fetchUrl(decoded, QUrl::StrictMode);
388 
389  if (fetchUrl.isValid()) {
390  qDebug() << "PaymentServer::handleURIOrFile : fetchRequest(" << fetchUrl << ")";
391  fetchRequest(fetchUrl);
392  } else {
393  qWarning() << "PaymentServer::handleURIOrFile : Invalid URL: " << fetchUrl;
394  emit message(tr("URI handling"),
395  tr("Payment request fetch URL is invalid: %1").arg(fetchUrl.toString()),
396  CClientUIInterface::ICON_WARNING);
397  }
398 
399  return;
400  } else // normal URI
401  {
402  SendCoinsRecipient recipient;
403  if (GUIUtil::parseBitcoinURI(s, &recipient)) {
404  CBitcoinAddress address(recipient.address.toStdString());
405  if (!address.IsValid()) {
406  emit message(tr("URI handling"), tr("Invalid payment address %1").arg(recipient.address),
407  CClientUIInterface::MSG_ERROR);
408  } else
409  emit receivedPaymentRequest(recipient);
410  } else
411  emit message(tr("URI handling"),
412  tr("URI cannot be parsed! This can be caused by an invalid DIVX address or malformed URI parameters."),
413  CClientUIInterface::ICON_WARNING);
414 
415  return;
416  }
417  }
418 
419  if (QFile::exists(s)) // payment request file
420  {
421  PaymentRequestPlus request;
422  SendCoinsRecipient recipient;
423  if (!readPaymentRequestFromFile(s, request)) {
424  emit message(tr("Payment request file handling"),
425  tr("Payment request file cannot be read! This can be caused by an invalid payment request file."),
426  CClientUIInterface::ICON_WARNING);
427  } else if (processPaymentRequest(request, recipient))
428  emit receivedPaymentRequest(recipient);
429 
430  return;
431  }
432 }

434 void PaymentServer::handleURIConnection()
435 {
436  QLocalSocket* clientConnection = uriServer->nextPendingConnection();
437 
438  while (clientConnection->bytesAvailable() < (int)sizeof(quint32))
439  clientConnection->waitForReadyRead();
440 
441  connect(clientConnection, SIGNAL(disconnected()),
442  clientConnection, SLOT(deleteLater()));
443 
444  QDataStream in(clientConnection);
445  in.setVersion(QDataStream::Qt_4_0);
446  if (clientConnection->bytesAvailable() < (int)sizeof(quint16)) {
447  return;
448  }
449  QString msg;
450  in >> msg;
451 
452  handleURIOrFile(msg);
453 }

459 bool PaymentServer::readPaymentRequestFromFile(const QString& filename, PaymentRequestPlus& request)
460 {
461  QFile f(filename);
462  if (!f.open(QIODevice::ReadOnly)) {
463  qWarning() << QString("PaymentServer::%1: Failed to open %2").arg(__func__).arg(filename);
464  return false;
465  }
466 
467  // BIP70 DoS protection
468  if (f.size() > BIP70_MAX_PAYMENTREQUEST_SIZE) {
469  qWarning() << QString("PaymentServer::%1: Payment request %2 is too large (%3 bytes, allowed %4 bytes).")
470  .arg(__func__)
471  .arg(filename)
472  .arg(f.size())
473  .arg(BIP70_MAX_PAYMENTREQUEST_SIZE);
474  return false;
475  }
476 
477  QByteArray data = f.readAll();
478 
479  return request.parse(data);
480 }

482 bool PaymentServer::processPaymentRequest(PaymentRequestPlus& request, SendCoinsRecipient& recipient)
483 {
484  if (!optionsModel)
485  return false;
486 
487  if (request.IsInitialized()) {
488  const payments::PaymentDetails& details = request.getDetails();
489 
490  // Payment request network matches client network?
491  if (details.network() != Params().NetworkIDString()) {
492  emit message(tr("Payment request rejected"), tr("Payment request network doesn't match client network."),
493  CClientUIInterface::MSG_ERROR);
494 
495  return false;
496  }
497 
498  // Expired payment request?
499  if (details.has_expires() && (int64_t)details.expires() < GetTime()) {
500  emit message(tr("Payment request rejected"), tr("Payment request has expired."),
501  CClientUIInterface::MSG_ERROR);
502 
503  return false;
504  }
505  } else {
506  emit message(tr("Payment request error"), tr("Payment request is not initialized."),
507  CClientUIInterface::MSG_ERROR);
508 
509  return false;
510  }
511 
512  recipient.paymentRequest = request;
513  recipient.message = GUIUtil::HtmlEscape(request.getDetails().memo());
514 
515  request.getMerchant(certStore.get(), recipient.authenticatedMerchant);
516 
517  QList<std::pair<CScript, CAmount> > sendingTos = request.getPayTo();
518  QStringList addresses;
519 
520  foreach (const PAIRTYPE(CScript, CAmount) & sendingTo, sendingTos) {
521  // Extract and check destination addresses
522  CTxDestination dest;
523  if (ExtractDestination(sendingTo.first, dest)) {
524  // Append destination address
525  addresses.append(QString::fromStdString(CBitcoinAddress(dest).ToString()));
526  } else if (!recipient.authenticatedMerchant.isEmpty()) {
527  // Insecure payments to custom divx addresses are not supported
528  // (there is no good way to tell the user where they are paying in a way
529  // they'd have a chance of understanding).
530  emit message(tr("Payment request rejected"),
531  tr("Unverified payment requests to custom payment scripts are unsupported."),
532  CClientUIInterface::MSG_ERROR);
533  return false;
534  }
535 
536  // Extract and check amounts
537  CTxOut txOut(sendingTo.second, sendingTo.first);
538  if (txOut.IsDust(::minRelayTxFee)) {
539  emit message(tr("Payment request error"), tr("Requested payment amount of %1 is too small (considered dust).").arg(BitcoinUnits::formatWithUnit(optionsModel->getDisplayUnit(), sendingTo.second)),
540  CClientUIInterface::MSG_ERROR);
541 
542  return false;
543  }
544 
545  recipient.amount += sendingTo.second;
546  }
547  // Store addresses and format them to fit nicely into the GUI
548  recipient.address = addresses.join("<br />");
549 
550  if (!recipient.authenticatedMerchant.isEmpty()) {
551  qDebug() << "PaymentServer::processPaymentRequest : Secure payment request from " << recipient.authenticatedMerchant;
552  } else {
553  qDebug() << "PaymentServer::processPaymentRequest : Insecure payment request to " << addresses.join(", ");
554  }
555 
556  return true;
557 }

569 void PaymentServer::fetchPaymentACK(CWallet* wallet, SendCoinsRecipient recipient, QByteArray transaction)
570 {
571  const payments::PaymentDetails& details = recipient.paymentRequest.getDetails();
572  if (!details.has_payment_url())
573  return;
574 
575  QNetworkRequest netRequest;
576  netRequest.setAttribute(QNetworkRequest::User, BIP70_MESSAGE_PAYMENTACK);
577  netRequest.setUrl(QString::fromStdString(details.payment_url()));
578  netRequest.setHeader(QNetworkRequest::ContentTypeHeader, BIP71_MIMETYPE_PAYMENT);
579  netRequest.setRawHeader("User-Agent", CLIENT_NAME.c_str());
580  netRequest.setRawHeader("Accept", BIP71_MIMETYPE_PAYMENTACK);
581 
582  payments::Payment payment;
583  payment.set_merchant_data(details.merchant_data());
584  payment.add_transactions(transaction.data(), transaction.size());
585 
586  // Create a new refund address, or re-use:
587  QString account = tr("Refund from %1").arg(recipient.authenticatedMerchant);
588  std::string strAccount = account.toStdString();
589  set<CTxDestination> refundAddresses = wallet->GetAccountAddresses(strAccount);
590  if (!refundAddresses.empty()) {
591  CScript s = GetScriptForDestination(*refundAddresses.begin());
592  payments::Output* refund_to = payment.add_refund_to();
593  refund_to->set_script(&s[0], s.size());
594  } else {
595  CPubKey newKey;
596  if (wallet->GetKeyFromPool(newKey)) {
597  CKeyID keyID = newKey.GetID();
598  wallet->SetAddressBook(keyID, strAccount, "refund");
599 
600  CScript s = GetScriptForDestination(keyID);
601  payments::Output* refund_to = payment.add_refund_to();
602  refund_to->set_script(&s[0], s.size());
603  } else {
604  // This should never happen, because sending coins should have
605  // just unlocked the wallet and refilled the keypool.
606  qWarning() << "PaymentServer::fetchPaymentACK : Error getting refund key, refund_to not set";
607  }
608  }
609 
610  int length = payment.ByteSize();
611  netRequest.setHeader(QNetworkRequest::ContentLengthHeader, length);
612  QByteArray serData(length, '\0');
613  if (payment.SerializeToArray(serData.data(), length)) {
614  netManager->post(netRequest, serData);
615  } else {
616  // This should never happen, either.
617  qWarning() << "PaymentServer::fetchPaymentACK : Error serializing payment message";
618  }
619 }

621 void PaymentServer::netRequestFinished(QNetworkReply* reply)
622 {
623  reply->deleteLater();
624 
625  // BIP70 DoS protection
626  if (reply->size() > BIP70_MAX_PAYMENTREQUEST_SIZE) {
627  QString msg = tr("Payment request %1 is too large (%2 bytes, allowed %3 bytes).")
628  .arg(reply->request().url().toString())
629  .arg(reply->size())
630  .arg(BIP70_MAX_PAYMENTREQUEST_SIZE);
631 
632  qWarning() << QString("PaymentServer::%1:").arg(__func__) << msg;
633  emit message(tr("Payment request DoS protection"), msg, CClientUIInterface::MSG_ERROR);
634  return;
635  }
636 
637  if (reply->error() != QNetworkReply::NoError) {
638  QString msg = tr("Error communicating with %1: %2")
639  .arg(reply->request().url().toString())
640  .arg(reply->errorString());
641 
642  qWarning() << "PaymentServer::netRequestFinished: " << msg;
643  emit message(tr("Payment request error"), msg, CClientUIInterface::MSG_ERROR);
644  return;
645  }
646 
647  QByteArray data = reply->readAll();
648 
649  QString requestType = reply->request().attribute(QNetworkRequest::User).toString();
650  if (requestType == BIP70_MESSAGE_PAYMENTREQUEST) {
651  PaymentRequestPlus request;
652  SendCoinsRecipient recipient;
653  if (!request.parse(data)) {
654  qWarning() << "PaymentServer::netRequestFinished : Error parsing payment request";
655  emit message(tr("Payment request error"),
656  tr("Payment request cannot be parsed!"),
657  CClientUIInterface::MSG_ERROR);
658  } else if (processPaymentRequest(request, recipient))
659  emit receivedPaymentRequest(recipient);
660 
661  return;
662  } else if (requestType == BIP70_MESSAGE_PAYMENTACK) {
663  payments::PaymentACK paymentACK;
664  if (!paymentACK.ParseFromArray(data.data(), data.size())) {
665  QString msg = tr("Bad response from server %1")
666  .arg(reply->request().url().toString());
667 
668  qWarning() << "PaymentServer::netRequestFinished : " << msg;
669  emit message(tr("Payment request error"), msg, CClientUIInterface::MSG_ERROR);
670  } else {
