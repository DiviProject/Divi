48  }

52  if (interface->isValid()) {
53  mode = Freedesktop;
54  }

58  if (MacNotificationHandler::instance()->hasUserNotificationCenterSupport()) {
59  mode = UserNotificationCenter;

61  // Check if Growl is installed (based on Qt's tray icon implementation)
62  CFURLRef cfurl;
63  OSStatus status = LSGetApplicationForInfo(kLSUnknownType, kLSUnknownCreator, CFSTR("growlTicket"), kLSRolesAll, 0, &cfurl);
64  if (status != kLSApplicationNotFoundErr) {
65  CFBundleRef bundle = CFBundleCreate(0, cfurl);
66  if (CFStringCompare(CFBundleGetIdentifier(bundle), CFSTR("com.Growl.GrowlHelperApp"), kCFCompareCaseInsensitive | kCFCompareBackwards) == kCFCompareEqualTo) {
67  if (CFStringHasSuffix(CFURLGetString(cfurl), CFSTR("/Growl.app/")))
68  mode = Growl13;
69  else
70  mode = Growl12;
71  }
72  CFRelease(cfurl);
73  CFRelease(bundle);

113 // Image configuration settings
114 const int CHANNELS = 4;
115 const int BYTES_PER_PIXEL = 4;
116 const int BITS_PER_SAMPLE = 8;
117 
118 FreedesktopImage::FreedesktopImage(const QImage& img) : width(img.width()),
119  height(img.height()),
120  stride(img.width() * BYTES_PER_PIXEL),
121  hasAlpha(true),
122  channels(CHANNELS),
123  bitsPerSample(BITS_PER_SAMPLE)
124 {
125  // Convert 00xAARRGGBB to RGBA bytewise (endian-independent) format
126  QImage tmp = img.convertToFormat(QImage::Format_ARGB32);
127  const uint32_t* data = reinterpret_cast<const uint32_t*>(tmp.bits());
128 
129  unsigned int num_pixels = width * height;
130  image.resize(num_pixels * BYTES_PER_PIXEL);
131 
132  for (unsigned int ptr = 0; ptr < num_pixels; ++ptr) {
133  image[ptr * BYTES_PER_PIXEL + 0] = data[ptr] >> 16; // R
134  image[ptr * BYTES_PER_PIXEL + 1] = data[ptr] >> 8; // G
135  image[ptr * BYTES_PER_PIXEL + 2] = data[ptr]; // B
136  image[ptr * BYTES_PER_PIXEL + 3] = data[ptr] >> 24; // A
137  }
138 }

167 void Notificator::notifyDBus(Class cls, const QString& title, const QString& text, const QIcon& icon, int millisTimeout)
168 {
169  Q_UNUSED(cls);
170  // Arguments for DBus call:
171  QList<QVariant> args;
172 
173  // Program Name:
174  args.append(programName);
175 
176  // Unique ID of this notification type:
177  args.append(0U);
178 
179  // Application Icon, empty string
180  args.append(QString());
181 
182  // Summary
183  args.append(title);
184 
185  // Body
186  args.append(text);
187 
188  // Actions (none, actions are deprecated)
189  QStringList actions;
190  args.append(actions);
191 
192  // Hints
193  QVariantMap hints;
194 
195  // If no icon specified, set icon based on class
196  QIcon tmpicon;
197  if (icon.isNull()) {
198  QStyle::StandardPixmap sicon = QStyle::SP_MessageBoxQuestion;
199  switch (cls) {
200  case Information:
201  sicon = QStyle::SP_MessageBoxInformation;
202  break;
203  case Warning:
204  sicon = QStyle::SP_MessageBoxWarning;
205  break;
206  case Critical:
207  sicon = QStyle::SP_MessageBoxCritical;
208  break;
209  default:
210  break;
211  }
212  tmpicon = QApplication::style()->standardIcon(sicon);
213  } else {
214  tmpicon = icon;
215  }
216  hints["icon_data"] = FreedesktopImage::toVariant(tmpicon.pixmap(FREEDESKTOP_NOTIFICATION_ICON_SIZE).toImage());
217  args.append(hints);
218 
219  // Timeout (in msec)
220  args.append(millisTimeout);
221 
222  // "Fire and forget"
223  interface->callWithArgumentList(QDBus::NoBlock, "Notify", args);
224 }

305 void Notificator::notify(Class cls, const QString& title, const QString& text, const QIcon& icon, int millisTimeout)
306 {
307  switch (mode) {
308 #ifdef USE_DBUS
309  case Freedesktop:
310  notifyDBus(cls, title, text, icon, millisTimeout);
311  break;
312 #endif
313  case QSystemTray:
314  notifySystray(cls, title, text, icon, millisTimeout);
315  break;
316 #ifdef Q_OS_MAC
317  case UserNotificationCenter:
318  notifyMacUserNotificationCenter(cls, title, text, icon);
319  break;
320  case Growl12:
321  case Growl13:
322  notifyGrowl(cls, title, text, icon);
323  break;
324 #endif
325  default:
326  if (cls == Critical) {
327  // Fall back to old fashioned pop-up dialog if critical and no other notification available
328  QMessageBox::critical(parent, title, text, QMessageBox::Ok, QMessageBox::Ok);
329  }
330  break;
331  }
