55 /* Determine address type from address purpose */
56 static AddressTableEntry::Type translateTransactionType(const QString& strPurpose, bool isMine)
57 {
58  AddressTableEntry::Type addressType = AddressTableEntry::Hidden;
59  // "refund" addresses aren't shown, and change addresses aren't in mapAddressBook at all.
60  if (strPurpose == "send")
61  addressType = AddressTableEntry::Sending;
62  else if (strPurpose == "receive")
63  addressType = AddressTableEntry::Receiving;
64  else if (strPurpose == "unknown" || strPurpose == "") // if purpose not set, guess
65  addressType = (isMine ? AddressTableEntry::Receiving : AddressTableEntry::Sending);
66  return addressType;
67 }

79  void refreshAddressTable()
80  {
81  cachedAddressTable.clear();
82  {
83  LOCK(wallet->cs_wallet);
84  BOOST_FOREACH (const PAIRTYPE(CTxDestination, CAddressBookData) & item, wallet->mapAddressBook) {
85  const CBitcoinAddress& address = item.first;
86  bool fMine = IsMine(*wallet, address.Get());
87  AddressTableEntry::Type addressType = translateTransactionType(
88  QString::fromStdString(item.second.purpose), fMine);
89  const std::string& strName = item.second.name;
90  cachedAddressTable.append(AddressTableEntry(addressType,
91  QString::fromStdString(strName),
92  QString::fromStdString(address.ToString())));
93  }
94  }
95  // qLowerBound() and qUpperBound() require our cachedAddressTable list to be sorted in asc order
96  // Even though the map is already sorted this re-sorting step is needed because the originating map
97  // is sorted by binary address, not by base58() address.
98  qSort(cachedAddressTable.begin(), cachedAddressTable.end(), AddressTableEntryLessThan());
99  }

101  void updateEntry(const QString& address, const QString& label, bool isMine, const QString& purpose, int status)
102  {
103  // Find address / label in model
104  QList<AddressTableEntry>::iterator lower = qLowerBound(
105  cachedAddressTable.begin(), cachedAddressTable.end(), address, AddressTableEntryLessThan());
106  QList<AddressTableEntry>::iterator upper = qUpperBound(
107  cachedAddressTable.begin(), cachedAddressTable.end(), address, AddressTableEntryLessThan());
108  int lowerIndex = (lower - cachedAddressTable.begin());
109  int upperIndex = (upper - cachedAddressTable.begin());
110  bool inModel = (lower != upper);
111  AddressTableEntry::Type newEntryType = translateTransactionType(purpose, isMine);
112 
113  switch (status) {
114  case CT_NEW:
115  if (inModel) {
116  qWarning() << "AddressTablePriv::updateEntry : Warning: Got CT_NEW, but entry is already in model";
117  break;
118  }
119  parent->beginInsertRows(QModelIndex(), lowerIndex, lowerIndex);
120  cachedAddressTable.insert(lowerIndex, AddressTableEntry(newEntryType, label, address));
121  parent->endInsertRows();
122  break;
123  case CT_UPDATED:
124  if (!inModel) {
125  qWarning() << "AddressTablePriv::updateEntry : Warning: Got CT_UPDATED, but entry is not in model";
126  break;
127  }
128  lower->type = newEntryType;
129  lower->label = label;
130  parent->emitDataChanged(lowerIndex);
131  break;
132  case CT_DELETED:
133  if (!inModel) {
134  qWarning() << "AddressTablePriv::updateEntry : Warning: Got CT_DELETED, but entry is not in model";
135  break;
136  }
137  parent->beginRemoveRows(QModelIndex(), lowerIndex, upperIndex - 1);
138  cachedAddressTable.erase(lower, upper);
139  parent->endRemoveRows();
140  break;
141  }
142  }

257 bool AddressTableModel::setData(const QModelIndex& index, const QVariant& value, int role)
258 {
259  if (!index.isValid())
260  return false;
261  AddressTableEntry* rec = static_cast<AddressTableEntry*>(index.internalPointer());
262  std::string strPurpose = (rec->type == AddressTableEntry::Sending ? "send" : "receive");
263  editStatus = OK;
264 
265  if (role == Qt::EditRole) {
266  LOCK(wallet->cs_wallet); /* For SetAddressBook / DelAddressBook */
267  CTxDestination curAddress = CBitcoinAddress(rec->address.toStdString()).Get();
268  if (index.column() == Label) {
269  // Do nothing, if old label == new label
270  if (rec->label == value.toString()) {
271  editStatus = NO_CHANGES;
272  return false;
273  }
274  wallet->SetAddressBook(curAddress, value.toString().toStdString(), strPurpose);
275  } else if (index.column() == Address) {
276  CTxDestination newAddress = CBitcoinAddress(value.toString().toStdString()).Get();
277  // Refuse to set invalid address, set error status and return false
278  if (boost::get<CNoDestination>(&newAddress)) {
279  editStatus = INVALID_ADDRESS;
280  return false;
281  }
282  // Do nothing, if old address == new address
283  else if (newAddress == curAddress) {
284  editStatus = NO_CHANGES;
285  return false;
286  }
287  // Check for duplicate addresses to prevent accidental deletion of addresses, if you try
288  // to paste an existing address over another address (with a different label)
289  else if (wallet->mapAddressBook.count(newAddress)) {
290  editStatus = DUPLICATE_ADDRESS;
291  return false;
292  }
293  // Double-check that we're not overwriting a receiving address
294  else if (rec->type == AddressTableEntry::Sending) {
295  // Remove old entry
296  wallet->DelAddressBook(curAddress);
297  // Add new entry with new address
298  wallet->SetAddressBook(newAddress, rec->label.toStdString(), strPurpose);
299  }
300  }
301  return true;
302  }
303  return false;
304 }

362 QString AddressTableModel::addRow(const QString& type, const QString& label, const QString& address)
363 {
364  std::string strLabel = label.toStdString();
365  std::string strAddress = address.toStdString();
366 
367  editStatus = OK;
368 
369  if (type == Send) {
370  if (!walletModel->validateAddress(address)) {
371  editStatus = INVALID_ADDRESS;
372  return QString();
373  }
374  // Check for duplicate addresses
375  {
376  LOCK(wallet->cs_wallet);
377  if (wallet->mapAddressBook.count(CBitcoinAddress(strAddress).Get())) {
378  editStatus = DUPLICATE_ADDRESS;
379  return QString();
380  }
381  }
382  } else if (type == Receive) {
383  // Generate a new address to associate with given label
384  CPubKey newKey;
385  if (!wallet->GetKeyFromPool(newKey)) {
386  WalletModel::UnlockContext ctx(walletModel->requestUnlock(true));
387  if (!ctx.isValid()) {
388  // Unlock wallet failed or was cancelled
389  editStatus = WALLET_UNLOCK_FAILURE;
390  return QString();
391  }
392  if (!wallet->GetKeyFromPool(newKey)) {
393  editStatus = KEY_GENERATION_FAILURE;
394  return QString();
395  }
396  }
397  strAddress = CBitcoinAddress(newKey.GetID()).ToString();
398  } else {
399  return QString();
400  }
401 
402  // Add entry
403  {
404  LOCK(wallet->cs_wallet);
405  wallet->SetAddressBook(CBitcoinAddress(strAddress).Get(), strLabel,
406  (type == Send ? "send" : "receive"));
407  }
408  return QString::fromStdString(strAddress);
409 }
