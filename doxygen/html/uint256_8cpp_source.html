89 template <unsigned int BITS>
90 base_uint<BITS>& base_uint<BITS>::operator/=(const base_uint& b)
91 {
92  base_uint<BITS> div = b; // make a copy, so we can shift.
93  base_uint<BITS> num = *this; // make a copy, so we can subtract.
94  *this = 0; // the quotient.
95  int num_bits = num.bits();
96  int div_bits = div.bits();
97  if (div_bits == 0)
98  throw uint_error("Division by zero");
99  if (div_bits > num_bits) // the result is certainly 0.
100  return *this;
101  int shift = num_bits - div_bits;
102  div <<= shift; // shift so that div and nun align.
103  while (shift >= 0) {
104  if (num >= div) {
105  num -= div;
106  pn[shift / 32] |= (1 << (shift & 31)); // set a bit of the result.
107  }
108  div >>= 1; // shift back.
109  shift--;
110  }
111  // num now contains the remainder of the division.
112  return *this;
113 }

163 void base_uint<BITS>::SetHex(const char* psz)
164 {

166 
167  // skip leading spaces
168  while (isspace(*psz))
169  psz++;
170 
171  // skip 0x
172  if (psz[0] == '0' && tolower(psz[1]) == 'x')
173  psz += 2;
174 
175  // hex string to uint

178  psz++;

272 uint256& uint256::SetCompact(uint32_t nCompact, bool* pfNegative, bool* pfOverflow)
273 {
274  int nSize = nCompact >> 24;
275  uint32_t nWord = nCompact & 0x007fffff;
276  if (nSize <= 3) {
277  nWord >>= 8 * (3 - nSize);
278  *this = nWord;
279  } else {
280  *this = nWord;
281  *this <<= 8 * (nSize - 3);
282  }
283  if (pfNegative)
284  *pfNegative = nWord != 0 && (nCompact & 0x00800000) != 0;
285  if (pfOverflow)
286  *pfOverflow = nWord != 0 && ((nSize > 34) ||
287  (nWord > 0xff && nSize > 33) ||
288  (nWord > 0xffff && nSize > 32));
289  return *this;
290 }

292 uint32_t uint256::GetCompact(bool fNegative) const
293 {
294  int nSize = (bits() + 7) / 8;
295  uint32_t nCompact = 0;
296  if (nSize <= 3) {
297  nCompact = GetLow64() << 8 * (3 - nSize);
298  } else {
299  uint256 bn = *this >> 8 * (nSize - 3);
300  nCompact = bn.GetLow64();
301  }
302  // The 0x00800000 bit denotes the sign.
303  // Thus, if it is already set, divide the mantissa by 256 and increase the exponent.
304  if (nCompact & 0x00800000) {
305  nCompact >>= 8;
306  nSize++;
307  }
308  assert((nCompact & ~0x007fffff) == 0);
309  assert(nSize < 256);
310  nCompact |= nSize << 24;
311  nCompact |= (fNegative && (nCompact & 0x007fffff) ? 0x00800000 : 0);
312  return nCompact;
313 }

357 uint64_t uint256::GetHash(const uint256& salt) const
358 {
359  uint32_t a, b, c;
360  a = b = c = 0xdeadbeef + (WIDTH << 2);
361 
362  a += pn[0] ^ salt.pn[0];
363  b += pn[1] ^ salt.pn[1];
364  c += pn[2] ^ salt.pn[2];
365  HashMix(a, b, c);
366  a += pn[3] ^ salt.pn[3];
367  b += pn[4] ^ salt.pn[4];
368  c += pn[5] ^ salt.pn[5];

371  b += pn[7] ^ salt.pn[7];
372  HashFinal(a, b, c);
373
