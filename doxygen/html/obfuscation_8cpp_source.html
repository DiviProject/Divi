54

59

62

65  LogPrintf("dsa -- not a Masternode! \n");
66  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
67

72  CTransaction txCollateral;

74

77  errorID = ERR_MN_LIST;
78  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
79  return;
80  }
81 
82  if (sessionUsers == 0) {
83  if (pmn->nLastDsq != 0 &&
84  pmn->nLastDsq + mnodeman.CountEnabled(ActiveProtocol()) / 5 > mnodeman.nDsqCount) {
85  LogPrintf("dsa -- last dsq too recent, must wait. %s \n", pfrom->addr.ToString());
86  errorID = ERR_RECENT;

88  return;
89  }
90  }
91

93  LogPrintf("dsa -- not compatible with existing transactions! \n");
94  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
95  return;
96  } else {
97  LogPrintf("dsa -- is compatible, please submit! \n");
98  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_ACCEPTED, errorID);
99  return;
100  }

103  TRY_LOCK(cs_obfuscation, lockRecv);
104  if (!lockRecv) return;
105 
106  if (pfrom->nVersion < ActiveProtocol()) {
107  return;
108  }
109

111  vRecv >> dsq;
112 
113  CService addr;
114  if (!dsq.GetAddress(addr)) return;
115  if (!dsq.CheckSignature()) return;
116 
117  if (dsq.IsExpired()) return;
118

120  if (pmn == NULL) return;
121

123  if (dsq.ready) {
124  if (!pSubmittedToMasternode) return;
125  if ((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)addr) {

159  if (pfrom->nVersion < ActiveProtocol()) {
160  LogPrintf("dsi -- incompatible version! \n");
161  errorID = ERR_VERSION;
162  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);

164  return;
165  }
166

168  LogPrintf("dsi -- not a Masternode! \n");
169  errorID = ERR_NOT_A_MN;
170  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
171 
172  return;
173  }

176  CAmount nAmount;
177  CTransaction txCollateral;
178  std::vector<CTxOut> out;
179  vRecv >> in >> nAmount >> txCollateral >> out;
180 
181  //do we have enough users in the current session?
182  if (!IsSessionReady()) {
183  LogPrintf("dsi -- session not complete! \n");
184  errorID = ERR_SESSION;
185  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
186  return;
187  }
188 
189  //do we have the same denominations as the current session?
190  if (!IsCompatibleWithEntries(out)) {
191  LogPrintf("dsi -- not compatible with existing transactions! \n");
192  errorID = ERR_EXISTING_TX;
193  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
194  return;
195  }
196 
197  //check it like a transaction
198  {
199  CAmount nValueIn = 0;
200  CAmount nValueOut = 0;
201  bool missingTx = false;
202 
203  CValidationState state;
204  CMutableTransaction tx;
205 
206  BOOST_FOREACH (const CTxOut o, out) {
207  nValueOut += o.nValue;
208  tx.vout.push_back(o);
209 
210  if (o.scriptPubKey.size() != 25) {
211  LogPrintf("dsi - non-standard pubkey detected! %s\n", o.scriptPubKey.ToString());
212  errorID = ERR_NON_STANDARD_PUBKEY;
213  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
214  return;
215  }

217  LogPrintf("dsi - invalid script! %s\n", o.scriptPubKey.ToString());

220  return;
221  }
222  }
223 
224  BOOST_FOREACH (const CTxIn i, in) {
225  tx.vin.push_back(i);
226 
227  LogPrint("obfuscation", "dsi -- tx in %s\n", i.ToString());
228 
229  CTransaction tx2;
230  uint256 hash;
231  if (GetTransaction(i.prevout.hash, tx2, hash, true)) {
232  if (tx2.vout.size() > i.prevout.n) {
233  nValueIn += tx2.vout[i.prevout.n].nValue;
234  }
235  } else {
236  missingTx = true;
237  }
238  }
239 
240  if (nValueIn > OBFUSCATION_POOL_MAX) {
241  LogPrintf("dsi -- more than Obfuscation pool max! %s\n", tx.ToString());
242  errorID = ERR_MAXIMUM;
243  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
244  return;
245  }
246 
247  if (!missingTx) {
248  if (nValueIn - nValueOut > nValueIn * .01) {
249  LogPrintf("dsi -- fees are too high! %s\n", tx.ToString());
250  errorID = ERR_FEES;
251  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
252  return;
253  }
254  } else {
255  LogPrintf("dsi -- missing input tx! %s\n", tx.ToString());
256  errorID = ERR_MISSING_TX;
257  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
258  return;
259  }
260 
261  {
262  LOCK(cs_main);
263  if (!AcceptableInputs(mempool, state, CTransaction(tx), false, NULL, false, true)) {
264  LogPrintf("dsi -- transaction not valid! \n");
265  errorID = ERR_INVALID_TX;
266  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
267  return;
268  }
269  }
270  }
271 
272  if (AddEntry(in, nAmount, txCollateral, out, errorID)) {
273  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_ACCEPTED, errorID);
274  Check();
275 
276  RelayStatus(sessionID, GetState(), GetEntriesCount(), MASTERNODE_RESET);
277  } else {
278  pfrom->PushMessage("dssu", sessionID, GetState(), GetEntriesCount(), MASTERNODE_REJECTED, errorID);
279  }
280 
281  } else if (strCommand == "dssu") { //Obfuscation status update
282  if (pfrom->nVersion < ActiveProtocol()) {
283  return;
284  }
285 
286  if (!pSubmittedToMasternode) return;
287  if ((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)pfrom->addr) {
288  //LogPrintf("dssu - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString(), pfrom->addr.ToString());
289  return;
290  }
291 
292  int sessionIDMessage;
293  int state;
294  int entriesCount;
295  int accepted;
296  int errorID;
297  vRecv >> sessionIDMessage >> state >> entriesCount >> accepted >> errorID;
298 
299  LogPrint("obfuscation", "dssu - state: %i entriesCount: %i accepted: %i error: %s \n", state, entriesCount, accepted, GetMessageByID(errorID));
300 
301  if ((accepted != 1 && accepted != 0) && sessionID != sessionIDMessage) {
302  LogPrintf("dssu - message doesn't match current Obfuscation session %d %d\n", sessionID, sessionIDMessage);
303  return;
304  }
305 
306  StatusUpdate(state, entriesCount, accepted, errorID, sessionIDMessage);
307 
308  } else if (strCommand == "dss") { //Obfuscation Sign Final Tx
309 
310  if (pfrom->nVersion < ActiveProtocol()) {
311  return;
312  }
313 
314  vector<CTxIn> sigs;
315  vRecv >> sigs;
316 
317  bool success = false;
318  int count = 0;
319 
320  BOOST_FOREACH (const CTxIn item, sigs) {
321  if (AddScriptSig(item)) success = true;
322  LogPrint("obfuscation", " -- sigs count %d %d\n", (int)sigs.size(), count);
323  count++;
324  }
325 
326  if (success) {
327  obfuScationPool.Check();
328  RelayStatus(obfuScationPool.sessionID, obfuScationPool.GetState(), obfuScationPool.GetEntriesCount(), MASTERNODE_RESET);
329  }
330  } else if (strCommand == "dsf") { //Obfuscation Final tx
331  if (pfrom->nVersion < ActiveProtocol()) {
332  return;
333  }
334 
335  if (!pSubmittedToMasternode) return;
336  if ((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)pfrom->addr) {
337  //LogPrintf("dsc - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString(), pfrom->addr.ToString());
338  return;
339  }
340 
341  int sessionIDMessage;
342  CTransaction txNew;
343  vRecv >> sessionIDMessage >> txNew;
344 
345  if (sessionID != sessionIDMessage) {
346  LogPrint("obfuscation", "dsf - message doesn't match current Obfuscation session %d %d\n", sessionID, sessionIDMessage);
347  return;
348  }
349 
350  //check to see if input is spent already? (and probably not confirmed)
351  SignFinalTransaction(txNew, pfrom);
352 
353  } else if (strCommand == "dsc") { //Obfuscation Complete
354 
355  if (pfrom->nVersion < ActiveProtocol()) {
356  return;
357  }
358 
359  if (!pSubmittedToMasternode) return;
360  if ((CNetAddr)pSubmittedToMasternode->addr != (CNetAddr)pfrom->addr) {
361  //LogPrintf("dsc - message doesn't match current Masternode - %s != %s\n", pSubmittedToMasternode->addr.ToString(), pfrom->addr.ToString());
362  return;
363  }
364 
365  int sessionIDMessage;
366  bool error;
367  int errorID;
368  vRecv >> sessionIDMessage >> error >> errorID;
369 
370  if (sessionID != sessionIDMessage) {
371  LogPrint("obfuscation", "dsc - message doesn't match current Obfuscation session %d %d\n", obfuScationPool.sessionID, sessionIDMessage);
372  return;
373  }
374 
375  obfuScationPool.CompletedTransaction(error, errorID);
376  }

516 //
517 void CObfuscationPool::Check()
518 {
519  if (fMasterNode) LogPrint("obfuscation", "CObfuscationPool::Check() - entries count %lu\n", entries.size());
520  //printf("CObfuscationPool::Check() %d - %d - %d\n", state, anonTx.CountEntries(), GetTimeMillis()-lastTimeChanged);
521 
522  if (fMasterNode) {
523  LogPrint("obfuscation", "CObfuscationPool::Check() - entries count %lu\n", entries.size());
524 
525  // If entries is full, then move on to the next phase
526  if (state == POOL_STATUS_ACCEPTING_ENTRIES && (int)entries.size() >= GetMaxPoolTransactions()) {
527  LogPrint("obfuscation", "CObfuscationPool::Check() -- TRYING TRANSACTION \n");
528  UpdateState(POOL_STATUS_FINALIZE_TRANSACTION);
529  }
530  }
531 
532  // create the finalized transaction for distribution to the clients
533  if (state == POOL_STATUS_FINALIZE_TRANSACTION) {
534  LogPrint("obfuscation", "CObfuscationPool::Check() -- FINALIZE TRANSACTIONS\n");
535  UpdateState(POOL_STATUS_SIGNING);
536 
537  if (fMasterNode) {
538  CMutableTransaction txNew;
539 
540  // make our new transaction
541  for (unsigned int i = 0; i < entries.size(); i++) {
542  BOOST_FOREACH (const CTxOut& v, entries[i].vout)
543  txNew.vout.push_back(v);
544 
545  BOOST_FOREACH (const CTxDSIn& s, entries[i].sev)
546  txNew.vin.push_back(s);
547  }
548 
549  // shuffle the outputs for improved anonymity
550  std::random_shuffle(txNew.vin.begin(), txNew.vin.end(), randomizeList);
551  std::random_shuffle(txNew.vout.begin(), txNew.vout.end(), randomizeList);
552 
553 
554  LogPrint("obfuscation", "Transaction 1: %s\n", txNew.ToString());
555  finalTransaction = txNew;
556 
557  // request signatures from clients
558  RelayFinalTransaction(sessionID, finalTransaction);
559  }
560  }
561 
562  // If we have all of the signatures, try to compile the transaction
563  if (fMasterNode && state == POOL_STATUS_SIGNING && SignaturesComplete()) {
564  LogPrint("obfuscation", "CObfuscationPool::Check() -- SIGNING\n");
565  UpdateState(POOL_STATUS_TRANSMISSION);
566 
567  CheckFinalTransaction();
568  }
569 
570  // reset if we're here for 10 seconds
571  if ((state == POOL_STATUS_ERROR || state == POOL_STATUS_SUCCESS) && GetTimeMillis() - lastTimeChanged >= 10000) {
572  LogPrint("obfuscation", "CObfuscationPool::Check() -- timeout, RESETTING\n");
573  UnlockCoins();
574  SetNull();
575  if (fMasterNode) RelayStatus(sessionID, GetState(), GetEntriesCount(), MASTERNODE_RESET);
576  }
577 }

579 void CObfuscationPool::CheckFinalTransaction()
580 {
581  if (!fMasterNode) return; // check and relay final tx only on masternode
582 
583  CWalletTx txNew = CWalletTx(pwalletMain, finalTransaction);
584 
585  LOCK2(cs_main, pwalletMain->cs_wallet);
586  {
587  LogPrint("obfuscation", "Transaction 2: %s\n", txNew.ToString());
588 
589  // See if the transaction is valid
590  if (!txNew.AcceptToMemoryPool(false, true, true)) {
591  LogPrintf("CObfuscationPool::Check() - CommitTransaction : Error: Transaction not valid\n");
592  SetNull();
593 
594  // not much we can do in this case
595  UpdateState(POOL_STATUS_ACCEPTING_ENTRIES);
596  RelayCompletedTransaction(sessionID, true, ERR_INVALID_TX);
597  return;
598  }
599 
600  LogPrintf("CObfuscationPool::Check() -- IS MASTER -- TRANSMITTING OBFUSCATION\n");
601 
602  // sign a message
603 
604  int64_t sigTime = GetAdjustedTime();
605  std::string strMessage = txNew.GetHash().ToString() + boost::lexical_cast<std::string>(sigTime);
606  std::string strError = "";
607  std::vector<unsigned char> vchSig;
608  CKey key2;
609  CPubKey pubkey2;
610 
611  if (!obfuScationSigner.SetKey(strMasterNodePrivKey, strError, key2, pubkey2)) {
612  LogPrintf("CObfuscationPool::Check() - ERROR: Invalid Masternodeprivkey: '%s'\n", strError);
613  return;
614  }
615

652 //
653 // Charge clients a fee if they're abusive
654 //
655 // Why bother? Obfuscation uses collateral to ensure abuse to the process is kept to a minimum.
656 // The submission and signing stages in Obfuscation are completely separate. In the cases where
657 // a client submits a transaction then refused to sign, there must be a cost. Otherwise they
658 // would be able to do this over and over again and bring the mixing to a hault.
659 //
660 // How does this work? Messages to Masternodes come in via "dsi", these require a valid collateral
661 // transaction for the client to be able to enter the pool. This transaction is kept by the Masternode
662 // until the transaction is either complete or fails.
663 //
664 void CObfuscationPool::ChargeFees()
665 {
666  if (!fMasterNode) return;
667 
668  //we don't need to charge collateral for every offence.
669  int offences = 0;
670  int r = rand() % 100;
671  if (r > 33) return;
672 
673  if (state == POOL_STATUS_ACCEPTING_ENTRIES) {
674  BOOST_FOREACH (const CTransaction& txCollateral, vecSessionCollateral) {
675  bool found = false;
676  BOOST_FOREACH (const CObfuScationEntry& v, entries) {
677  if (v.collateral == txCollateral) {
678  found = true;
679  }
680  }
681 
682  // This queue entry didn't send us the promised transaction
683  if (!found) {
684  LogPrintf("CObfuscationPool::ChargeFees -- found uncooperative node (didn't send transaction). Found offence.\n");
685  offences++;
686  }
687  }
688  }
689 
690  if (state == POOL_STATUS_SIGNING) {
691  // who didn't sign?
692  BOOST_FOREACH (const CObfuScationEntry v, entries) {
693  BOOST_FOREACH (const CTxDSIn s, v.sev) {
694  if (!s.fHasSig) {
695  LogPrintf("CObfuscationPool::ChargeFees -- found uncooperative node (didn't sign). Found offence\n");
696  offences++;
697  }
698  }
699  }
700  }
701 
702  r = rand() % 100;
703  int target = 0;
704 
705  //mostly offending?
706  if (offences >= Params().PoolMaxTransactions() - 1 && r > 33) return;
707 
708  //everyone is an offender? That's not right
709  if (offences >= Params().PoolMaxTransactions()) return;
710 
711  //charge one of the offenders randomly
712  if (offences > 1) target = 50;
713 
714  //pick random client to charge
715  r = rand() % 100;
716 
717  if (state == POOL_STATUS_ACCEPTING_ENTRIES) {
718  BOOST_FOREACH (const CTransaction& txCollateral, vecSessionCollateral) {
719  bool found = false;
720  BOOST_FOREACH (const CObfuScationEntry& v, entries) {
721  if (v.collateral == txCollateral) {
722  found = true;
723  }
724  }
725

735  LogPrintf("CObfuscationPool::ChargeFees() : Error: Transaction not valid");
736  }
737  wtxCollateral.RelayWalletTransaction();
738  return;
739  }
740  }
741  }
742 
743  if (state == POOL_STATUS_SIGNING) {
744  // who didn't sign?
745  BOOST_FOREACH (const CObfuScationEntry v, entries) {
746  BOOST_FOREACH (const CTxDSIn s, v.sev) {
747  if (!s.fHasSig && r > target) {
748  LogPrintf("CObfuscationPool::ChargeFees -- found uncooperative node (didn't sign). charging fees.\n");

750  CWalletTx wtxCollateral = CWalletTx(pwalletMain, v.collateral);
751 
752  // Broadcast

758  return;
759  }
760  }
761  }

831  it = vecObfuscationQueue.erase(it);
832  } else
833  ++it;
834  c++;
