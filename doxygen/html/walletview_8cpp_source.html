178 void WalletView::processNewTransaction(const QModelIndex& parent, int start, int /*end*/)
179 {
180  // Prevent balloon-spam when initial block download is in progress
181  if (!walletModel || !clientModel || clientModel->inInitialBlockDownload())
182  return;
183 
184  TransactionTableModel* ttm = walletModel->getTransactionTableModel();
185  if (!ttm || ttm->processingQueuedTransactions())
186  return;
187 
188  QString date = ttm->index(start, TransactionTableModel::Date, parent).data().toString();
189  qint64 amount = ttm->index(start, TransactionTableModel::Amount, parent).data(Qt::EditRole).toULongLong();
190  QString type = ttm->index(start, TransactionTableModel::Type, parent).data().toString();
191  QString address = ttm->index(start, TransactionTableModel::ToAddress, parent).data().toString();
192 
193  emit incomingTransaction(date, walletModel->getOptionsModel()->getDisplayUnit(), amount, type, address);
194 }

358 void WalletView::toggleLockWallet()

363  WalletModel::EncryptionStatus encStatus = walletModel->getEncryptionStatus();
364 
365  // Unlock the wallet when requested
366  if (encStatus == walletModel->Locked) {
367  AskPassphraseDialog dlg(AskPassphraseDialog::UnlockAnonymize, this, walletModel);

372  walletModel->setWalletLocked(true);
