28 CScript ParseScript(std::string s)
29 {
30  CScript result;
31 
32  static map<string, opcodetype> mapOpNames;
33 
34  if (mapOpNames.empty()) {
35  for (int op = 0; op <= OP_ZEROCOINSPEND; op++) {
36  // Allow OP_RESERVED to get into mapOpNames
37  if (op < OP_NOP && op != OP_RESERVED)
38  continue;
39 
40  const char* name = GetOpName((opcodetype)op);
41  if (strcmp(name, "OP_UNKNOWN") == 0)
42  continue;
43  string strName(name);
44  mapOpNames[strName] = (opcodetype)op;
45  // Convenience: OP_ADD and just ADD are both recognized:
46  replace_first(strName, "OP_", "");
47  mapOpNames[strName] = (opcodetype)op;
48  }
49  }
50 
51  vector<string> words;
52  split(words, s, is_any_of(" \t\n"), token_compress_on);
53 
54  for (std::vector<std::string>::const_iterator w = words.begin(); w != words.end(); ++w) {
55  if (w->empty()) {
56  // Empty string, ignore. (boost::split given '' will return one word)
57  } else if (all(*w, is_digit()) ||
58  (starts_with(*w, "-") && all(string(w->begin() + 1, w->end()), is_digit()))) {
59  // Number
60  int64_t n = atoi64(*w);
61  result << n;
62  } else if (starts_with(*w, "0x") && (w->begin() + 2 != w->end()) && IsHex(string(w->begin() + 2, w->end()))) {
63  // Raw hex data, inserted NOT pushed onto stack:
64  std::vector<unsigned char> raw = ParseHex(string(w->begin() + 2, w->end()));
65  result.insert(result.end(), raw.begin(), raw.end());
66  } else if (w->size() >= 2 && starts_with(*w, "'") && ends_with(*w, "'")) {
67  // Single-quoted string, pushed as data. NOTE: this is poor-man's
68  // parsing, spaces/tabs/newlines in single-quoted strings won't work.
69  std::vector<unsigned char> value(w->begin() + 1, w->end() - 1);
70  result << value;
71  } else if (mapOpNames.count(*w)) {
72  // opcode, e.g. OP_ADD or ADD:
73  result << mapOpNames[*w];
74  } else {
75  throw runtime_error("script parse error");
76  }
77  }
78 
79  return result;
80 }
