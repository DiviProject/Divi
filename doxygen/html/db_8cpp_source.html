162 bool CDBEnv::Salvage(std::string strFile, bool fAggressive, std::vector<CDBEnv::KeyValPair>& vResult)
163 {
164  LOCK(cs_db);
165  assert(mapFileUseCount.count(strFile) == 0);
166 
167  u_int32_t flags = DB_SALVAGE;
168  if (fAggressive)
169  flags |= DB_AGGRESSIVE;
170 
171  stringstream strDump;
172 
173  Db db(&dbenv, 0);
174  int result = db.verify(strFile.c_str(), NULL, &strDump, flags);
175  if (result == DB_VERIFY_BAD) {
176  LogPrintf("CDBEnv::Salvage : Database salvage found errors, all data may not be recoverable.\n");
177  if (!fAggressive) {
178  LogPrintf("CDBEnv::Salvage : Rerun with aggressive mode to ignore errors and continue.\n");
179  return false;
180  }
181  }
182  if (result != 0 && result != DB_VERIFY_BAD) {
183  LogPrintf("CDBEnv::Salvage : Database salvage failed with result %d.\n", result);
184  return false;
185  }
186 
187  // Format of bdb dump is ascii lines:
188  // header lines...
189  // HEADER=END
190  // hexadecimal key
191  // hexadecimal value
192  // ... repeated
193  // DATA=END
194 
195  string strLine;
196  while (!strDump.eof() && strLine != "HEADER=END")
197  getline(strDump, strLine); // Skip past header
198 
199  std::string keyHex, valueHex;
200  while (!strDump.eof() && keyHex != "DATA=END") {
201  getline(strDump, keyHex);
202  if (keyHex != "DATA=END") {
203  getline(strDump, valueHex);
204  vResult.push_back(make_pair(ParseHex(keyHex), ParseHex(valueHex)));
205  }
206  }
207 
208  return (result == 0);
209 }

332 bool CDB::Rewrite(const string& strFile, const char* pszSkip)
333 {
334  while (true) {
335  {
336  LOCK(bitdb.cs_db);
337  if (!bitdb.mapFileUseCount.count(strFile) || bitdb.mapFileUseCount[strFile] == 0) {
338  // Flush log data to the dat file
339  bitdb.CloseDb(strFile);
340  bitdb.CheckpointLSN(strFile);
341  bitdb.mapFileUseCount.erase(strFile);
342 
343  bool fSuccess = true;
344  LogPrintf("CDB::Rewrite : Rewriting %s...\n", strFile);
345  string strFileRes = strFile + ".rewrite";
346  { // surround usage of db with extra {}
347  CDB db(strFile.c_str(), "r");
348  Db* pdbCopy = new Db(&bitdb.dbenv, 0);
349 
350  int ret = pdbCopy->open(NULL, // Txn pointer
351  strFileRes.c_str(), // Filename
352  "main", // Logical db name
353  DB_BTREE, // Database type
354  DB_CREATE, // Flags
355  0);
356  if (ret > 0) {
357  LogPrintf("CDB::Rewrite : Can't create database file %s\n", strFileRes);
358  fSuccess = false;
359  }
360 
361  Dbc* pcursor = db.GetCursor();
362  if (pcursor)
363  while (fSuccess) {
364  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
365  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
366  int ret = db.ReadAtCursor(pcursor, ssKey, ssValue, DB_NEXT);
367  if (ret == DB_NOTFOUND) {
368  pcursor->close();
369  break;
370  } else if (ret != 0) {
371  pcursor->close();
372  fSuccess = false;
373  break;
374  }
375  if (pszSkip &&
376  strncmp(&ssKey[0], pszSkip, std::min(ssKey.size(), strlen(pszSkip))) == 0)
377  continue;
378  if (strncmp(&ssKey[0], "\x07version", 8) == 0) {
379  // Update version:
380  ssValue.clear();
381  ssValue << CLIENT_VERSION;
382  }
383  Dbt datKey(&ssKey[0], ssKey.size());
384  Dbt datValue(&ssValue[0], ssValue.size());
385  int ret2 = pdbCopy->put(NULL, &datKey, &datValue, DB_NOOVERWRITE);
386  if (ret2 > 0)
387  fSuccess = false;
388  }
389  if (fSuccess) {
390  db.Close();
391  bitdb.CloseDb(strFile);
392  if (pdbCopy->close(0))
393  fSuccess = false;
394  delete pdbCopy;
395  }
396  }
397  if (fSuccess) {
398  Db dbA(&bitdb.dbenv, 0);
399  if (dbA.remove(strFile.c_str(), NULL, 0))
400  fSuccess = false;
401  Db dbB(&bitdb.dbenv, 0);
402  if (dbB.rename(strFileRes.c_str(), NULL, strFile.c_str(), 0))
403  fSuccess = false;
404  }
405  if (!fSuccess)
406  LogPrintf("CDB::Rewrite : Failed to rewrite database file %s\n", strFileRes);
407  return fSuccess;
408  }
409  }
410  MilliSleep(100);
411  }
412  return false;
413 }

416 void CDBEnv::Flush(bool fShutdown)
417 {
418  int64_t nStart = GetTimeMillis();
419  // Flush log data to the actual data file on all files that are not in use
420  LogPrint("db", "CDBEnv::Flush : Flush(%s)%s\n", fShutdown ? "true" : "false", fDbEnvInit ? "" : " database not started");
421  if (!fDbEnvInit)
422  return;
423  {
424  LOCK(cs_db);
425  map<string, int>::iterator mi = mapFileUseCount.begin();
426  while (mi != mapFileUseCount.end()) {
427  string strFile = (*mi).first;
428  int nRefCount = (*mi).second;
429  LogPrint("db", "CDBEnv::Flush : Flushing %s (refcount = %d)...\n", strFile, nRefCount);
430  if (nRefCount == 0) {
431  // Move log data to the dat file
432  CloseDb(strFile);
433  LogPrint("db", "CDBEnv::Flush : %s checkpoint\n", strFile);
434  dbenv.txn_checkpoint(0, 0, 0);
435  LogPrint("db", "CDBEnv::Flush : %s detach\n", strFile);
436  if (!fMockDb)
437  dbenv.lsn_reset(strFile.c_str(), 0);
438  LogPrint("db", "CDBEnv::Flush : %s closed\n", strFile);
439  mapFileUseCount.erase(mi++);
440  } else
441  mi++;
442  }
443  LogPrint("db", "CDBEnv::Flush : Flush(%s)%s took %15dms\n", fShutdown ? "true" : "false", fDbEnvInit ? "" : " database not started", GetTimeMillis() - nStart);
444  if (fShutdown) {
445  char** listp;
446  if (mapFileUseCount.empty()) {
447  dbenv.log_archive(&listp, DB_ARCH_REMOVE);
448  Close();
449  if (!fMockDb)
450  boost::filesystem::remove_all(boost::filesystem::path(strPath) / "database");
451  }
452  }
453  }
