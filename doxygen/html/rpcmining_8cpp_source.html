39 Value GetNetworkHashPS(int lookup, int height)
40 {
41  CBlockIndex* pb = chainActive.Tip();
42 
43  if (height >= 0 && height < chainActive.Height())
44  pb = chainActive[height];
45 
46  if (pb == NULL || !pb->nHeight)
47  return 0;
48 
49  // If lookup is -1, then use blocks since last difficulty change.
50  if (lookup <= 0)
51  lookup = pb->nHeight % 2016 + 1;
52 
53  // If lookup is larger than chain, then set it to chain length.
54  if (lookup > pb->nHeight)
55  lookup = pb->nHeight;
56 
57  CBlockIndex* pb0 = pb;
58  int64_t minTime = pb0->GetBlockTime();
59  int64_t maxTime = minTime;
60  for (int i = 0; i < lookup; i++) {
61  pb0 = pb0->pprev;
62  int64_t time = pb0->GetBlockTime();
63  minTime = std::min(time, minTime);
64  maxTime = std::max(time, maxTime);
65  }
66 
67  // In case there's a situation where minTime == maxTime, we don't want a divide by zero exception.
68  if (minTime == maxTime)
69  return 0;
70 
71  uint256 workDiff = pb->nChainWork - pb0->nChainWork;
72  int64_t timeDiff = maxTime - minTime;
73 
74  return (int64_t)(workDiff.getdouble() / timeDiff);
75 }

114 Value setgenerate(const Array& params, bool fHelp)
115 {
116  if (fHelp || params.size() < 1 || params.size() > 2)
117  throw runtime_error(
118  "setgenerate generate ( genproclimit )\n"
119  "\nSet 'generate' true or false to turn generation on or off.\n"
120  "Generation is limited to 'genproclimit' processors, -1 is unlimited.\n"
121  "See the getgenerate call for the current setting.\n"
122  "\nArguments:\n"
123  "1. generate (boolean, required) Set to true to turn on generation, false to turn off.\n"
124  "2. genproclimit (numeric, optional) Set the processor limit for when generation is on. Can be -1 for unlimited.\n"
125  " Note: in -regtest mode, genproclimit controls how many blocks are generated immediately.\n"
126  "\nResult\n"
127  "[ blockhashes ] (array, -regtest only) hashes of blocks generated\n"
128  "\nExamples:\n"
129  "\nSet the generation on with a limit of one processor\n" +
130  HelpExampleCli("setgenerate", "true 1") +
131  "\nCheck the setting\n" + HelpExampleCli("getgenerate", "") +
132  "\nTurn off generation\n" + HelpExampleCli("setgenerate", "false") +
133  "\nUsing json rpc\n" + HelpExampleRpc("setgenerate", "true, 1"));
134 
135  if (pwalletMain == NULL)
136  throw JSONRPCError(RPC_METHOD_NOT_FOUND, "Method not found (disabled)");
137 
138  bool fGenerate = true;
139  if (params.size() > 0)
140  fGenerate = params[0].get_bool();
141 
142  int nGenProcLimit = -1;
143  if (params.size() > 1) {
144  nGenProcLimit = params[1].get_int();
145  if (nGenProcLimit == 0)
146  fGenerate = false;
147  }
148 
149  // -regtest mode: don't return until nGenProcLimit blocks are generated
150  if (fGenerate && Params().MineBlocksOnDemand()) {
151  int nHeightStart = 0;
152  int nHeightEnd = 0;
153  int nHeight = 0;
154  int nGenerate = (nGenProcLimit > 0 ? nGenProcLimit : 1);
155  CReserveKey reservekey(pwalletMain);
156 
157  { // Don't keep cs_main locked
158  LOCK(cs_main);
159  nHeightStart = chainActive.Height();
160  nHeight = nHeightStart;
161  nHeightEnd = nHeightStart + nGenerate;
162  }
163  unsigned int nExtraNonce = 0;
164  Array blockHashes;
165  while (nHeight < nHeightEnd) {
166  unique_ptr<CBlockTemplate> pblocktemplate(CreateNewBlockWithKey(reservekey, pwalletMain, false));
167  if (!pblocktemplate.get())
168  throw JSONRPCError(RPC_INTERNAL_ERROR, "Wallet keypool empty");
169  CBlock* pblock = &pblocktemplate->block;
170  {
171  LOCK(cs_main);
172  IncrementExtraNonce(pblock, chainActive.Tip(), nExtraNonce);
173  }
174  while (!CheckProofOfWork(pblock->GetHash(), pblock->nBits)) {
175  // Yes, there is a chance every nonce could fail to satisfy the -regtest
176  // target -- 1 in 2^(2^32). That ain't gonna happen.
177  ++pblock->nNonce;
178  }
179  CValidationState state;
180  if (!ProcessNewBlock(state, NULL, pblock))
181  throw JSONRPCError(RPC_INTERNAL_ERROR, "ProcessNewBlock, block not accepted");
182  ++nHeight;
183  blockHashes.push_back(pblock->GetHash().GetHex());
184  }
185  return blockHashes;
186  } else // Not -regtest: start generate thread, return immediately
187  {
188  mapArgs["-gen"] = (fGenerate ? "1" : "0");
189  mapArgs["-genproclimit"] = itostr(nGenProcLimit);
190  GenerateBitcoins(fGenerate, pwalletMain, nGenProcLimit);
191  }
192 
193  return Value::null;
194 }

286 // NOTE: Assumes a conclusive result; if result is inconclusive, it must be handled by caller
287 static Value BIP22ValidationResult(const CValidationState& state)
288 {
289  if (state.IsValid())
290  return Value::null;
291 
292  std::string strRejectReason = state.GetRejectReason();
293  if (state.IsError())
294  throw JSONRPCError(RPC_VERIFY_ERROR, strRejectReason);
295  if (state.IsInvalid()) {
296  if (strRejectReason.empty())
297  return "rejected";
298  return strRejectReason;
299  }
300  // Should be impossible
301  return "valid?";
302 }

609 Value submitblock(const Array& params, bool fHelp)
610 {
611  if (fHelp || params.size() < 1 || params.size() > 2)
612  throw runtime_error(
613  "submitblock \"hexdata\" ( \"jsonparametersobject\" )\n"
614  "\nAttempts to submit new block to network.\n"
615  "The 'jsonparametersobject' parameter is currently ignored.\n"
616  "See https://en.bitcoin.it/wiki/BIP_0022 for full specification.\n"
617 
618  "\nArguments\n"
619  "1. \"hexdata\" (string, required) the hex-encoded block data to submit\n"
620  "2. \"jsonparametersobject\" (string, optional) object of optional parameters\n"
621  " {\n"
622  " \"workid\" : \"id\" (string, optional) if the server provided a workid, it MUST be included with submissions\n"
623  " }\n"
624  "\nResult:\n"
625  "\nExamples:\n" +
626  HelpExampleCli("submitblock", "\"mydata\"") + HelpExampleRpc("submitblock", "\"mydata\""));
627 
628  CBlock block;
629  if (!DecodeHexBlk(block, params[0].get_str()))
630  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "Block decode failed");
631 
632  uint256 hash = block.GetHash();
633  bool fBlockPresent = false;
634  {
635  LOCK(cs_main);
636  BlockMap::iterator mi = mapBlockIndex.find(hash);
637  if (mi != mapBlockIndex.end()) {
638  CBlockIndex* pindex = mi->second;
639  if (pindex->IsValid(BLOCK_VALID_SCRIPTS))
640  return "duplicate";
641  if (pindex->nStatus & BLOCK_FAILED_MASK)
642  return "duplicate-invalid";
643  // Otherwise, we might only have the header - process the block before returning
644  fBlockPresent = true;
645  }
646  }
647 
648  CValidationState state;
649  submitblock_StateCatcher sc(block.GetHash());
650  RegisterValidationInterface(&sc);
651  bool fAccepted = ProcessNewBlock(state, NULL, &block);
652  UnregisterValidationInterface(&sc);
653  if (fBlockPresent) {
654  if (fAccepted && !sc.found)
655  return "duplicate-inconclusive";
656  return "duplicate";
657  }
658  if (fAccepted) {
659  if (!sc.found)
660  return "inconclusive";
661  state = sc.state;
662  }
663  return BIP22ValidationResult(state);
664 }
