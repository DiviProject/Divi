99 static QString GetLangTerritory()
100 {
101  QSettings settings;
102  // Get desired locale (e.g. "de_DE")
103  // 1) System default language
104  QString lang_territory = QLocale::system().name();
105  // 2) Language from QSettings

107  if (!lang_territory_qsettings.isEmpty())
108  lang_territory = lang_territory_qsettings;
109  // 3) -lang command line argument
110  lang_territory = QString::fromStdString(GetArg("-lang", lang_territory.toStdString()));
111  return lang_territory;

115 static void initTranslations(QTranslator& qtTranslatorBase, QTranslator& qtTranslator, QTranslator& translatorBase, QTranslator& translator)
116 {
117  // Remove old translators
118  QApplication::removeTranslator(&qtTranslatorBase);
119  QApplication::removeTranslator(&qtTranslator);
120  QApplication::removeTranslator(&translatorBase);
121  QApplication::removeTranslator(&translator);
122 
123  // Get desired locale (e.g. "de_DE")
124  // 1) System default language
125  QString lang_territory = GetLangTerritory();
126 
127  // Convert to "de" only by truncating "_DE"
128  QString lang = lang_territory;
129  lang.truncate(lang_territory.lastIndexOf('_'));
130 
131  // Load language files for configured locale:
132  // - First load the translator for the base language, without territory
133  // - Then load the more specific locale translator
134 
135  // Load e.g. qt_de.qm
136  if (qtTranslatorBase.load("qt_" + lang, QLibraryInfo::location(QLibraryInfo::TranslationsPath)))
137  QApplication::installTranslator(&qtTranslatorBase);
138 
139  // Load e.g. qt_de_DE.qm
140  if (qtTranslator.load("qt_" + lang_territory, QLibraryInfo::location(QLibraryInfo::TranslationsPath)))
141  QApplication::installTranslator(&qtTranslator);
142 
143  // Load e.g. bitcoin_de.qm (shortcut "de" needs to be defined in divx.qrc)
144  if (translatorBase.load(lang, ":/translations/"))
145  QApplication::installTranslator(&translatorBase);
146 
147  // Load e.g. bitcoin_de_DE.qm (shortcut "de_DE" needs to be defined in divx.qrc)
148  if (translator.load(lang_territory, ":/translations/"))
149  QApplication::installTranslator(&translator);
150 }

288 void BitcoinCore::restart(QStringList args)
289 {
290  if (execute_restart) { // Only restart 1x, no matter how often a user clicks on a restart-button
291  execute_restart = false;
292  try {
293  qDebug() << __func__ << ": Running Restart in thread";
294  threadGroup.interrupt_all();
295  threadGroup.join_all();
296  PrepareShutdown();
297  qDebug() << __func__ << ": Shutdown finished";
298  emit shutdownResult(1);
299  CExplicitNetCleanup::callCleanup();
300  QProcess::startDetached(QApplication::applicationFilePath(), args);
301  qDebug() << __func__ << ": Restart initiated...";
302  QApplication::quit();
303  } catch (std::exception& e) {
304  handleRunawayException(&e);
305  } catch (...) {
306  handleRunawayException(NULL);
307  }
308  }

398 void BitcoinApplication::startThread()
399 {
400  if (coreThread)
401  return;
402  coreThread = new QThread(this);
403  BitcoinCore* executor = new BitcoinCore();
404  executor->moveToThread(coreThread);
405 
406  /* communication to and from thread */
407  connect(executor, SIGNAL(initializeResult(int)), this, SLOT(initializeResult(int)));
408  connect(executor, SIGNAL(shutdownResult(int)), this, SLOT(shutdownResult(int)));
409  connect(executor, SIGNAL(runawayException(QString)), this, SLOT(handleRunawayException(QString)));
410  connect(this, SIGNAL(requestedInitialize()), executor, SLOT(initialize()));
411  connect(this, SIGNAL(requestedShutdown()), executor, SLOT(shutdown()));
412  connect(window, SIGNAL(requestedRestart(QStringList)), executor, SLOT(restart(QStringList)));
413  /* make sure executor object is deleted in its own thread */
414  connect(this, SIGNAL(stopThread()), executor, SLOT(deleteLater()));
415  connect(this, SIGNAL(stopThread()), coreThread, SLOT(quit()));
416 
417  coreThread->start();
418 }

450 void BitcoinApplication::initializeResult(int retval)
451 {
452  qDebug() << __func__ << ": Initialization result: " << retval;
453  // Set exit result: 0 if successful, 1 if failure
454  returnValue = retval ? 0 : 1;
455  if (retval) {
456 #ifdef ENABLE_WALLET
457  PaymentServer::LoadRootCAs();
458  paymentServer->setOptionsModel(optionsModel);
459 #endif
460 
461  clientModel = new ClientModel(optionsModel);
462  window->setClientModel(clientModel);
463 
464 #ifdef ENABLE_WALLET
465  if (pwalletMain) {
466  walletModel = new WalletModel(pwalletMain, optionsModel);
467 
468  window->addWallet(BitcoinGUI::DEFAULT_WALLET, walletModel);
469  window->setCurrentWallet(BitcoinGUI::DEFAULT_WALLET);
470 
471  connect(walletModel, SIGNAL(coinsSent(CWallet*, SendCoinsRecipient, QByteArray)),
472  paymentServer, SLOT(fetchPaymentACK(CWallet*, const SendCoinsRecipient&, QByteArray)));
473  }
474 #endif
475 
476  // If -min option passed, start window minimized.
477  if (GetBoolArg("-min", false)) {
478  window->showMinimized();
479  } else {
480  window->show();
481  }
482  emit splashFinished(window);
483 
484 #ifdef ENABLE_WALLET
485  // Now that initialization/startup is done, process any command-line
486  // DIVX: URIs or payment requests:
487  connect(paymentServer, SIGNAL(receivedPaymentRequest(SendCoinsRecipient)),
488  window, SLOT(handlePaymentRequest(SendCoinsRecipient)));
489  connect(window, SIGNAL(receivedURI(QString)),
490  paymentServer, SLOT(handleURIOrFile(QString)));
491  connect(paymentServer, SIGNAL(message(QString, QString, unsigned int)),
492  window, SLOT(message(QString, QString, unsigned int)));
493  QTimer::singleShot(100, paymentServer, SLOT(uiReady()));
494 #endif
495  } else {
496  quit(); // Exit main loop
497  }
498 }
