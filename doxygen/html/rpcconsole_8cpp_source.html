96 bool parseCommandLine(std::vector<std::string>& args, const std::string& strCommand)
97 {
98  enum CmdParseState {
99  STATE_EATING_SPACES,
100  STATE_ARGUMENT,
101  STATE_SINGLEQUOTED,
102  STATE_DOUBLEQUOTED,
103  STATE_ESCAPE_OUTER,
104  STATE_ESCAPE_DOUBLEQUOTED
105  } state = STATE_EATING_SPACES;
106  std::string curarg;
107  foreach (char ch, strCommand) {
108  switch (state) {
109  case STATE_ARGUMENT: // In or after argument
110  case STATE_EATING_SPACES: // Handle runs of whitespace
111  switch (ch) {
112  case '"':
113  state = STATE_DOUBLEQUOTED;
114  break;
115  case '\'':
116  state = STATE_SINGLEQUOTED;
117  break;
118  case '\\':
119  state = STATE_ESCAPE_OUTER;
120  break;
121  case ' ':
122  case '\n':
123  case '\t':
124  if (state == STATE_ARGUMENT) // Space ends argument
125  {
126  args.push_back(curarg);
127  curarg.clear();
128  }
129  state = STATE_EATING_SPACES;
130  break;
131  default:
132  curarg += ch;
133  state = STATE_ARGUMENT;
134  }
135  break;
136  case STATE_SINGLEQUOTED: // Single-quoted string
137  switch (ch) {
138  case '\'':
139  state = STATE_ARGUMENT;
140  break;
141  default:
142  curarg += ch;
143  }
144  break;
145  case STATE_DOUBLEQUOTED: // Double-quoted string
146  switch (ch) {
147  case '"':
148  state = STATE_ARGUMENT;
149  break;
150  case '\\':
151  state = STATE_ESCAPE_DOUBLEQUOTED;
152  break;
153  default:
154  curarg += ch;
155  }
156  break;
157  case STATE_ESCAPE_OUTER: // '\' outside quotes
158  curarg += ch;
159  state = STATE_ARGUMENT;
160  break;
161  case STATE_ESCAPE_DOUBLEQUOTED: // '\' in double-quoted text
162  if (ch != '"' && ch != '\\') curarg += '\\'; // keep '\' for everything but the quote and '\' itself
163  curarg += ch;
164  state = STATE_DOUBLEQUOTED;
165  break;
166  }
167  }
168  switch (state) // final state
169  {
170  case STATE_EATING_SPACES:
171  return true;
172  case STATE_ARGUMENT:
173  args.push_back(curarg);
174  return true;
175  default: // ERROR to end in one of the other states
176  return false;
177  }
178 }

180 void RPCExecutor::request(const QString& command)
181 {
182  std::vector<std::string> args;
183  if (!parseCommandLine(args, command.toStdString())) {
184  emit reply(RPCConsole::CMD_ERROR, QString("Parse error: unbalanced ' or \""));
185  return;
186  }
187  if (args.empty())
188  return; // Nothing to do
189  try {
190  std::string strPrint;
191  // Convert argument list to JSON objects in method-dependent way,
192  // and pass it along with the method name to the dispatcher.
193  json_spirit::Value result = tableRPC.execute(
194  args[0],
195  RPCConvertValues(args[0], std::vector<std::string>(args.begin() + 1, args.end())));
196 
197  // Format result reply
198  if (result.type() == json_spirit::null_type)
199  strPrint = "";
200  else if (result.type() == json_spirit::str_type)
201  strPrint = result.get_str();
202  else
203  strPrint = write_string(result, true);
204 
205  emit reply(RPCConsole::CMD_REPLY, QString::fromStdString(strPrint));
206  } catch (json_spirit::Object& objError) {
207  try // Nice formatting for standard-format error
208  {
209  int code = find_value(objError, "code").get_int();
210  std::string message = find_value(objError, "message").get_str();
211  emit reply(RPCConsole::CMD_ERROR, QString::fromStdString(message) + " (code " + QString::number(code) + ")");
212  } catch (std::runtime_error&) // raised when converting to invalid type, i.e. missing code or message
213  { // Show raw JSON object
214  emit reply(RPCConsole::CMD_ERROR, QString::fromStdString(write_string(json_spirit::Value(objError), false)));
215  }
216  } catch (std::exception& e) {
217  emit reply(RPCConsole::CMD_ERROR, QString("Error: ") + QString::fromStdString(e.what()));
218  }
219 }

275 bool RPCConsole::eventFilter(QObject* obj, QEvent* event)
276 {
277  if (event->type() == QEvent::KeyPress) // Special key handling
278  {
279  QKeyEvent* keyevt = static_cast<QKeyEvent*>(event);
280  int key = keyevt->key();
281  Qt::KeyboardModifiers mod = keyevt->modifiers();
282  switch (key) {
283  case Qt::Key_Up:
284  if (obj == ui->lineEdit) {
285  browseHistory(-1);
286  return true;
287  }
288  break;
289  case Qt::Key_Down:
290  if (obj == ui->lineEdit) {
291  browseHistory(1);
292  return true;
293  }
294  break;
295  case Qt::Key_PageUp: /* pass paging keys to messages widget */
296  case Qt::Key_PageDown:
297  if (obj == ui->lineEdit) {
298  QApplication::postEvent(ui->messagesWidget, new QKeyEvent(*keyevt));
299  return true;
300  }
301  break;
302  case Qt::Key_Return:
303  case Qt::Key_Enter:
304  // forward these events to lineEdit
305  if(obj == autoCompleter->popup()) {
306  QApplication::postEvent(ui->lineEdit, new QKeyEvent(*keyevt));
307  return true;
308  }
309  break;
310  default:
311  // Typing in messages widget brings focus to line edit, and redirects key there
312  // Exclude most combinations and keys that emit no text, except paste shortcuts
313  if (obj == ui->messagesWidget && ((!mod && !keyevt->text().isEmpty() && key != Qt::Key_Tab) ||
314  ((mod & Qt::ControlModifier) && key == Qt::Key_V) ||
315  ((mod & Qt::ShiftModifier) && key == Qt::Key_Insert))) {
316  ui->lineEdit->setFocus();
317  QApplication::postEvent(ui->lineEdit, new QKeyEvent(*keyevt));
318  return true;
319  }
320  }
321  }
322  return QDialog::eventFilter(obj, event);
323 }

556 void RPCConsole::on_lineEdit_returnPressed()
557 {
558  QString cmd = ui->lineEdit->text();
559  ui->lineEdit->clear();
560 
561  if (!cmd.isEmpty()) {
562  message(CMD_REQUEST, cmd);
563  emit cmdRequest(cmd);
564  // Remove command, if already in history
565  history.removeOne(cmd);
566  // Append command to history
567  history.append(cmd);
568  // Enforce maximum history size
569  while (history.size() > CONSOLE_HISTORY)
570  history.removeFirst();
571  // Set pointer to end of history
572  historyPtr = history.size();
573  // Scroll console view to end
574  scrollToEnd();
575  }
576 }

591 void RPCConsole::startExecutor()
592 {
593  QThread* thread = new QThread;
594  RPCExecutor* executor = new RPCExecutor();
595  executor->moveToThread(thread);
596 
597  // Replies from executor object must go to this object
598  connect(executor, SIGNAL(reply(int, QString)), this, SLOT(message(int, QString)));
599  // Requests from this object must go to executor
600  connect(this, SIGNAL(cmdRequest(QString)), executor, SLOT(request(QString)));
601 
602  // On stopExecutor signal
603  // - queue executor for deletion (in execution thread)
604  // - quit the Qt event loop in the execution thread
605  connect(this, SIGNAL(stopExecutor()), executor, SLOT(deleteLater()));
606  connect(this, SIGNAL(stopExecutor()), thread, SLOT(quit()));
607  // Queue the thread for deletion (in this thread) when it is finished
608  connect(thread, SIGNAL(finished()), thread, SLOT(deleteLater()));
609 
610  // Default implementation of QThread::run() simply spins up an event loop in the thread,
611  // which is what we want.
612  thread->start();
613 }

716 void RPCConsole::peerLayoutChanged()
717 {
718  if (!clientModel)
719  return;
720 
721  const CNodeCombinedStats* stats = NULL;
722  bool fUnselect = false;
723  bool fReselect = false;
724 
725  if (cachedNodeid == -1) // no node selected yet
726  return;
727 
728  // find the currently selected row
729  int selectedRow;
730  QModelIndexList selectedModelIndex = ui->peerWidget->selectionModel()->selectedIndexes();
731  if (selectedModelIndex.isEmpty())
732  selectedRow = -1;
733  else
734  selectedRow = selectedModelIndex.first().row();
735 
736  // check if our detail node has a row in the table (it may not necessarily
737  // be at selectedRow since its position can change after a layout change)
738  int detailNodeRow = clientModel->getPeerTableModel()->getRowByNodeId(cachedNodeid);
739 
740  if (detailNodeRow < 0) {
741  // detail node dissapeared from table (node disconnected)
742  fUnselect = true;
743  cachedNodeid = -1;
744  ui->peerHeading->setText(tr("Select a peer to view detailed information."));
745  } else {
746  if (detailNodeRow != selectedRow) {
747  // detail node moved position
748  fUnselect = true;
749  fReselect = true;
750  }
751 
752  // get fresh stats on the detail node.
753  stats = clientModel->getPeerTableModel()->getNodeStats(detailNodeRow);
754  }
755 
756  if (fUnselect && selectedRow >= 0) {
757  ui->peerWidget->selectionModel()->select(QItemSelection(selectedModelIndex.first(), selectedModelIndex.last()),
758  QItemSelectionModel::Deselect);
759  }
760 
761  if (fReselect) {
762  ui->peerWidget->selectRow(detailNodeRow);
763  }
764 
765  if (stats)
766  updateNodeDetail(stats);
767 }
