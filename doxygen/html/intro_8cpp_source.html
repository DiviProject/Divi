20 /* Minimum free space (in bytes) needed for data directory */
21 static const uint64_t GB_BYTES = 1000000000LL;
22 static const uint64_t BLOCK_CHAIN_SIZE = 1LL * GB_BYTES;
23 
24 /* Check free space asynchronously to prevent hanging the UI thread.
25 
26  Up to one request to check a path is in flight to this thread; when the check()
27  function runs, the current path is requested from the associated Intro object.
28  The reply is sent back through a signal.
29 
30  This ensures that no queue of checking requests is built up while the user is
31  still entering the path, and that always the most recently entered path is checked as
32  soon as the thread becomes available.
33 */

63 void FreespaceChecker::check()
64 {
65  namespace fs = boost::filesystem;
66  QString dataDirStr = intro->getPathToCheck();
67  fs::path dataDir = GUIUtil::qstringToBoostPath(dataDirStr);
68  uint64_t freeBytesAvailable = 0;
69  int replyStatus = ST_OK;
70  QString replyMessage = tr("A new data directory will be created.");
71 
72  /* Find first parent that exists, so that fs::space does not fail */
73  fs::path parentDir = dataDir;
74  fs::path parentDirOld = fs::path();
75  while (parentDir.has_parent_path() && !fs::exists(parentDir)) {
76  parentDir = parentDir.parent_path();
77 
78  /* Check if we make any progress, break if not to prevent an infinite loop here */
79  if (parentDirOld == parentDir)
80  break;
81 
82  parentDirOld = parentDir;
83  }
84 
85  try {
86  freeBytesAvailable = fs::space(parentDir).available;
87  if (fs::exists(dataDir)) {
88  if (fs::is_directory(dataDir)) {
89  QString separator = "<code>" + QDir::toNativeSeparators("/") + tr("name") + "</code>";
90  replyStatus = ST_OK;
91  replyMessage = tr("Directory already exists. Add %1 if you intend to create a new directory here.").arg(separator);
92  } else {
93  replyStatus = ST_ERROR;
94  replyMessage = tr("Path already exists, and is not a directory.");
95  }
96  }
97  } catch (fs::filesystem_error& e) {
98  /* Parent directory does not exist or is not accessible */
99  replyStatus = ST_ERROR;
100  replyMessage = tr("Cannot create data directory here.");
101  }
102  emit reply(replyStatus, replyMessage, freeBytesAvailable);
103 }

148 bool Intro::pickDataDirectory()
149 {
150  namespace fs = boost::filesystem;
151  QSettings settings;
152  /* If data directory provided on command line, no need to look at settings
153  or show a picking dialog */
154  if (!GetArg("-datadir", "").empty())
155  return true;
156  /* 1) Default data directory for operating system */
157  QString dataDir = getDefaultDataDirectory();
158  /* 2) Allow QSettings to override default dir */
159  dataDir = settings.value("strDataDir", dataDir).toString();
160 
161  if (!fs::exists(GUIUtil::qstringToBoostPath(dataDir)) || GetBoolArg("-choosedatadir", false)) {
162  /* If current default data directory does not exist, let the user choose one */
163  Intro intro;
164  intro.setDataDirectory(dataDir);
165  intro.setWindowIcon(QIcon(":icons/bitcoin"));
166 
167  while (true) {
168  if (!intro.exec()) {
169  /* Cancel clicked */
170  return false;
171  }
172  dataDir = intro.getDataDirectory();
173  try {
174  TryCreateDirectory(GUIUtil::qstringToBoostPath(dataDir));
175  break;
176  } catch (fs::filesystem_error& e) {
177  QMessageBox::critical(0, tr("DIVI Core"),
178  tr("Error: Specified data directory \"%1\" cannot be created.").arg(dataDir));
179  /* fall through, back to choosing screen */
180  }
181  }
182 
183  settings.setValue("strDataDir", dataDir);
184  }
185  /* Only override -datadir if different from the default, to make it possible to
186  * override -datadir in the divi.conf file in the default data directory
187  * (to be consistent with divid behavior)
188  */
189  if (dataDir != getDefaultDataDirectory())
190  SoftSetArg("-datadir", GUIUtil::qstringToBoostPath(dataDir).string()); // use OS locale for path setting
191  return true;
192 }

194 void Intro::setStatus(int status, const QString& message, quint64 bytesAvailable)
195 {
196  switch (status) {
197  case FreespaceChecker::ST_OK:
198  ui->errorMessage->setText(message);
199  ui->errorMessage->setStyleSheet("");
200  break;
201  case FreespaceChecker::ST_ERROR:
202  ui->errorMessage->setText(tr("Error") + ": " + message);
203  ui->errorMessage->setStyleSheet("QLabel { color: #800000 }");
204  break;
205  }
206  /* Indicate number of bytes available */
207  if (status == FreespaceChecker::ST_ERROR) {
208  ui->freeSpace->setText("");
209  } else {
210  QString freeString = tr("%1 GB of free space available").arg(bytesAvailable / GB_BYTES);
211  if (bytesAvailable < BLOCK_CHAIN_SIZE) {
212  freeString += " " + tr("(of %1 GB needed)").arg(BLOCK_CHAIN_SIZE / GB_BYTES);
213  ui->freeSpace->setStyleSheet("QLabel { color: #800000 }");
214  } else {
215  ui->freeSpace->setStyleSheet("");
216  }
217  ui->freeSpace->setText(freeString + ".");
218  }
219  /* Don't allow confirm in ERROR state */
220  ui->buttonBox->button(QDialogButtonBox::Ok)->setEnabled(status != FreespaceChecker::ST_ERROR);
221 }

248 void Intro::startThread()
249 {
250  thread = new QThread(this);
251  FreespaceChecker* executor = new FreespaceChecker(this);
252  executor->moveToThread(thread);
253 
254  connect(executor, SIGNAL(reply(int, QString, quint64)), this, SLOT(setStatus(int, QString, quint64)));
255  connect(this, SIGNAL(requestCheck()), executor, SLOT(check()));
256  /* make sure executor object is deleted in its own thread */
257  connect(this, SIGNAL(stopThread()), executor, SLOT(deleteLater()));
258  connect(this, SIGNAL(stopThread()), thread, SLOT(quit()));
259 
260  thread->start();
261 }
