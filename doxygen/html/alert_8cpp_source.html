170 bool CAlert::ProcessAlert(bool fThread)
171 {
172  if (!CheckSignature())
173  return false;
174  if (!IsInEffect())
175  return false;
176 
177  // alert.nID=max is reserved for if the alert key is
178  // compromised. It must have a pre-defined message,
179  // must never expire, must apply to all versions,
180  // and must cancel all previous
181  // alerts or it will be ignored (so an attacker can't
182  // send an "everything is OK, don't panic" version that
183  // cannot be overridden):
184  int maxInt = std::numeric_limits<int>::max();
185  if (nID == maxInt) {
186  if (!(
187  nExpiration == maxInt &&
188  nCancel == (maxInt - 1) &&
189  nMinVer == 0 &&
190  nMaxVer == maxInt &&
191  setSubVer.empty() &&
192  nPriority == maxInt &&
193  strStatusBar == "URGENT: Alert key compromised, upgrade required"))
194  return false;
195  }
196 
197  {
198  LOCK(cs_mapAlerts);
199  // Cancel previous alerts
200  for (map<uint256, CAlert>::iterator mi = mapAlerts.begin(); mi != mapAlerts.end();) {
201  const CAlert& alert = (*mi).second;
202  if (Cancels(alert)) {
203  LogPrint("alert", "cancelling alert %d\n", alert.nID);
204  uiInterface.NotifyAlertChanged((*mi).first, CT_DELETED);
205  mapAlerts.erase(mi++);
206  } else if (!alert.IsInEffect()) {
207  LogPrint("alert", "expiring alert %d\n", alert.nID);
208  uiInterface.NotifyAlertChanged((*mi).first, CT_DELETED);
209  mapAlerts.erase(mi++);
210  } else
211  mi++;
212  }
213 
214  // Check if this alert has been cancelled
215  BOOST_FOREACH (PAIRTYPE(const uint256, CAlert) & item, mapAlerts) {
216  const CAlert& alert = item.second;
217  if (alert.Cancels(*this)) {
218  LogPrint("alert", "alert already cancelled by %d\n", alert.nID);
219  return false;
220  }
221  }
222 
223  // Add to mapAlerts
224  mapAlerts.insert(make_pair(GetHash(), *this));
225  // Notify UI and -alertnotify if it applies to me
226  if (AppliesToMe()) {
227  uiInterface.NotifyAlertChanged(GetHash(), CT_NEW);
228  Notify(strStatusBar, fThread);
229  }
230  }
231 
232  LogPrint("alert", "accepted alert %d, AppliesToMe()=%d\n", nID, AppliesToMe());
233  return true;
234 }

236 void CAlert::Notify(const std::string& strMessage, bool fThread)
237 {
238  std::string strCmd = GetArg("-alertnotify", "");
239  if (strCmd.empty()) return;
240 
241  // Alert text should be plain ascii coming from a trusted source, but to
242  // be safe we first strip anything not in safeChars, then add single quotes around
243  // the whole string before passing it to the shell:
244  std::string singleQuote("'");
245  std::string safeStatus = SanitizeString(strMessage);
246  safeStatus = singleQuote + safeStatus + singleQuote;
247  boost::replace_all(strCmd, "%s", safeStatus);
248 
249  if (fThread)
250  boost::thread t(runCommand, strCmd); // thread runs free
251  else
252  runCommand(strCmd);
253 }
