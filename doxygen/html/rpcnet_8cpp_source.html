188

190  if (it != vAddedNodes.end())
191  throw JSONRPCError(RPC_CLIENT_NODE_ALREADY_ADDED, "Error: Node already added");

194  if (it == vAddedNodes.end())
195  throw JSONRPCError(RPC_CLIENT_NODE_NOT_ADDED, "Error: Node has not been added.");
196  vAddedNodes.erase(it);
197  }
