120 static void RegisterLoad(const string& strInput)
121 {
122  // separate NAME:FILENAME in string
123  size_t pos = strInput.find(':');
124  if ((pos == string::npos) ||
125  (pos == 0) ||
126  (pos == (strInput.size() - 1)))
127  throw runtime_error("Register load requires NAME:FILENAME");
128 
129  string key = strInput.substr(0, pos);
130  string filename = strInput.substr(pos + 1, string::npos);
131 
132  FILE* f = fopen(filename.c_str(), "r");
133  if (!f) {
134  string strErr = "Cannot open file " + filename;
135  throw runtime_error(strErr);
136  }
137 
138  // load file chunks into one big buffer
139  string valStr;
140  while ((!feof(f)) && (!ferror(f))) {
141  char buf[4096];
142  int bread = fread(buf, 1, sizeof(buf), f);
143  if (bread <= 0)
144  break;
145 
146  valStr.insert(valStr.size(), buf, bread);
147  }
148 
149  if (ferror(f)) {
150  string strErr = "Error reading file " + filename;
151  throw runtime_error(strErr);
152  }
153 
154  fclose(f);
155 
156  // evaluate as JSON buffer register
157  RegisterSetJson(key, valStr);
158 }

329 static void MutateTxSign(CMutableTransaction& tx, const string& flagStr)
330 {
331  int nHashType = SIGHASH_ALL;
332 
333  if (flagStr.size() > 0)
334  if (!findSighashFlags(nHashType, flagStr))
335  throw runtime_error("unknown sighash flag/sign option");
336 
337  vector<CTransaction> txVariants;
338  txVariants.push_back(tx);
339 
340  // mergedTx will end up with all the signatures; it
341  // starts as a clone of the raw tx:
342  CMutableTransaction mergedTx(txVariants[0]);
343  bool fComplete = true;
344  CCoinsView viewDummy;
345  CCoinsViewCache view(&viewDummy);
346 
347  if (!registers.count("privatekeys"))
348  throw runtime_error("privatekeys register variable must be set.");
349  bool fGivenKeys = false;
350  CBasicKeyStore tempKeystore;
351  UniValue keysObj = registers["privatekeys"];
352  fGivenKeys = true;
353 
354  for (unsigned int kidx = 0; kidx < keysObj.count(); kidx++) {
355  if (!keysObj[kidx].isStr())
356  throw runtime_error("privatekey not a string");
357  CBitcoinSecret vchSecret;
358  bool fGood = vchSecret.SetString(keysObj[kidx].getValStr());
359  if (!fGood)
360  throw runtime_error("privatekey not valid");
361 
362  CKey key = vchSecret.GetKey();
363  tempKeystore.AddKey(key);
364  }
365 
366  // Add previous txouts given in the RPC call:
367  if (!registers.count("prevtxs"))
368  throw runtime_error("prevtxs register variable must be set.");
369  UniValue prevtxsObj = registers["prevtxs"];
370  {
371  for (unsigned int previdx = 0; previdx < prevtxsObj.count(); previdx++) {
372  UniValue prevOut = prevtxsObj[previdx];
373  if (!prevOut.isObject())
374  throw runtime_error("expected prevtxs internal object");
375 
376  map<string, UniValue::VType> types = map_list_of("txid", UniValue::VSTR)("vout", UniValue::VNUM)("scriptPubKey", UniValue::VSTR);
377  if (!prevOut.checkObject(types))
378  throw runtime_error("prevtxs internal object typecheck fail");
379 
380  uint256 txid = ParseHashUV(prevOut["txid"], "txid");
381 
382  int nOut = atoi(prevOut["vout"].getValStr());
383  if (nOut < 0)
384  throw runtime_error("vout must be positive");
385 
386  vector<unsigned char> pkData(ParseHexUV(prevOut["scriptPubKey"], "scriptPubKey"));
387  CScript scriptPubKey(pkData.begin(), pkData.end());
388 
389  {
390  CCoinsModifier coins = view.ModifyCoins(txid);
391  if (coins->IsAvailable(nOut) && coins->vout[nOut].scriptPubKey != scriptPubKey) {
392  string err("Previous output scriptPubKey mismatch:\n");
393  err = err + coins->vout[nOut].scriptPubKey.ToString() + "\nvs:\n" +
394  scriptPubKey.ToString();
395  throw runtime_error(err);
396  }
397  if ((unsigned int)nOut >= coins->vout.size())
398  coins->vout.resize(nOut + 1);
399  coins->vout[nOut].scriptPubKey = scriptPubKey;
400  coins->vout[nOut].nValue = 0; // we don't know the actual output value
401  }
402 
403  // if redeemScript given and private keys given,
404  // add redeemScript to the tempKeystore so it can be signed:
405  if (fGivenKeys && scriptPubKey.IsPayToScriptHash() &&
406  prevOut.exists("redeemScript")) {
407  UniValue v = prevOut["redeemScript"];
408  vector<unsigned char> rsData(ParseHexUV(v, "redeemScript"));
409  CScript redeemScript(rsData.begin(), rsData.end());
410  tempKeystore.AddCScript(redeemScript);
411  }
412  }
413  }
414 
415  const CKeyStore& keystore = tempKeystore;
416 
417  bool fHashSingle = ((nHashType & ~SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE);
418 
419  // Sign what we can:
420  for (unsigned int i = 0; i < mergedTx.vin.size(); i++) {
421  CTxIn& txin = mergedTx.vin[i];
422  const CCoins* coins = view.AccessCoins(txin.prevout.hash);
423  if (!coins || !coins->IsAvailable(txin.prevout.n)) {
424  fComplete = false;
425  continue;
426  }
427  const CScript& prevPubKey = coins->vout[txin.prevout.n].scriptPubKey;
428 
429  txin.scriptSig.clear();
430  // Only sign SIGHASH_SINGLE if there's a corresponding output:
431  if (!fHashSingle || (i < mergedTx.vout.size()))
432  SignSignature(keystore, prevPubKey, mergedTx, i, nHashType);
433 
434  // ... and merge in other signatures:
435  BOOST_FOREACH (const CTransaction& txv, txVariants) {
436  txin.scriptSig = CombineSignatures(prevPubKey, mergedTx, i, txin.scriptSig, txv.vin[i].scriptSig);
437  }
438  if (!VerifyScript(txin.scriptSig, prevPubKey, STANDARD_SCRIPT_VERIFY_FLAGS, MutableTransactionSignatureChecker(&mergedTx, i)))
439  fComplete = false;
440  }
441 
442  if (fComplete) {
443  // do nothing... for now
444  // perhaps store this for later optional JSON output
445  }
446 
447  tx = mergedTx;
448 }

450 static void MutateTx(CMutableTransaction& tx, const string& command, const string& commandVal)
451 {
452  if (command == "nversion")
453  MutateTxVersion(tx, commandVal);
454  else if (command == "locktime")
455  MutateTxLocktime(tx, commandVal);
456 
457  else if (command == "delin")
458  MutateTxDelInput(tx, commandVal);
459  else if (command == "in")
460  MutateTxAddInput(tx, commandVal);
461 
462  else if (command == "delout")
463  MutateTxDelOutput(tx, commandVal);
464  else if (command == "outaddr")
465  MutateTxAddOutAddr(tx, commandVal);
466  else if (command == "outscript")
467  MutateTxAddOutScript(tx, commandVal);
468 
469  else if (command == "sign")
470  MutateTxSign(tx, commandVal);
471 
472  else if (command == "load")
473  RegisterLoad(commandVal);
474 
475  else if (command == "set")
476  RegisterSet(commandVal);
477 
478  else
479  throw runtime_error("unknown command");
480 }

554

556  string strHexTx(argv[1]);
557  if (strHexTx == "-") // "-" implies standard input
558  strHexTx = readStdin();

562 
563  startArg = 2;
564  } else

567  CMutableTransaction tx(txDecodeTmp);

573  if (eqpos == string::npos)

577  value = arg.substr(eqpos + 1);

594  }
595
