135 void WalletModel::pollBalanceChanged()
136 {
137  // Get required locks upfront. This avoids the GUI from getting stuck on
138  // periodical polls if the core is holding the locks for a longer time -
139  // for example, during a wallet rescan.
140  TRY_LOCK(cs_main, lockMain);
141  if (!lockMain)
142  return;
143  TRY_LOCK(wallet->cs_wallet, lockWallet);
144  if (!lockWallet)
145  return;
146 
147  if (fForceCheckBalanceChanged || chainActive.Height() != cachedNumBlocks || nZeromintPercentage != cachedZeromintPercentage || cachedTxLocks != nCompleteTXLocks) {
148  fForceCheckBalanceChanged = false;
149 
150  // Balance and number of transactions might have changed
151  cachedNumBlocks = chainActive.Height();
152  cachedZeromintPercentage = nZeromintPercentage;
153 
154  checkBalanceChanged();
155  if (transactionTableModel) {
156  transactionTableModel->updateConfirmations();
157  }
158  }
159 }

169 void WalletModel::checkBalanceChanged()
170 {
171  TRY_LOCK(cs_main, lockMain);
172  if (!lockMain) return;
173 
174  CAmount newBalance = getBalance();
175  CAmount newUnconfirmedBalance = getUnconfirmedBalance();
176  CAmount newImmatureBalance = getImmatureBalance();
177  CAmount newZerocoinBalance = getZerocoinBalance();
178  CAmount newUnconfirmedZerocoinBalance = getUnconfirmedZerocoinBalance();
179  CAmount newImmatureZerocoinBalance = getImmatureZerocoinBalance();
180  CAmount newWatchOnlyBalance = 0;
181  CAmount newWatchUnconfBalance = 0;
182  CAmount newWatchImmatureBalance = 0;
183  if (haveWatchOnly()) {
184  newWatchOnlyBalance = getWatchBalance();
185  newWatchUnconfBalance = getWatchUnconfirmedBalance();
186  newWatchImmatureBalance = getWatchImmatureBalance();
187  }
188 
189  if (cachedBalance != newBalance || cachedUnconfirmedBalance != newUnconfirmedBalance || cachedImmatureBalance != newImmatureBalance ||
190  cachedZerocoinBalance != newZerocoinBalance || cachedUnconfirmedZerocoinBalance != newUnconfirmedZerocoinBalance || cachedImmatureZerocoinBalance != newImmatureZerocoinBalance ||
191  cachedWatchOnlyBalance != newWatchOnlyBalance || cachedWatchUnconfBalance != newWatchUnconfBalance || cachedWatchImmatureBalance != newWatchImmatureBalance ||
192  cachedTxLocks != nCompleteTXLocks ) {
193  cachedBalance = newBalance;
194  cachedUnconfirmedBalance = newUnconfirmedBalance;
195  cachedImmatureBalance = newImmatureBalance;
196  cachedZerocoinBalance = newZerocoinBalance;
197  cachedUnconfirmedZerocoinBalance = newUnconfirmedZerocoinBalance;
198  cachedImmatureZerocoinBalance = newImmatureZerocoinBalance;
199  cachedTxLocks = nCompleteTXLocks;
200  cachedWatchOnlyBalance = newWatchOnlyBalance;
201  cachedWatchUnconfBalance = newWatchUnconfBalance;
202  cachedWatchImmatureBalance = newWatchImmatureBalance;
203  emit balanceChanged(newBalance, newUnconfirmedBalance, newImmatureBalance,
204  newZerocoinBalance, newUnconfirmedZerocoinBalance, newImmatureZerocoinBalance,
205  newWatchOnlyBalance, newWatchUnconfBalance, newWatchImmatureBalance);
206  }

245 WalletModel::SendCoinsReturn WalletModel::prepareTransaction(WalletModelTransaction& transaction, const CCoinControl* coinControl)
246 {
247  CAmount total = 0;
248  QList<SendCoinsRecipient> recipients = transaction.getRecipients();
249  std::vector<std::pair<CScript, CAmount> > vecSend;
250 
251  if (recipients.empty()) {
252  return OK;
253  }
254 
255  if (isAnonymizeOnlyUnlocked()) {
256  return AnonymizeOnlyUnlocked;
257  }
258 
259  QSet<QString> setAddress; // Used to detect duplicates
260  int nAddresses = 0;
261 
262  // Pre-check input data for validity
263  foreach (const SendCoinsRecipient& rcp, recipients) {
264  if (rcp.paymentRequest.IsInitialized()) { // PaymentRequest...
265  CAmount subtotal = 0;
266  const payments::PaymentDetails& details = rcp.paymentRequest.getDetails();
267  for (int i = 0; i < details.outputs_size(); i++) {
268  const payments::Output& out = details.outputs(i);
269  if (out.amount() <= 0) continue;
270  subtotal += out.amount();
271  const unsigned char* scriptStr = (const unsigned char*)out.script().data();
272  CScript scriptPubKey(scriptStr, scriptStr + out.script().size());
273  vecSend.push_back(std::pair<CScript, CAmount>(scriptPubKey, out.amount()));
274  }
275  if (subtotal <= 0) {
276  return InvalidAmount;
277  }
278  total += subtotal;
279  } else { // User-entered divi address / amount:
280  if (!validateAddress(rcp.address)) {
281  return InvalidAddress;
282  }
283  if (rcp.amount <= 0) {
284  return InvalidAmount;
285  }
286  setAddress.insert(rcp.address);
287  ++nAddresses;
288 
289  CScript scriptPubKey = GetScriptForDestination(CBitcoinAddress(rcp.address.toStdString()).Get());
290  vecSend.push_back(std::pair<CScript, CAmount>(scriptPubKey, rcp.amount));
291 
292  total += rcp.amount;
293  }
294  }
295  if (setAddress.size() != nAddresses) {
296  return DuplicateAddress;
297  }
298 
299  CAmount nBalance = getBalance(coinControl);
300 
301  if (total > nBalance) {
302  return AmountExceedsBalance;
303  }
304 
305  {
306  LOCK2(cs_main, wallet->cs_wallet);
307 
308  transaction.newPossibleKeyChange(wallet);
309  CAmount nFeeRequired = 0;
310  std::string strFailReason;
311 
312  CWalletTx* newTx = transaction.getTransaction();
313  CReserveKey* keyChange = transaction.getPossibleKeyChange();
314 
315 
316  if (recipients[0].useSwiftTX && total > GetSporkValue(SPORK_5_MAX_VALUE) * COIN) {
317  emit message(tr("Send Coins"), tr("SwiftX doesn't support sending values that high yet. Transactions are currently limited to %1 DIV.").arg(GetSporkValue(SPORK_5_MAX_VALUE)),
318  CClientUIInterface::MSG_ERROR);
319  return TransactionCreationFailed;
320  }
321 
322  bool fCreated = wallet->CreateTransaction(vecSend, *newTx, *keyChange, nFeeRequired, strFailReason, coinControl, recipients[0].inputType, recipients[0].useSwiftTX);
323  transaction.setTransactionFee(nFeeRequired);
324 
325  if (recipients[0].useSwiftTX && newTx->GetValueOut() > GetSporkValue(SPORK_5_MAX_VALUE) * COIN) {
326  emit message(tr("Send Coins"), tr("SwiftX doesn't support sending values that high yet. Transactions are currently limited to %1 DIV.").arg(GetSporkValue(SPORK_5_MAX_VALUE)),
327  CClientUIInterface::MSG_ERROR);
328  return TransactionCreationFailed;
329  }
330 
331  if (!fCreated) {
332  if ((total + nFeeRequired) > nBalance) {
333  return SendCoinsReturn(AmountWithFeeExceedsBalance);
334  }
335  emit message(tr("Send Coins"), QString::fromStdString(strFailReason),
336  CClientUIInterface::MSG_ERROR);
337  return TransactionCreationFailed;
338  }
339 
340  // reject insane fee
341  if (nFeeRequired > ::minRelayTxFee.GetFee(transaction.getTransactionSize()) * 10000)
342  return InsaneFee;
343  }
344 
345  return SendCoinsReturn(OK);
346 }

348 WalletModel::SendCoinsReturn WalletModel::sendCoins(WalletModelTransaction& transaction)
349 {
350  QByteArray transaction_array; /* store serialized transaction */
351 
352  if (isAnonymizeOnlyUnlocked()) {
353  return AnonymizeOnlyUnlocked;
354  }
355 
356  {
357  LOCK2(cs_main, wallet->cs_wallet);
358  CWalletTx* newTx = transaction.getTransaction();
359  QList<SendCoinsRecipient> recipients = transaction.getRecipients();
360 
361  // Store PaymentRequests in wtx.vOrderForm in wallet.
362  foreach (const SendCoinsRecipient& rcp, recipients) {
363  if (rcp.paymentRequest.IsInitialized()) {
364  std::string key("PaymentRequest");
365  std::string value;
366  rcp.paymentRequest.SerializeToString(&value);
367  newTx->vOrderForm.push_back(make_pair(key, value));
368  } else if (!rcp.message.isEmpty()) // Message from normal divi:URI (divi:XyZ...?message=example)
369  {
370  newTx->vOrderForm.push_back(make_pair("Message", rcp.message.toStdString()));
371  }
372  }
373 
374  CReserveKey* keyChange = transaction.getPossibleKeyChange();
375 
376  transaction.getRecipients();
377 
378  if (!wallet->CommitTransaction(*newTx, *keyChange, (recipients[0].useSwiftTX) ? "ix" : "tx"))
379  return TransactionCommitFailed;
380 
381  CTransaction* t = (CTransaction*)newTx;
382  CDataStream ssTx(SER_NETWORK, PROTOCOL_VERSION);
383  ssTx << *t;
384  transaction_array.append(&(ssTx[0]), ssTx.size());
385  }
386 
387  // Add addresses / update labels that we've sent to to the address book,
388  // and emit coinsSent signal for each recipient
389  foreach (const SendCoinsRecipient& rcp, transaction.getRecipients()) {
390  // Don't touch the address book when we have a payment request
391  if (!rcp.paymentRequest.IsInitialized()) {
392  std::string strAddress = rcp.address.toStdString();
393  CTxDestination dest = CBitcoinAddress(strAddress).Get();
394  std::string strLabel = rcp.label.toStdString();
395  {
396  LOCK(wallet->cs_wallet);
397 
398  std::map<CTxDestination, CAddressBookData>::iterator mi = wallet->mapAddressBook.find(dest);
399 
400  // Check if we have a new address or an updated label
401  if (mi == wallet->mapAddressBook.end()) {
402  wallet->SetAddressBook(dest, strLabel, "send");
403  } else if (mi->second.name != strLabel) {
404  wallet->SetAddressBook(dest, strLabel, ""); // "" means don't change purpose
405  }
406  }
407  }
408  emit coinsSent(wallet, rcp, transaction_array);
409  }
410  checkBalanceChanged(); // update balance immediately, otherwise there could be a short noticeable delay until pollBalanceChanged hits
411 
412  return SendCoinsReturn(OK);
413 }

657 {

660

662  std::vector<COutPoint> vLockedCoins;
663  wallet->ListLockedCoins(vLockedCoins);
664 
665  // add locked coins
666  BOOST_FOREACH (const COutPoint& outpoint, vLockedCoins) {
667  if (!wallet->mapWallet.count(outpoint.hash)) continue;
668  int nDepth = wallet->mapWallet[outpoint.hash].GetDepthInMainChain();
669  if (nDepth < 0) continue;
670  COutput out(&wallet->mapWallet[outpoint.hash], outpoint.n, nDepth, true);
671  if (outpoint.n < out.tx->vout.size() && wallet->IsMine(out.tx->vout[outpoint.n]) == ISMINE_SPENDABLE)
672  vCoins.push_back(out);
673  }
674 
675  BOOST_FOREACH (const COutput& out, vCoins) {
676  COutput cout = out;
677 
678  while (wallet->IsChange(cout.tx->vout[cout.i]) && cout.tx->vin.size() > 0 && wallet->IsMine(cout.tx->vin[0])) {
