155 // Init
156 class CInit
157 {
158 public:
159  CInit()
160  {
161  // Init OpenSSL library multithreading support
162  ppmutexOpenSSL = (CCriticalSection**)OPENSSL_malloc(CRYPTO_num_locks() * sizeof(CCriticalSection*));
163  for (int i = 0; i < CRYPTO_num_locks(); i++)
164  ppmutexOpenSSL[i] = new CCriticalSection();
165  CRYPTO_set_locking_callback(locking_callback);
166 
167  // OpenSSL can optionally load a config file which lists optional loadable modules and engines.
168  // We don't use them so we don't require the config. However some of our libs may call functions
169  // which attempt to load the config file, possibly resulting in an exit() or crash if it is missing
170  // or corrupt. Explicitly tell OpenSSL not to try to load the file. The result for our libs will be
171  // that the config appears to have been loaded and there are no modules/engines available.
172  OPENSSL_no_config();
173 
174 #ifdef WIN32
175  // Seed OpenSSL PRNG with current contents of the screen
176  RAND_screen();
177 #endif
178 
179  // Seed OpenSSL PRNG with performance counter
180  RandAddSeed();
181  }
182  ~CInit()
183  {
184  // Securely erase the memory used by the PRNG
185  RAND_cleanup();
186  // Shutdown OpenSSL library multithreading support
187  CRYPTO_set_locking_callback(NULL);
188  for (int i = 0; i < CRYPTO_num_locks(); i++)
189  delete ppmutexOpenSSL[i];
190  OPENSSL_free(ppmutexOpenSSL);
191  }
192 } instance_of_cinit;

226 {

228  if (!fDebug)

231  // Give each thread quick access to -debug settings.
232  // This helps prevent issues debugging global destructors,

236  if (ptrCategory.get() == NULL) {

315 void ParseParameters(int argc, const char* const argv[])
316 {
317  mapArgs.clear();
318  mapMultiArgs.clear();
319 
320  for (int i = 1; i < argc; i++) {
321  std::string str(argv[i]);
322  std::string strValue;
323  size_t is_index = str.find('=');
324  if (is_index != std::string::npos) {
325  strValue = str.substr(is_index + 1);
326  str = str.substr(0, is_index);
327  }
328 #ifdef WIN32
329  boost::to_lower(str);
330  if (boost::algorithm::starts_with(str, "/"))
331  str = "-" + str.substr(1);
332 #endif
333 
334  if (str[0] != '-')
335  break;
336 
337  // Interpret --foo as -foo.
338  // If both --foo and -foo are set, the last takes effect.
339  if (str.length() > 1 && str[1] == '-')
340  str = str.substr(1);
341  InterpretNegativeSetting(str, strValue);
342 
343  mapArgs[str] = strValue;
344  mapMultiArgs[str].push_back(strValue);
345  }
346 }

509 void ReadConfigFile(map<string, string>& mapSettingsRet,
510  map<string, vector<string> >& mapMultiSettingsRet)
511 {
512  boost::filesystem::ifstream streamConfig(GetConfigFile());
513  if (!streamConfig.good()) {
514  // Create empty divi.conf if it does not exist
515  FILE* configFile = fopen(GetConfigFile().string().c_str(), "a");
516  if (configFile != NULL)
517  fclose(configFile);
518  return; // Nothing to read, so just return
519  }
520 
521  set<string> setOptions;
522  setOptions.insert("*");
523 
524  for (boost::program_options::detail::config_file_iterator it(streamConfig, setOptions), end; it != end; ++it) {
525  // Don't overwrite existing settings so command line settings override divi.conf
526  string strKey = string("-") + it->string_key;
527  string strValue = it->value[0];
528  InterpretNegativeSetting(strKey, strValue);
529  if (mapSettingsRet.count(strKey) == 0)
530  mapSettingsRet[strKey] = strValue;
531  mapMultiSettingsRet[strKey].push_back(strValue);
532  }
533  // If datadir is changed in .conf file:
534  ClearDatadirCache();
535 }

638 void AllocateFileRange(FILE* file, unsigned int offset, unsigned int length)
639 {
640 #if defined(WIN32)
641  // Windows-specific version
642  HANDLE hFile = (HANDLE)_get_osfhandle(_fileno(file));
643  LARGE_INTEGER nFileSize;
644  int64_t nEndPos = (int64_t)offset + length;
645  nFileSize.u.LowPart = nEndPos & 0xFFFFFFFF;
646  nFileSize.u.HighPart = nEndPos >> 32;
647  SetFilePointerEx(hFile, nFileSize, 0, FILE_BEGIN);
648  SetEndOfFile(hFile);
649 #elif defined(MAC_OSX)
650  // OSX specific version
651  fstore_t fst;
652  fst.fst_flags = F_ALLOCATECONTIG;
653  fst.fst_posmode = F_PEOFPOSMODE;
654  fst.fst_offset = 0;
655  fst.fst_length = (off_t)offset + length;
656  fst.fst_bytesalloc = 0;
657  if (fcntl(fileno(file), F_PREALLOCATE, &fst) == -1) {
658  fst.fst_flags = F_ALLOCATEALL;
659  fcntl(fileno(file), F_PREALLOCATE, &fst);
660  }
661  ftruncate(fileno(file), fst.fst_length);
662 #elif defined(__linux__)
663  // Version using posix_fallocate
664  off_t nEndPos = (off_t)offset + length;
665  posix_fallocate(fileno(file), 0, nEndPos);
666 #else
667  // Fallback version
668  // TODO: just write one byte per block
669  static const char buf[65536] = {};
670  fseek(file, offset, SEEK_SET);
671  while (length > 0) {
672  unsigned int now = 65536;
673  if (length < now)
674  now = length;
675  fwrite(buf, 1, now, file); // allowed to fail; this function is advisory anyway
676  length -= now;
677  }
678 #endif
679 }

772 void SetupEnvironment()
773 {
774 // On most POSIX systems (e.g. Linux, but not BSD) the environment's locale
775 // may be invalid, in which case the "C" locale is used as fallback.
776 #if !defined(WIN32) && !defined(MAC_OSX) && !defined(__FreeBSD__) && !defined(__OpenBSD__)
777  try {

786  // boost::filesystem::path, which is then used to explicitly imbue the path.
787  std::locale loc = boost::filesystem::path::imbue(std::locale::classic());
788  boost::filesystem::path::imbue(loc);
