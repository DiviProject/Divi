107  void AddPayee(CScript payeeIn, int nIncrement)
108  {
109  LOCK(cs_vecPayments);
110 
111  BOOST_FOREACH (CMasternodePayee& payee, vecPayments) {
112  if (payee.scriptPubKey == payeeIn) {
113  payee.nVotes += nIncrement;
114  return;
115  }
116  }
117 
118  CMasternodePayee c(payeeIn, nIncrement);
119  vecPayments.push_back(c);
120  }
121 
122  bool GetPayee(CScript& payee)
123  {
124  LOCK(cs_vecPayments);
125 
126  int nVotes = -1;
127  BOOST_FOREACH (CMasternodePayee& p, vecPayments) {
128  if (p.nVotes > nVotes) {
129  payee = p.scriptPubKey;
130  nVotes = p.nVotes;
131  }
132  }
133 
134  return (nVotes > -1);
135  }
136 
137  bool HasPayeeWithVotes(CScript payee, int nVotesReq)
138  {
139  LOCK(cs_vecPayments);
140 
141  BOOST_FOREACH (CMasternodePayee& p, vecPayments) {
142  if (p.nVotes >= nVotesReq && p.scriptPubKey == payee) return true;
143  }
144 
145  return false;
146  }
