136 QString BitcoinUnits::format(int unit, const CAmount& nIn, bool fPlus, SeparatorStyle separators)
137 {
138  // Note: not using straight sprintf here because we do NOT want
139  // localized number formatting.
140  if (!valid(unit))
141  return QString(); // Refuse to format invalid unit
142  qint64 n = (qint64)nIn;
143  qint64 coin = factor(unit);
144  int num_decimals = decimals(unit);
145  qint64 n_abs = (n > 0 ? n : -n);
146  qint64 quotient = n_abs / coin;
147  qint64 remainder = n_abs % coin;
148  QString quotient_str = QString::number(quotient);
149  QString remainder_str = QString::number(remainder).rightJustified(num_decimals, '0');
150 
151  // Use SI-style thin space separators as these are locale independent and can't be
152  // confused with the decimal marker.
153  QChar thin_sp(THIN_SP_CP);
154  int q_size = quotient_str.size();
155  if (separators == separatorAlways || (separators == separatorStandard && q_size > 4))
156  for (int i = 3; i < q_size; i += 3)
157  quotient_str.insert(q_size - i, thin_sp);
158 
159  if (n < 0)
160  quotient_str.insert(0, '-');
161  else if (fPlus && n > 0)
162  quotient_str.insert(0, '+');
163 
164  if (num_decimals <= 0)
165  return quotient_str;
166 
167  return quotient_str + QString(".") + remainder_str;
168 }

216 bool BitcoinUnits::parse(int unit, const QString& value, CAmount* val_out)
217 {
218  if (!valid(unit) || value.isEmpty())
219  return false; // Refuse to parse invalid unit or empty string
220  int num_decimals = decimals(unit);
221 
222  // Ignore spaces and thin spaces when parsing
223  QStringList parts = removeSpaces(value).split(".");
224 
225  if (parts.size() > 2) {
226  return false; // More than one dot
227  }
228  QString whole = parts[0];
229  QString decimals;
230 
231  if (parts.size() > 1) {
232  decimals = parts[1];
233  }
234  if (decimals.size() > num_decimals) {
235  return false; // Exceeds max precision
236  }
237  bool ok = false;
238  QString str = whole + decimals.leftJustified(num_decimals, '0');
239 
240  if (str.size() > 18) {
241  return false; // Longer numbers will exceed 63 bits
242  }
243  CAmount retvalue(str.toLongLong(&ok));
244  if (val_out) {
245  *val_out = retvalue;
246  }
247  return ok;
248 }
