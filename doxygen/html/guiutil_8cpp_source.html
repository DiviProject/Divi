128 bool parseBitcoinURI(const QUrl& uri, SendCoinsRecipient* out)
129 {
130  // return if URI is not valid or is no DIVI: URI
131  if (!uri.isValid() || uri.scheme() != QString(URI_SCHEME))
132  return false;
133 
134  SendCoinsRecipient rv;
135  rv.address = uri.path();
136  // Trim any following forward slash which may have been added by the OS
137  if (rv.address.endsWith("/")) {
138  rv.address.truncate(rv.address.length() - 1);
139  }
140  rv.amount = 0;
141 
142 #if QT_VERSION < 0x050000
143  QList<QPair<QString, QString> > items = uri.queryItems();
144 #else
145  QUrlQuery uriQuery(uri);
146  QList<QPair<QString, QString> > items = uriQuery.queryItems();
147 #endif
148  for (QList<QPair<QString, QString> >::iterator i = items.begin(); i != items.end(); i++) {
149  bool fShouldReturnFalse = false;
150  if (i->first.startsWith("req-")) {
151  i->first.remove(0, 4);
152  fShouldReturnFalse = true;
153  }
154 
155  if (i->first == "label") {
156  rv.label = i->second;
157  fShouldReturnFalse = false;
158  }
159  if (i->first == "message") {
160  rv.message = i->second;
161  fShouldReturnFalse = false;
162  } else if (i->first == "amount") {
163  if (!i->second.isEmpty()) {
164  if (!BitcoinUnits::parse(BitcoinUnits::DIV, i->second, &rv.amount)) {
165  return false;
166  }
167  }
168  fShouldReturnFalse = false;
169  }
170 
171  if (fShouldReturnFalse)
172  return false;
173  }
174  if (out) {
175  *out = rv;
176  }
177  return true;
178 }

180 bool parseBitcoinURI(QString uri, SendCoinsRecipient* out)
181 {
182  // Convert divi:// to divi:
183  //
184  // Cannot handle this later, because divi:// will cause Qt to see the part after // as host,
185  // which will lower-case it (and thus invalidate the address).
186  if (uri.startsWith(URI_SCHEME "://", Qt::CaseInsensitive)) {
187  uri.replace(0, std::strlen(URI_SCHEME) + 3, URI_SCHEME ":");
188  }
189  QUrl uriInstance(uri);
190  return parseBitcoinURI(uriInstance, out);

201  }
202 
203  if (!info.label.isEmpty()) {

208 
209  if (!info.message.isEmpty()) {
210  QString msg(QUrl::toPercentEncoding(info.message));
211  ret += QString("%1message=%2").arg(paramCount == 0 ? "?" : "&").arg(msg);
212  paramCount++;
213  }

258 {

264  myDir = QDesktopServices::storageLocation(QDesktopServices::DocumentsLocation);
265 #else
266  myDir = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
267 #endif
268  } else {
269  myDir = dir;
270  }
271  /* Directly convert path to native OS path separators */
272  QString result = QDir::toNativeSeparators(QFileDialog::getSaveFileName(parent, caption, myDir, filter, &selectedFilter));
273 
274  /* Extract first suffix from filter pattern "Description (*.foo)" or "Description (*.foo *.bar ...) */
275  QRegExp filter_re(".* \\(\\*\\.(.*)[ \\)]");
276  QString selectedSuffix;
277  if (filter_re.exactMatch(selectedFilter)) {
278  selectedSuffix = filter_re.cap(1);

280 
281  /* Add suffix if needed */
282  QFileInfo info(result);
283  if (!result.isEmpty()) {
284  if (info.suffix().isEmpty() && !selectedSuffix.isEmpty()) {
285  /* No suffix specified, add selected suffix */
286  if (!result.endsWith("."))
287  result.append(".");
288  result.append(selectedSuffix);
289  }
290  }
291 
292  /* Return selected suffix if asked to */
293  if (selectedSuffixOut) {
294  *selectedSuffixOut = selectedSuffix;
295  }
296  return result;

385 void SubstituteFonts(const QString& language)
386 {
387 #if defined(Q_OS_MAC)
388 // Background:
389 // OSX's default font changed in 10.9 and QT is unable to find it with its
390 // usual fallback methods when building against the 10.7 sdk or lower.
391 // The 10.8 SDK added a function to let it find the correct fallback font.
392 // If this fallback is not properly loaded, some characters may fail to
393 // render correctly.
394 //
395 // The same thing happened with 10.10. .Helvetica Neue DeskInterface is now default.
396 //
397 // Solution: If building with the 10.7 SDK or lower and the user's platform
398 // is 10.9 or higher at runtime, substitute the correct font. This needs to
399 // happen before the QApplication is created.
400 #if defined(MAC_OS_X_VERSION_MAX_ALLOWED) && MAC_OS_X_VERSION_MAX_ALLOWED < MAC_OS_X_VERSION_10_8
401  if (floor(NSAppKitVersionNumber) > NSAppKitVersionNumber10_8) {
402  if (floor(NSAppKitVersionNumber) <= NSAppKitVersionNumber10_9)
403  /* On a 10.9 - 10.9.x system */
404  QFont::insertSubstitution(".Lucida Grande UI", "Lucida Grande");
405  else {
406  /* 10.10 or later system */
407  if (language == "zh_CN" || language == "zh_TW" || language == "zh_HK") // traditional or simplified Chinese
408  QFont::insertSubstitution(".Helvetica Neue DeskInterface", "Heiti SC");
409  else if (language == "ja") // Japanesee
410  QFont::insertSubstitution(".Helvetica Neue DeskInterface", "Songti SC");
411  else
412  QFont::insertSubstitution(".Helvetica Neue DeskInterface", "Lucida Grande");
413  }
414  }
415 #endif
416 #endif
417 }

593 bool SetStartOnSystemStartup(bool fAutoStart)
594 {
595  // If the shortcut exists already, remove it for updating
596  boost::filesystem::remove(StartupShortcutPath());
597 
598  if (fAutoStart) {
599  CoInitialize(NULL);
600 
601  // Get a pointer to the IShellLink interface.
602  IShellLink* psl = NULL;
603  HRESULT hres = CoCreateInstance(CLSID_ShellLink, NULL,
604  CLSCTX_INPROC_SERVER, IID_IShellLink,
605  reinterpret_cast<void**>(&psl));
606 
607  if (SUCCEEDED(hres)) {
608  // Get the current executable path
609  TCHAR pszExePath[MAX_PATH];
610  GetModuleFileName(NULL, pszExePath, sizeof(pszExePath));
611 
612  TCHAR pszArgs[5] = TEXT("-min");
613 
614  // Set the path to the shortcut target
615  psl->SetPath(pszExePath);
616  PathRemoveFileSpec(pszExePath);
617  psl->SetWorkingDirectory(pszExePath);
618  psl->SetShowCmd(SW_SHOWMINNOACTIVE);
619  psl->SetArguments(pszArgs);
620 
621  // Query IShellLink for the IPersistFile interface for
622  // saving the shortcut in persistent storage.
623  IPersistFile* ppf = NULL;
624  hres = psl->QueryInterface(IID_IPersistFile,
625  reinterpret_cast<void**>(&ppf));
626  if (SUCCEEDED(hres)) {
627  WCHAR pwsz[MAX_PATH];
628  // Ensure that the string is ANSI.
629  MultiByteToWideChar(CP_ACP, 0, StartupShortcutPath().string().c_str(), -1, pwsz, MAX_PATH);
630  // Save the link by calling IPersistFile::Save.
631  hres = ppf->Save(pwsz, TRUE);
632  ppf->Release();
633  psl->Release();
634  CoUninitialize();
635  return true;
636  }
637  psl->Release();
638  }
639  CoUninitialize();
640  return false;
641  }
642  return true;
643 }
