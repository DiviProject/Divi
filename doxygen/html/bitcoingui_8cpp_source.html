882 void BitcoinGUI::setNumBlocks(int count)
883 {
884  if (!clientModel)
885  return;
886 
887  // Prevent orphan statusbar messages (e.g. hover Quit in main menu, wait until chain-sync starts -> garbelled text)
888  statusBar()->clearMessage();
889 
890  // Acquire current block source
891  enum BlockSource blockSource = clientModel->getBlockSource();
892  switch (blockSource) {
893  case BLOCK_SOURCE_NETWORK:
894  progressBarLabel->setText(tr("Synchronizing with network..."));
895  break;
896  case BLOCK_SOURCE_DISK:
897  progressBarLabel->setText(tr("Importing blocks from disk..."));
898  break;
899  case BLOCK_SOURCE_REINDEX:
900  progressBarLabel->setText(tr("Reindexing blocks on disk..."));
901  break;
902  case BLOCK_SOURCE_NONE:
903  // Case: not Importing, not Reindexing and no network connection
904  progressBarLabel->setText(tr("No block source available..."));
905  break;
906  }
907 
908  QString tooltip;
909 
910  QDateTime lastBlockDate = clientModel->getLastBlockDate();
911  QDateTime currentDate = QDateTime::currentDateTime();
912  int secs = lastBlockDate.secsTo(currentDate);
913 
914  tooltip = tr("Processed %n blocks of transaction history.", "", count);
915 
916  // Set icon state: spinning if catching up, tick otherwise
917  // if(secs < 25*60) // 90*60 for bitcoin but we are 4x times faster
918  if (masternodeSync.IsBlockchainSynced()) {
919  QString strSyncStatus;
920  tooltip = tr("Up to date") + QString(".<br>") + tooltip;
921 
922  if (masternodeSync.IsSynced()) {
923  progressBarLabel->setVisible(false);
924  progressBar->setVisible(false);
925  labelBlocksIcon->setPixmap(QIcon(":/icons/synced").pixmap(STATUSBAR_ICONSIZE, STATUSBAR_ICONSIZE));
926  } else {
927  int nAttempt;
928  int progress = 0;
929 
930  labelBlocksIcon->setPixmap(QIcon(QString(
931  ":/movies/spinner-%1")
932  .arg(spinnerFrame, 3, 10, QChar('0')))
933  .pixmap(STATUSBAR_ICONSIZE, STATUSBAR_ICONSIZE));
934  spinnerFrame = (spinnerFrame + 1) % SPINNER_FRAMES;
935 
936 #ifdef ENABLE_WALLET
937  if (walletFrame)
938  walletFrame->showOutOfSyncWarning(false);
939 #endif // ENABLE_WALLET
940 
941  nAttempt = masternodeSync.RequestedMasternodeAttempt < MASTERNODE_SYNC_THRESHOLD ?
942  masternodeSync.RequestedMasternodeAttempt + 1 :
943  MASTERNODE_SYNC_THRESHOLD;
944  progress = nAttempt + (masternodeSync.RequestedMasternodeAssets - 1) * MASTERNODE_SYNC_THRESHOLD;
945  progressBar->setMaximum(4 * MASTERNODE_SYNC_THRESHOLD);
946  progressBar->setFormat(tr("Synchronizing additional data: %p%"));
947  progressBar->setValue(progress);
948  }
949 
950  strSyncStatus = QString(masternodeSync.GetSyncStatus().c_str());
951  progressBarLabel->setText(strSyncStatus);
952  tooltip = strSyncStatus + QString("<br>") + tooltip;
953  } else {
954  // Represent time from last generated block in human readable text
955  QString timeBehindText;
956  const int HOUR_IN_SECONDS = 60 * 60;
957  const int DAY_IN_SECONDS = 24 * 60 * 60;
958  const int WEEK_IN_SECONDS = 7 * 24 * 60 * 60;
959  const int YEAR_IN_SECONDS = 31556952; // Average length of year in Gregorian calendar
960  if (secs < 2 * DAY_IN_SECONDS) {
961  timeBehindText = tr("%n hour(s)", "", secs / HOUR_IN_SECONDS);
962  } else if (secs < 2 * WEEK_IN_SECONDS) {
963  timeBehindText = tr("%n day(s)", "", secs / DAY_IN_SECONDS);
964  } else if (secs < YEAR_IN_SECONDS) {
965  timeBehindText = tr("%n week(s)", "", secs / WEEK_IN_SECONDS);
966  } else {
967  int years = secs / YEAR_IN_SECONDS;
968  int remainder = secs % YEAR_IN_SECONDS;
969  timeBehindText = tr("%1 and %2").arg(tr("%n year(s)", "", years)).arg(tr("%n week(s)", "", remainder / WEEK_IN_SECONDS));
970  }
971 
972  progressBarLabel->setVisible(true);
973  progressBar->setFormat(tr("%1 behind. Scanning block %2").arg(timeBehindText).arg(count));
974  progressBar->setMaximum(1000000000);
975  progressBar->setValue(clientModel->getVerificationProgress() * 1000000000.0 + 0.5);
976  progressBar->setVisible(true);
977 
978  tooltip = tr("Catching up...") + QString("<br>") + tooltip;
979  if (count != prevBlocks) {
980  labelBlocksIcon->setPixmap(QIcon(QString(
981  ":/movies/spinner-%1")
982  .arg(spinnerFrame, 3, 10, QChar('0')))
983  .pixmap(STATUSBAR_ICONSIZE, STATUSBAR_ICONSIZE));
984  spinnerFrame = (spinnerFrame + 1) % SPINNER_FRAMES;
985  }
986  prevBlocks = count;
987 
988 #ifdef ENABLE_WALLET
989  if (walletFrame)
990  walletFrame->showOutOfSyncWarning(true);
991 #endif // ENABLE_WALLET
992 
993  tooltip += QString("<br>");
994  tooltip += tr("Last received block was generated %1 ago.").arg(timeBehindText);
995  tooltip += QString("<br>");
996  tooltip += tr("Transactions after this will not yet be visible.");
997  }
998 
999  // Don't word-wrap this (fixed-width) tooltip
1000  tooltip = QString("<nobr>") + tooltip + QString("</nobr>");
1001 
1002  labelBlocksIcon->setToolTip(tooltip);
1003  progressBarLabel->setToolTip(tooltip);
1004  progressBar->setToolTip(tooltip);
1005 }
