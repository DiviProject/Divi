66 bool PaymentRequestPlus::getMerchant(X509_STORE* certStore, QString& merchant) const
67 {
68  merchant.clear();
69 
70  if (!IsInitialized())
71  return false;
72 
73  // One day we'll support more PKI types, but just
74  // x509 for now:
75  const EVP_MD* digestAlgorithm = NULL;
76  if (paymentRequest.pki_type() == "x509+sha256") {
77  digestAlgorithm = EVP_sha256();
78  } else if (paymentRequest.pki_type() == "x509+sha1") {
79  digestAlgorithm = EVP_sha1();
80  } else if (paymentRequest.pki_type() == "none") {
81  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: pki_type == none";
82  return false;
83  } else {
84  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: unknown pki_type " << QString::fromStdString(paymentRequest.pki_type());
85  return false;
86  }
87 
88  payments::X509Certificates certChain;
89  if (!certChain.ParseFromString(paymentRequest.pki_data())) {
90  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: error parsing pki_data";
91  return false;
92  }
93 
94  std::vector<X509*> certs;
95  const QDateTime currentTime = QDateTime::currentDateTime();
96  for (int i = 0; i < certChain.certificate_size(); i++) {
97  QByteArray certData(certChain.certificate(i).data(), certChain.certificate(i).size());
98  QSslCertificate qCert(certData, QSsl::Der);
99  if (currentTime < qCert.effectiveDate() || currentTime > qCert.expiryDate()) {
100  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: certificate expired or not yet active: " << qCert;
101  return false;
102  }
103 #if QT_VERSION >= 0x050000
104  if (qCert.isBlacklisted()) {
105  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: certificate blacklisted: " << qCert;
106  return false;
107  }
108 #endif
109  const unsigned char* data = (const unsigned char*)certChain.certificate(i).data();
110  X509* cert = d2i_X509(NULL, &data, certChain.certificate(i).size());
111  if (cert)
112  certs.push_back(cert);
113  }
114  if (certs.empty()) {
115  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: empty certificate chain";
116  return false;
117  }
118 
119  // The first cert is the signing cert, the rest are untrusted certs that chain
120  // to a valid root authority. OpenSSL needs them separately.
121  STACK_OF(X509)* chain = sk_X509_new_null();
122  for (int i = certs.size() - 1; i > 0; i--) {
123  sk_X509_push(chain, certs[i]);
124  }
125  X509* signing_cert = certs[0];
126 
127  // Now create a "store context", which is a single use object for checking,
128  // load the signing cert into it and verify.
129  X509_STORE_CTX* store_ctx = X509_STORE_CTX_new();
130  if (!store_ctx) {
131  qWarning() << "PaymentRequestPlus::getMerchant : Payment request: error creating X509_STORE_CTX";
132  return false;
133  }
134 
135  char* website = NULL;
136  bool fResult = true;
137  try {
138  if (!X509_STORE_CTX_init(store_ctx, certStore, signing_cert, chain)) {
139  int error = X509_STORE_CTX_get_error(store_ctx);
140  throw SSLVerifyError(X509_verify_cert_error_string(error));
141  }
142 
143  // Now do the verification!
144  int result = X509_verify_cert(store_ctx);
145  if (result != 1) {
146  int error = X509_STORE_CTX_get_error(store_ctx);
147  throw SSLVerifyError(X509_verify_cert_error_string(error));
148  }
149  X509_NAME* certname = X509_get_subject_name(signing_cert);
150 
151  // Valid cert; check signature:
152  payments::PaymentRequest rcopy(paymentRequest); // Copy
153  rcopy.set_signature(std::string(""));
154  std::string data_to_verify; // Everything but the signature
155  rcopy.SerializeToString(&data_to_verify);
156 
157  EVP_MD_CTX ctx;
158  EVP_PKEY* pubkey = X509_get_pubkey(signing_cert);
159  EVP_MD_CTX_init(&ctx);
160  if (!EVP_VerifyInit_ex(&ctx, digestAlgorithm, NULL) ||
161  !EVP_VerifyUpdate(&ctx, data_to_verify.data(), data_to_verify.size()) ||
162  !EVP_VerifyFinal(&ctx, (const unsigned char*)paymentRequest.signature().data(), paymentRequest.signature().size(), pubkey)) {
163  throw SSLVerifyError("Bad signature, invalid PaymentRequest.");
164  }
165 
166  // OpenSSL API for getting human printable strings from certs is baroque.
167  int textlen = X509_NAME_get_text_by_NID(certname, NID_commonName, NULL, 0);
168  website = new char[textlen + 1];
169  if (X509_NAME_get_text_by_NID(certname, NID_commonName, website, textlen + 1) == textlen && textlen > 0) {
170  merchant = website;
171  } else {
172  throw SSLVerifyError("Bad certificate, missing common name.");
173  }
174  // TODO: detect EV certificates and set merchant = business name instead of unfriendly NID_commonName ?
175  } catch (SSLVerifyError& err) {
176  fResult = false;
177  qWarning() << "PaymentRequestPlus::getMerchant : SSL error: " << err.what();
178  }
179 
180  if (website)
181  delete[] website;
182  X509_STORE_CTX_free(store_ctx);
183  for (unsigned int i = 0; i < certs.size(); i++)
184  X509_free(certs[i]);
185 
186  return fResult;
187 }
